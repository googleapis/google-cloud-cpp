if (("${CMAKE_CXX_COMPILER_ID}" STREQUAL "GNU")
    AND (CMAKE_CXX_COMPILER_VERSION VERSION_EQUAL 7.5.0)
    AND (Protobuf_VERSION VERSION_GREATER_EQUAL 23.0)
    AND (Protobuf_VERSION VERSION_LESS 24.4))
    message(
        WARNING "Cannot build google/cloud/dialogflow_es - "
                " the code generated by Protobuf between (at least)"
                " 23.0 and 24.3 triggers an internal compiler error"
                " with GCC == 7.5.0")
    return()
endif ()

include(GoogleCloudCppLibrary)

google_cloud_cpp_add_gapic_library(
    dialogflow_es "Dialogflow API"
    SERVICE_DIRS "__EMPTY__"
    BACKWARDS_COMPAT_PROTO_TARGETS "cloud_dialogflow_v2_protos")

if (BUILD_TESTING AND GOOGLE_CLOUD_CPP_ENABLE_CXX_EXCEPTIONS)
    add_executable(dialogflow_es_quickstart "quickstart/quickstart.cc")
    target_link_libraries(dialogflow_es_quickstart
                          PRIVATE google-cloud-cpp::dialogflow_es)
    google_cloud_cpp_add_common_options(dialogflow_es_quickstart)
    add_test(
        NAME dialogflow_es_quickstart
        COMMAND cmake -P "${PROJECT_SOURCE_DIR}/cmake/quickstart-runner.cmake"
                $<TARGET_FILE:dialogflow_es_quickstart> GOOGLE_CLOUD_PROJECT)
    set_tests_properties(dialogflow_es_quickstart
                         PROPERTIES LABELS "integration-test;quickstart")
endif ()
