cmake_minimum_required(VERSION 3.6)
project(verify-cxx-version-capture CXX)

if (NOT CMAKE_CXX_STANDARD)
    set(CMAKE_CXX_STANDARD 17)
endif ()
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(google_cloud_cpp_common REQUIRED)
find_package(GTest CONFIG REQUIRED)

include(CTest)

add_executable(capture_versions_test capture_versions_test.cc)
target_compile_definitions(capture_versions_test
                           PRIVATE GOOGLE_CLOUD_CPP_TEST_DIFFERENT_CPP_VERSIONS)
target_link_libraries(
    capture_versions_test PRIVATE google-cloud-cpp::common GTest::gmock_main
                                  GTest::gmock GTest::gtest)
add_test(NAME "capture_versions_test" COMMAND "capture_versions_test")
