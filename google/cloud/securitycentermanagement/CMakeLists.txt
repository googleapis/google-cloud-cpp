include(GoogleCloudCppLibrary)

google_cloud_cpp_add_gapic_library(
    securitycentermanagement "Security Center Management API"
    SERVICE_DIRS "v1/")

if (BUILD_TESTING AND GOOGLE_CLOUD_CPP_ENABLE_CXX_EXCEPTIONS)
    add_executable(securitycentermanagement_quickstart
                   "quickstart/quickstart.cc")
    target_link_libraries(securitycentermanagement_quickstart
                          PRIVATE google-cloud-cpp::securitycentermanagement)
    google_cloud_cpp_add_common_options(securitycentermanagement_quickstart)
    add_test(
        NAME securitycentermanagement_quickstart
        COMMAND
            cmake -P "${PROJECT_SOURCE_DIR}/cmake/quickstart-runner.cmake"
            $<TARGET_FILE:securitycentermanagement_quickstart>
            GOOGLE_CLOUD_PROJECT)
    set_tests_properties(securitycentermanagement_quickstart
                         PROPERTIES LABELS "integration-test;quickstart")
endif ()
