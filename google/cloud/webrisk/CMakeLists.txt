if (("${CMAKE_CXX_COMPILER_ID}" STREQUAL "GNU")
    AND (CMAKE_CXX_COMPILER_VERSION VERSION_EQUAL 7.5.0)
    AND (Protobuf_VERSION VERSION_GREATER_EQUAL 23.0)
    AND (Protobuf_VERSION VERSION_LESS 24.4))
    message(
        WARNING "Cannot build google/cloud/webrisk - "
                " the code generated by Protobuf between (at least)"
                " 23.0 and 24.3 triggers an internal compiler error"
                " with GCC == 7.5.0")
    return()
endif ()

include(GoogleCloudCppLibrary)

google_cloud_cpp_add_gapic_library(webrisk "Web Risk API"
                                   SERVICE_DIRS "__EMPTY__" "v1/")

if (BUILD_TESTING AND GOOGLE_CLOUD_CPP_ENABLE_CXX_EXCEPTIONS)
    add_executable(webrisk_quickstart "quickstart/quickstart.cc")
    target_link_libraries(webrisk_quickstart PRIVATE google-cloud-cpp::webrisk)
    google_cloud_cpp_add_common_options(webrisk_quickstart)
    add_test(
        NAME webrisk_quickstart
        COMMAND cmake -P "${PROJECT_SOURCE_DIR}/cmake/quickstart-runner.cmake"
                $<TARGET_FILE:webrisk_quickstart>)
    set_tests_properties(
        webrisk_quickstart
        PROPERTIES
            LABELS "integration-test;quickstart" ENVIRONMENT
            GOOGLE_CLOUD_CPP_USER_PROJECT=$ENV{GOOGLE_CLOUD_CPP_USER_PROJECT})
endif ()
