26 /**
27  * Represent a (possibly non-continuous) set of row keys.
28  *
29  * Cloud Bigtable can scan non-continuous sets of rows, these sets can include
30  * a mix of specific row keys and ranges as defined by `bigtable::RowRange`.
31  */
32 class RowSet {
33  public:
34  /// Create an empty set.
35  RowSet() {}
36 
37  RowSet(RowSet&& rhs) noexcept = default;
38  RowSet& operator=(RowSet&& rhs) noexcept = default;
39  RowSet(RowSet const& rhs) = default;
40  RowSet& operator=(RowSet const& rhs) = default;
41 
42  template <typename... Arg>
43  RowSet(Arg&&... a) {
44  // Generate a better error message when the parameters do not match.
45  static_assert(internal::conjunction<IsValidAppendAllArg<Arg>...>::value,
46  "RowSet variadic constructor arguments must be convertible "
47  "to bigtable::RowRange or std::string");
48  AppendAll(std::forward<Arg&&>(a)...);
49  }
50 
51  /// Add @p range to the set.
52  void Append(RowRange range) {
53  *row_set_.add_row_ranges() = std::move(range).as_proto();
54  }
55 
56  /**
57  * Add @p row_key to the set, minimize copies when possible.
58  */
59  void Append(std::string row_key) {
60  *row_set_.add_row_keys() = std::move(row_key);
61  }
62 
63  /**
64  * Modify this object to contain the ranges and keys inside @p range.
65  *
66  * This function removes any rowkeys outside @p range, it removes any row
67  * ranges that do not insersect with @p range, and keeps only the intersection
68  * for those ranges that do intersect @p range.
69  */
70  RowSet Intersect(bigtable::RowRange const& range) const;
71 
72  /**
73  * Returns true if the set is empty.
74  *
75  * A row set is empty iff passing it to a ReadRows call would never
76  * cause it to return rows. This is true if the set consists of only
77  * empty ranges.
78  *
79  * Note that a default constructed RowSet is not empty, since it
80  * matches all rows in the table.
81  */
82  bool IsEmpty() const;
83 
84  ::google::bigtable::v2::RowSet const& as_proto() const& { return row_set_; }
85  ::google::bigtable::v2::RowSet&& as_proto() && { return std::move(row_set_); }
86 
87  private:
88  template <typename T>
89  struct IsValidAppendAllArg {
90  using value_type = T;
91  using type =
92  std::integral_constant<bool,
93  std::is_convertible<T, std::string>::value ||
94  std::is_convertible<T, RowRange>::value>;
95  static constexpr bool value = type::value;
96  };
97 
98  /// Append the arguments to the rowset.
99  template <typename H, typename... Tail>
100  void AppendAll(H&& head, Tail&&... a) {
101  // We cannot use the initializer list expression here because the types
102  // may be all different.
103  Append(std::forward<H>(head));
104  AppendAll(std::forward<Tail&&>(a)...);
105  }
106 
107  /// Terminate the recursion.
108  void AppendAll() {}
109 
110  private:
111  ::google::bigtable::v2::RowSet row_set_;
112 };
