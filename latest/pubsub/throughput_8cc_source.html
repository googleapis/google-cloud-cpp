180  std::chrono::steady_clock::time_point start) {
181  auto const now = std::chrono::steady_clock::now();
182  if (now >= start + config.maximum_runtime) return true;
183  if (samples >= config.maximum_samples) return true;
184  if (now < start + config.minimum_runtime) return false;
185  return samples >= config.minimum_samples;

243 class PublishWorker {
244  public:
245  PublishWorker(Config config, int id) : config_(std::move(config)), id_(id) {}
246 
247  void Shutdown() {
248  std::unique_lock<std::mutex> lk(mu_);
249  shutdown_ = true;
250  cv_.notify_all();
251  }
252 
253  void Run() {
254  auto publisher = CreatePublisher(config_);
255 
256  auto gen = google::cloud::internal::DefaultPRNG(std::random_device{}());
257  auto const data = google::cloud::internal::Sample(
258  gen, static_cast<int>(config_.payload_size), "0123456789");
259 
260  using std::chrono::duration_cast;
261  using std::chrono::steady_clock;
262 
263  // We typically want to send tens of thousands or a million messages
264  // per second, but sleeping for just one microsecond (or less) does not
265  // work, the sleep call takes about 100us. We pace every kPacingCount
266  // messages instead.
267  auto constexpr kPacingCount = 8192;
268  auto const enable_pacing =
269  config_.publisher_target_messages_per_second != 0;
270  auto const pacing_period = [&] {
271  using std::chrono::microseconds;
272  if (config_.publisher_target_messages_per_second == 0) {
273  return microseconds(0);
274  }
275  return microseconds(std::chrono::seconds(1)) * kPacingCount /
276  config_.publisher_target_messages_per_second;
277  }();
278 
279  auto const start = std::chrono::steady_clock::now();
280  auto pacing_time = start + pacing_period;
281  for (std::int64_t i = 0; NotShutdownAndReady(); ++i) {
282  auto const elapsed =
283  duration_cast<std::chrono::microseconds>(steady_clock::now() - start);
284  auto message = pubsub::MessageBuilder{}
285  .SetAttributes({
286  {"sendTime", std::to_string(elapsed.count())},
287  {"clientId", std::to_string(id_)},
288  {"sequenceNumber", std::to_string(i)},
289  })
290  .SetData(data)
291  .Build();
292  auto const bytes = MessageSize(message);
293  publisher.Publish(std::move(message))
294  .then([this, bytes](future<StatusOr<std::string>> f) {
295  ++ack_count;
296  ack_bytes.fetch_add(bytes);
297  if (!f.get()) ++error_count;
298  OnAck();
299  });
300  ++send_count;
301  send_bytes.fetch_add(bytes);
302  if (enable_pacing && (i + 1) % kPacingCount == 0) {
303  auto const now = steady_clock::now();
304  if (now < pacing_time) std::this_thread::sleep_for(pacing_time - now);
305  pacing_time = now + pacing_period;
306  }
307  }
308  WaitUntilAllAcked();
309  }
310 
311  std::int64_t hwm_count() const { return hwm_count_; }
312  std::int64_t lwm_count() const { return lwm_count_; }
313 
314  private:
315  bool NotShutdownAndReady() {
316  std::unique_lock<std::mutex> lk(mu_);
317  cv_.wait(lk, [&] { return !blocked_ || shutdown_; });
318  if (shutdown_) return false;
319  pending_ += config_.payload_size;
320  if (pending_ <= config_.publisher_pending_hwm) return true;
321  blocked_ = true;
322  ++hwm_count_;
323  return true;
324  }
325 
326  void WaitUntilAllAcked() {
327  std::unique_lock<std::mutex> lk(mu_);
328  cv_.wait(lk, [&] { return pending_ == 0; });
329  }
330 
331  void OnAck() {
332  std::unique_lock<std::mutex> lk(mu_);
333  pending_ -= config_.payload_size;
334  if (pending_ == 0) {
335  blocked_ = false;
336  cv_.notify_all();
337  return;
338  }
339  if (pending_ > config_.publisher_pending_lwm) return;
340  if (!blocked_) return;
341  ++lwm_count_;
342  blocked_ = false;
343  cv_.notify_all();
344  }
345 
346  Config const config_;
347  int const id_;
348  std::mutex mu_;
349  std::condition_variable cv_;
350  bool shutdown_ = false;
351  bool blocked_ = false;
352  std::int64_t pending_ = 0;
353  std::int64_t lwm_count_ = 0;
354  std::int64_t hwm_count_ = 0;
355 };

357 void PublisherTask(Config const& config) {
358  std::vector<std::shared_ptr<PublishWorker>> workers;
359  int task_id = 0;
360  std::generate_n(
361  std::back_inserter(workers), config.publisher_thread_count,
362  [&] { return std::make_shared<PublishWorker>(config, task_id++); });
363  std::vector<std::thread> tasks(workers.size());
364  auto activate = [](std::shared_ptr<PublishWorker> const& w) {
365  return std::thread{[w] { w->Run(); }};
366  };
367  std::transform(workers.begin(), workers.end(), tasks.begin(), activate);
368 
369  auto const start = std::chrono::steady_clock::now();
370  for (int i = 0; !Done(config, i, start); ++i) {
371  using std::chrono::steady_clock;
372  Timer usage;
373  usage.Start();
374  auto const start_send_count = send_count.load();
375  auto const start_send_bytes = send_bytes.load();
376  auto const start_ack_count = ack_count.load();
377  auto const start_ack_bytes = ack_bytes.load();
378  std::this_thread::sleep_for(config.iteration_duration);
379  auto const send_count_last = send_count.load() - start_send_count;
380  auto const send_bytes_last = send_bytes.load() - start_send_bytes;
381  auto const ack_count_last = ack_count.load() - start_ack_count;
382  auto const ack_bytes_last = ack_bytes.load() - start_ack_bytes;
383  usage.Stop();
384  PrintResult("Pub", i, send_count_last, send_bytes_last, usage);
385  PrintResult("Ack", i, ack_count_last, ack_bytes_last, usage);
386  }
387 
388  for (auto& w : workers) w->Shutdown();
389  for (auto& t : tasks) t.join();
390  auto const hwm_count = std::accumulate(
391  workers.begin(), workers.end(), std::int64_t{0},
392  [](std::int64_t a, std::shared_ptr<PublishWorker> const& w) {
393  return a + w->hwm_count();
394  });
395  auto const lwm_count = std::accumulate(
396  workers.begin(), workers.end(), std::int64_t{0},
397  [](std::int64_t a, std::shared_ptr<PublishWorker> const& w) {
398  return a + w->lwm_count();
399  });
400  std::cout << "# Publisher: error_count=" << error_count
401  << ", ack_count=" << ack_count << ", send_count=" << send_count
402  << ", hwm_count=" << hwm_count << ", lwm_count=" << lwm_count
403  << std::endl;
404 }

431 void SubscriberTask(Config const& config) {
432  std::vector<pubsub::Subscriber> subscribers;
433  std::generate_n(std::back_inserter(subscribers),
434  config.subscriber_thread_count,
435  [config] { return CreateSubscriber(config); });
436 
437  std::atomic<std::int64_t> received_count{0};
438  std::atomic<std::int64_t> received_bytes{0};
439  auto handler = [&received_count, &received_bytes](pubsub::Message const& m,
440  pubsub::AckHandler h) {
441  ++received_count;
442  received_bytes.fetch_add(MessageSize(m));
443  std::move(h).ack();
444  };
445 
446  std::vector<future<Status>> sessions;
447  std::transform(
448  subscribers.begin(), subscribers.end(), std::back_inserter(sessions),
449  [&handler](pubsub::Subscriber s) { return s.Subscribe(handler); });
450 
451  auto const start = std::chrono::steady_clock::now();
452  for (int i = 0; !Done(config, i, start); ++i) {
453  using std::chrono::steady_clock;
454  Timer usage;
455  usage.Start();
456  auto const start_count = received_count.load();
457  auto const start_bytes = received_bytes.load();
458  std::this_thread::sleep_for(config.iteration_duration);
459  auto const count = received_count.load() - start_count;
460  auto const bytes = received_bytes.load() - start_bytes;
461  usage.Stop();
462  PrintResult("Sub", i, count, bytes, usage);
463  }
464  for (auto& s : sessions) s.cancel();
465  Status last_status;
466  std::int64_t last_received_count = 0;
467  for (auto& s : sessions) {
468  auto status = s.get();
469  auto const current = received_count.load();
470  if (last_status == status && last_received_count == current) continue;
471  last_status = std::move(status);
472  last_received_count = current;
473  std::lock_guard<std::mutex> lk(cout_mu);
474  std::cout << "# status=" << last_status << ", count=" << last_received_count
475  << std::endl;
476  }
477 }
