68 void SubscriptionMessageQueue::OnRead(
69  std::unique_lock<std::mutex> lk,
70  google::pubsub::v1::StreamingPullResponse r) {
71  auto handle_response = [&] {
72  shutdown_manager_->FinishedOperation("OnRead");
73  for (auto& m : *r.mutable_received_messages()) {
74  auto key = m.message().ordering_key();
75  if (key.empty()) {
76  // Empty key, requires no ordering and therefore immediately runnable.
77  runnable_messages_.push_back(std::move(m));
78  continue;
79  }
80  // The message requires ordering, find out if there is an existing queue
81  // for its ordering key, and insert one if necessary.
82  auto loc = queues_.insert({key, {}});
83  // There is no queue for this ordering key, that means no other messages
84  // are present, we can push the message to the runnable queue. We leave
85  // the per-ordering-key queue as a marker for any other incoming messages
86  // with the same ordering key.
87  if (loc.second) {
88  runnable_messages_.push_back(std::move(m));
89  continue;
90  }
91  // Insert the messages into the existing queue.
92  loc.first->second.push_back(std::move(m));
93  }
94  DrainQueue(std::move(lk));
95  };
96  auto bulk_nack = [&] {
97  lk.unlock();
98  std::vector<std::string> ack_ids(r.mutable_received_messages()->size());
99  std::transform(r.mutable_received_messages()->begin(),
100  r.mutable_received_messages()->end(), ack_ids.begin(),
101  [&](google::pubsub::v1::ReceivedMessage& m) {
102  return std::move(*m.mutable_ack_id());
103  });
104  (void)source_->BulkNack(std::move(ack_ids));
105  };
106  if (!shutdown_manager_->StartOperation(__func__, "OnRead", handle_response)) {
107  bulk_nack();
108  }
109 }

111 void SubscriptionMessageQueue::Shutdown(std::unique_lock<std::mutex> lk) {
112  shutdown_ = true;
113  available_slots_ = 0;
114  QueueByOrderingKey queues;
115  queues.swap(queues_);
116  std::deque<google::pubsub::v1::ReceivedMessage> runnable_messages;
117  runnable_messages.swap(runnable_messages_);
118  lk.unlock();
119 
120  std::vector<std::string> ack_ids;
121  for (auto& kv : queues) {
122  for (auto& m : kv.second) {
123  ack_ids.push_back(std::move(*m.mutable_ack_id()));
124  }
125  }
126  for (auto& m : runnable_messages) {
127  ack_ids.push_back(std::move(*m.mutable_ack_id()));
128  }
129 
130  if (ack_ids.empty()) return;
131  source_->BulkNack(std::move(ack_ids));
132 }

134 void SubscriptionMessageQueue::DrainQueue(std::unique_lock<std::mutex> lk) {
135  while (!runnable_messages_.empty() && available_slots_ > 0 && !shutdown_) {
136  auto m = std::move(runnable_messages_.front());
137  runnable_messages_.pop_front();
138  --available_slots_;
139  // No need to track messages without an ordering key, as there is no action
140  // to take in their HandlerDone() member function.
141  if (!m.message().ordering_key().empty()) {
142  ordering_key_by_ack_id_[m.ack_id()] = m.message().ordering_key();
143  }
144  // Don't hold a lock during the callback, as the callee may call `Read()`
145  // or something similar.
146  lk.unlock();
147  callback_(std::move(m));
148  lk.lock();
149  }
150 }

152 void SubscriptionMessageQueue::HandlerDone(std::string const& ack_id) {
153  std::unique_lock<std::mutex> lk(mu_);
154  // Find out the ordering key for this message.
155  auto loc = ordering_key_by_ack_id_.find(ack_id);
156  // Messages without an ordering key are not inserted in the collection (see
157  // `DrainQueue()`), so this happens routinely.
158  if (loc == ordering_key_by_ack_id_.end()) return;
159  auto key = std::move(loc->second);
160  ordering_key_by_ack_id_.erase(loc);
161  auto ql = queues_.find(key);
162  // This is purely defensive, but should not happen.
163  if (ql == queues_.end()) return;
164  if (ql->second.empty()) {
165  // There are no more messages for this ordering key, remove the queue, as it
166  // also serves as a marker to order the next message.
167  queues_.erase(ql);
168  return;
169  }
170  auto m = std::move(ql->second.front());
171  ql->second.pop_front();
172  runnable_messages_.push_back(std::move(m));
173  DrainQueue(std::move(lk));
174 }
