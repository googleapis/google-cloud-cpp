50 bool SessionShutdownManager::FinishedOperation(char const* name) {
51  std::unique_lock<std::mutex> lk(mu_);
52  auto decrease_count = [&] { return --ops_[name]; };
53  GCP_LOG(TRACE) << "operation <" << name << "> finished"

56  << ", result=" << result_ << ", count=" << decrease_count();
57  bool r = shutdown_;

64  std::unique_lock<std::mutex> lk(mu_);

74 void SessionShutdownManager::SignalOnShutdown(std::unique_lock<std::mutex> lk) {
75  GCP_LOG(TRACE) << __func__ << "() - maybe signal"
76  << ", shutdown=" << shutdown_ << ", signaled=" << signaled_
77  << ", outstanding_operations=" << outstanding_operations_
78  << ", result=" << result_ << ", ops=" << FormatOps(ops_);
79  if (outstanding_operations_ > 0 || !shutdown_ || signaled_) return;
80  // No other thread will go beyond this point, as `signaled_` is only set
81  // once.
82  signaled_ = true;
83  // As satisfying the `done_` promise might trigger callbacks we should release
84  // the lock before doing so. But we also need to modify any variables with
85  // the lock held:
86  auto p = std::move(done_);
87  auto s = std::move(result_);
88  lk.unlock();
89  p.set_value(std::move(s));
90 }
