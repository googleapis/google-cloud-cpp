33 TEST(SubscriptionSessionTest, ScheduleCallbacks) {
34  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
35  pubsub::Subscription const subscription("test-project", "test-subscription");
36 
37  std::mutex mu;
38  int count = 0;
39  EXPECT_CALL(*mock, AsyncPull(_, _, _))
40  .Times(AtLeast(1))
41  .WillRepeatedly([&](google::cloud::CompletionQueue&,
42  std::unique_ptr<grpc::ClientContext>,
43  google::pubsub::v1::PullRequest const& request) {
44  EXPECT_EQ(subscription.FullName(), request.subscription());
45  google::pubsub::v1::PullResponse response;
46  for (int i = 0; i != 2; ++i) {
47  auto& m = *response.add_received_messages();
48  std::lock_guard<std::mutex> lk(mu);
49  m.set_ack_id("test-ack-id-" + std::to_string(count));
50  m.set_delivery_attempt(42);
51  m.mutable_message()->set_message_id("test-message-id-" +
52  std::to_string(count));
53  ++count;
54  }
55  return make_ready_future(make_status_or(response));
56  });
57 
58  std::atomic<int> expected_ack_id{0};
59  EXPECT_CALL(*mock, AsyncAcknowledge(_, _, _))
60  .Times(AtLeast(1))
61  .WillRepeatedly(
62  [&](google::cloud::CompletionQueue&,
63  std::unique_ptr<grpc::ClientContext>,
64  google::pubsub::v1::AcknowledgeRequest const& request) {
65  EXPECT_EQ(subscription.FullName(), request.subscription());
66  for (auto const& a : request.ack_ids()) {
67  EXPECT_EQ("test-ack-id-" + std::to_string(expected_ack_id), a);
68  ++expected_ack_id;
69  }
70  return make_ready_future(Status{});
71  });
72  EXPECT_CALL(*mock, AsyncModifyAckDeadline(_, _, _))
73  .WillRepeatedly([](google::cloud::CompletionQueue&,
74  std::unique_ptr<grpc::ClientContext>,
75  google::pubsub::v1::ModifyAckDeadlineRequest const&) {
76  return make_ready_future(Status{});
77  });
78 
79  google::cloud::CompletionQueue cq;
80  std::vector<std::thread> tasks;
81  std::generate_n(std::back_inserter(tasks), 4,
82  [&] { return std::thread([&cq] { cq.Run(); }); });
83  std::set<std::thread::id> ids;
84  auto const main_id = std::this_thread::get_id();
85  std::transform(tasks.begin(), tasks.end(), std::inserter(ids, ids.end()),
86  [](std::thread const& t) { return t.get_id(); });
87 
88  std::atomic<int> expected_message_id{0};
89  auto handler = [&](pubsub::Message const& m, pubsub::AckHandler h) {
90  EXPECT_EQ(42, h.delivery_attempt());
91  EXPECT_EQ("test-message-id-" + std::to_string(expected_message_id),
92  m.message_id());
93  auto pos = ids.find(std::this_thread::get_id());
94  EXPECT_NE(ids.end(), pos);
95  EXPECT_NE(main_id, std::this_thread::get_id());
96  // Increment the counter before acking, as the ack() may trigger a new call
97  // before this function gets to run.
98  ++expected_message_id;
99  std::move(h).ack();
100  };
101 
102  auto session = SubscriptionSession::Create(
103  mock, cq, {subscription.FullName(), handler, {}});
104  auto response = session->Start();
105  while (expected_ack_id.load() < 100) {
106  auto s = response.wait_for(std::chrono::milliseconds(5));
107  if (s != std::future_status::timeout) break;
108  }
109  response.cancel();
110  EXPECT_STATUS_OK(response.get());
111 
112  cq.Shutdown();
113  for (auto& t : tasks) t.join();
114 }

206 TEST(SubscriptionSessionTest, UpdateAckDeadlines) {
207  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
208  pubsub::Subscription const subscription("test-project", "test-subscription");
209 
210  std::mutex mu;
211  int count = 0;
212  auto generate_3 = [&](google::cloud::CompletionQueue&,
213  std::unique_ptr<grpc::ClientContext>,
214  google::pubsub::v1::PullRequest const& request) {
215  EXPECT_EQ(subscription.FullName(), request.subscription());
216  google::pubsub::v1::PullResponse response;
217  for (int i = 0; i != 3; ++i) {
218  auto& m = *response.add_received_messages();
219  std::lock_guard<std::mutex> lk(mu);
220  m.set_ack_id("test-ack-id-" + std::to_string(count));
221  m.mutable_message()->set_message_id("test-message-id-" +
222  std::to_string(count));
223  ++count;
224  }
225  return make_ready_future(make_status_or(response));
226  };
227  auto generate_ack_response =
228  [&](google::cloud::CompletionQueue&, std::unique_ptr<grpc::ClientContext>,
229  google::pubsub::v1::AcknowledgeRequest const&) {
230  return make_ready_future(Status{});
231  };
232  // The basic expectations, pull some data, ack each message, pull more data,
233  // must be in a specific order.
234  {
235  InSequence sequence;
236  EXPECT_CALL(*mock, AsyncPull(_, _, _)).WillOnce(generate_3);
237  EXPECT_CALL(*mock, AsyncAcknowledge(_, _, _))
238  .Times(3)
239  .WillRepeatedly(generate_ack_response);
240  EXPECT_CALL(*mock, AsyncPull(_, _, _)).WillOnce(generate_3);
241  EXPECT_CALL(*mock, AsyncAcknowledge(_, _, _))
242  .Times(3)
243  .WillRepeatedly(generate_ack_response);
244  EXPECT_CALL(*mock, AsyncPull(_, _, _)).WillOnce(generate_3);
245  EXPECT_CALL(*mock, AsyncAcknowledge(_, _, _))
246  .Times(3)
247  .WillRepeatedly(generate_ack_response);
248  }
249 
250  // The expectations for timers and AsyncModifyAckDeadline are more complex.
251  // The setup is as follows:
252  // - The subscription handler stops after receiving specific messages,
253  // e.g. test-ack-id-2.
254  // - Because it is stopped, the timer to update the deadlines will eventually
255  // expire, and (again eventually) will contain only the messages that have
256  // not been acked.
257  // - When that happens we signal the handler to continue.
258  //
259  // The set of points when to stop is expressed in this array of "tripwires":
260  struct Tripwire {
261  // When is this triggered
262  std::string const ack_id;
263  // How many times it needs to repeat before the barrier is lifted
264  int repeats;
265  // The barrier to wake up the waiting thread
266  promise<void> barrier;
267  } tripwires[] = {
268  {"test-ack-id-2", 1, {}},
269  {"test-ack-id-5", 2, {}},
270  };
271  std::mutex tripwires_mu;
272 
273  // This is how we mock all the AsyncModifyAckDeadline() calls, signaling the
274  // handler at the right points.
275  EXPECT_CALL(*mock, AsyncModifyAckDeadline(_, _, _))
276  .WillRepeatedly(
277  [&](google::cloud::CompletionQueue&,
278  std::unique_ptr<grpc::ClientContext>,
279  google::pubsub::v1::ModifyAckDeadlineRequest const& r) {
280  auto ids_are = [&r](std::string const& s) {
281  return r.ack_ids_size() == 1 && r.ack_ids(0) == s;
282  };
283  std::lock_guard<std::mutex> lk(tripwires_mu);
284  for (auto& tw : tripwires) {
285  if (!ids_are(tw.ack_id)) continue;
286  if (--tw.repeats == 0) tw.barrier.set_value();
287  }
288  return make_ready_future(Status{});
289  });
290 
291  // Now unto the handler, basically it
292  promise<void> enough_messages;
293  std::atomic<int> message_count{0};
294  auto constexpr kMaximumMessages = 9;
295  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
296  for (auto& tw : tripwires) {
297  // Note the lack of locking, this is Okay because the values we use either
298  // never changes (`ack_id` is const) or they are thread-safe by design:
299  // (barrier is a promise<void> which better be thread-safe.
300  if (tw.ack_id == h.ack_id()) tw.barrier.get_future().get();
301  }
302  // When enough messages are received signal the main thread to stop the
303  // test.
304  if (++message_count == kMaximumMessages) enough_messages.set_value();
305  std::move(h).ack();
306  };
307 
308  google::cloud::CompletionQueue cq;
309  std::vector<std::thread> pool;
310  // we need more than one thread because `handler()` blocks and the CQ needs to
311  // make progress.
312  std::generate_n(std::back_inserter(pool), 4,
313  [&cq] { return std::thread{[&cq] { cq.Run(); }}; });
314 
315  auto session = SubscriptionSession::Create(
316  mock, cq,
317  {subscription.FullName(), handler,
318  pubsub::SubscriptionOptions{}.set_max_deadline_time(
319  std::chrono::seconds(60))});
320  session->SetTestRefreshPeriod(std::chrono::milliseconds(50));
321  auto response = session->Start();
322  enough_messages.get_future()
323  .then([&](future<void>) { response.cancel(); })
324  .get();
325  EXPECT_STATUS_OK(response.get());
326 
327  cq.Shutdown();
328  for (auto& t : pool) t.join();
329 }
