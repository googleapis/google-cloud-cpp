277  std::chrono::duration<Rep, Period> d) {
278  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(d);
279  auto nanos =
280  std::chrono::duration_cast<std::chrono::nanoseconds>(d - seconds);
281  google::protobuf::Duration result;

283  result.set_nanos(static_cast<std::int32_t>(nanos.count()));
