28 class SubscriptionSessionImpl
29  : public std::enable_shared_from_this<SubscriptionSessionImpl> {
30  public:
31  static future<Status> Create(
32  pubsub::SubscriberOptions const& options,
33  google::cloud::CompletionQueue executor,
34  std::shared_ptr<SessionShutdownManager> shutdown_manager,
35  std::shared_ptr<SubscriptionBatchSource> source,
36  pubsub::SubscriberConnection::SubscribeParams p) {
37  auto queue =
38  SubscriptionMessageQueue::Create(shutdown_manager, std::move(source));
39  auto concurrency_control = SubscriptionConcurrencyControl::Create(
40  executor, shutdown_manager, std::move(queue),
41  options.max_concurrency());
42 
43  auto self = std::make_shared<SubscriptionSessionImpl>(
44  std::move(executor), std::move(shutdown_manager),
45  std::move(concurrency_control), options.shutdown_polling_period());
46 
47  auto weak = std::weak_ptr<SubscriptionSessionImpl>(self);
48  auto result = self->shutdown_manager_->Start(promise<Status>([weak] {
49  if (auto self = weak.lock()) self->InitiateApplicationShutdown();
50  }));
51  // Create a (periodic) timer, this serves two purposes:
52  // 1) Effectively the timer owns "self" and the timer is owned by the
53  // CompletionQueue.
54  // 2) When the completion queue is shutdown, the timer is canceled and
55  // `self` gets a chance to shutdown the pipeline.
56  self->ScheduleTimer();
57  self->pipeline_->Start(std::move(p.callback));
58  return result.then([weak](future<Status> f) {
59  if (auto self = weak.lock()) self->ShutdownCompleted();
60  return f.get();
61  });
62  }

82  void MarkAsShutdown(std::unique_lock<std::mutex> lk, ShutdownState reason,
83  char const* where) {
84  auto pipeline = pipeline_;
85  auto shutdown_manager = shutdown_manager_;
86  if (reason == kShutdownByCompletionQueue) pipeline_.reset();
87  shutdown_state_ = reason;
88  lk.unlock();
89  shutdown_manager->MarkAsShutdown(where, {});
90  if (pipeline) pipeline->Shutdown();
91  }

93  void InitiateApplicationShutdown() {
94  GCP_LOG(TRACE) << __func__ << "()";
95  std::unique_lock<std::mutex> lk(mu_);
96  switch (shutdown_state_) {
97  case kNotInShutdown:
98  MarkAsShutdown(std::move(lk), kShutdownByApplication, __func__);
99  break;
100  case kShutdownByApplication:
101  case kShutdownByCompletionQueue:
102  case kShutdownCompleted:
103  break;
104  }
105  }
106 
107  void ShutdownCompleted() {
108  ShutdownCompleted(std::unique_lock<std::mutex>(mu_));
109  }
110  void ShutdownCompleted(std::unique_lock<std::mutex>) {
111  GCP_LOG(TRACE) << __func__ << "()";
112  pipeline_.reset();
113  shutdown_state_ = kShutdownCompleted;
114  if (timer_.valid()) timer_.cancel();
115  }
116 
117  void ScheduleTimer() {
118  GCP_LOG(TRACE) << __func__ << "()";
119  using TimerArg = future<StatusOr<std::chrono::system_clock::time_point>>;
120 
121  auto self = shared_from_this();
122  auto t = cq_.MakeRelativeTimer(shutdown_polling_period_)
123  .then([self](TimerArg f) { self->OnTimer(!f.get().ok()); });
124  std::unique_lock<std::mutex> lk(mu_);
125  if (shutdown_state_ == kShutdownByCompletionQueue) return;
126  timer_ = std::move(t);
127  }
128 
129  void OnTimer(bool cancelled) {
130  GCP_LOG(TRACE) << __func__ << "(" << cancelled << ")";
131  if (!cancelled) {
132  ScheduleTimer();
133  return;
134  }
135  std::unique_lock<std::mutex> lk(mu_);
136  switch (shutdown_state_) {
137  case kNotInShutdown:
138  MarkAsShutdown(std::move(lk), kShutdownByCompletionQueue, __func__);
139  break;
140  case kShutdownByApplication:
141  ShutdownCompleted(std::move(lk));
142  break;
143  case kShutdownByCompletionQueue:
144  case kShutdownCompleted:
145  break;
146  }
147  }

160 future<Status> CreateSubscriptionSession(
161  pubsub::Subscription const& subscription,
162  pubsub::SubscriberOptions const& options,
163  std::shared_ptr<pubsub_internal::SubscriberStub> const& stub,
164  google::cloud::CompletionQueue const& executor, std::string client_id,
165  pubsub::SubscriberConnection::SubscribeParams p,
166  std::unique_ptr<pubsub::RetryPolicy const> retry_policy,
167  std::unique_ptr<pubsub::BackoffPolicy const> backoff_policy) {
168  auto shutdown_manager = std::make_shared<SessionShutdownManager>();
169  auto batch = std::make_shared<StreamingSubscriptionBatchSource>(
170  executor, shutdown_manager, stub, subscription.FullName(),
171  std::move(client_id), options, std::move(retry_policy),
172  std::move(backoff_policy));
173  auto lease_management = SubscriptionLeaseManagement::Create(
174  executor, shutdown_manager, std::move(batch),
175  options.max_deadline_time());
176 
177  return SubscriptionSessionImpl::Create(
178  options, std::move(executor), std::move(shutdown_manager),
179  std::move(lease_management), std::move(p));
180 }

200  auto shutdown_manager = std::make_shared<SessionShutdownManager>();
201  auto batch = std::make_shared<StreamingSubscriptionBatchSource>(

203  "test-client-id", options, std::move(retry_policy),
204  std::move(backoff_policy));
205 
206  auto cq = executor; // need a copy to make it mutable
207  auto timer = [cq](std::chrono::system_clock::time_point) mutable {
208  return cq.MakeRelativeTimer(std::chrono::milliseconds(50))
209  .then([](future<StatusOr<std::chrono::system_clock::time_point>> f) {
210  return f.get().status();
211  });
212  };
213  auto lease_management = SubscriptionLeaseManagement::CreateForTesting(
214  executor, shutdown_manager, timer, std::move(batch),
215  options.max_deadline_time());

218  options, std::move(executor), std::move(shutdown_manager),
219  std::move(lease_management), std::move(p));
