73 void StreamingSubscriptionBatchSource::StartStream(
74  std::shared_ptr<pubsub::RetryPolicy> retry_policy,
75  std::shared_ptr<pubsub::BackoffPolicy> backoff_policy) {
76  // Starting a stream is a 4 step process.
77  // 1. Create a new SubscriberStub::AsyncPullStream object.
78  // 2. Call Start() on it, which is asynchronous, which might fail, but rarely
79  // does
80  // 3. Call Write() on it, which is also asynchronous and almost always
81  // succeeds, could fail if the endpoint is invalid
82  // 4. Call Read() on it, which is also asynchronous, and it is the first
83  // chance to get errors such as kPermissionDenied, or kNotFound.
84  // Once Read() completes we can repeatedly call Read() and/or Write().
85  // Because steps 2 through 4 may fail with transient errors we need to wrap
86  // these steps in an asynchronous retry loop.
87 
88  auto request = InitialRequest();
89  auto stream = stub_->AsyncStreamingPull(
90  cq_, absl::make_unique<grpc::ClientContext>(), request);
91  if (!stream) {
92  OnRetryFailure(Status(StatusCode::kUnknown, "null stream"));
93  return;
94  }
95  shutdown_manager_->StartOperation(__func__, "InitialStart", [&] {
96  stream_ = std::move(stream);
97  RetryLoopState rs{std::move(retry_policy), std::move(backoff_policy)};
98  auto weak = WeakFromThis();
99  stream_->Start().then([weak, rs, request](future<bool> f) {
100  if (auto s = weak.lock()) s->OnStart(rs, request, f.get());
101  });
102  });
103 }

120 void StreamingSubscriptionBatchSource::OnStart(
121  RetryLoopState rs, google::pubsub::v1::StreamingPullRequest const& request,
122  bool ok) {
123  shutdown_manager_->FinishedOperation("InitialStart");
124  if (!ok) {
125  OnInitialError(std::move(rs));
126  return;
127  }
128  shutdown_manager_->StartOperation(__func__, "InitialWrite", [&] {
129  auto weak = WeakFromThis();
130  stream_->Write(request, grpc::WriteOptions{}.set_write_through())
131  .then([weak, rs](future<bool> f) mutable {
132  if (auto s = weak.lock()) s->OnInitialWrite(std::move(rs), f.get());
133  });
134  });
135 }

137 void StreamingSubscriptionBatchSource::OnInitialWrite(RetryLoopState const& rs,
138  bool ok) {
139  shutdown_manager_->FinishedOperation("InitialWrite");
140  if (!ok) {
141  OnInitialError(std::move(rs));
142  return;
143  }
144  auto scheduled =
145  shutdown_manager_->StartOperation(__func__, "InitialRead", [&] {
146  auto weak = WeakFromThis();
147  stream_->Read().then(
148  [weak,
149  rs](future<
150  absl::optional<google::pubsub::v1::StreamingPullResponse>>
151  f) {
152  if (auto s = weak.lock())
153  s->OnInitialRead(std::move(rs), f.get());
154  });
155  });
156  // This is very rare, but it can happen if the session enters shutdown while
157  // the initial setup is in progress.
158  if (!scheduled) OnInitialError(std::move(rs));
159 }

161 void StreamingSubscriptionBatchSource::OnInitialRead(
162  RetryLoopState rs,
163  absl::optional<google::pubsub::v1::StreamingPullResponse> response) {
164  shutdown_manager_->FinishedOperation("InitialRead");
165  if (!response.has_value()) {
166  OnInitialError(std::move(rs));
167  return;
168  }
169 
170  std::unique_lock<std::mutex> lk(mu_);
171  ChangeState(lk, StreamState::kActive, __func__, "success");
172  status_ = Status{};
173  lk.unlock();
174  auto const scheduled =
175  shutdown_manager_->StartOperation(__func__, "read", [&] {
176  OnRead(std::move(response));
177  shutdown_manager_->FinishedOperation("read");
178  });
179  if (!scheduled) {
180  Shutdown();
181  lk.lock();
182  ShutdownStream(std::move(lk), "early-shutdown");
183  }
184 }

186 void StreamingSubscriptionBatchSource::OnInitialError(RetryLoopState rs) {
187  auto weak = WeakFromThis();
188  auto const scheduled =
189  shutdown_manager_->StartOperation(__func__, "finish", [&] {
190  stream_->Finish().then([weak, rs](future<Status> f) {
191  if (auto s = weak.lock()) s->OnInitialFinish(std::move(rs), f.get());
192  });
193  shutdown_manager_->FinishedOperation("finish");
194  });
195  if (!scheduled) {
196  shutdown_manager_->FinishedOperation("stream");
197  }
198 }

200 void StreamingSubscriptionBatchSource::OnInitialFinish(RetryLoopState rs,
201  Status status) {
202  if (!rs.retry_policy->OnFailure(status)) {
203  OnRetryFailure(std::move(status));
204  return;
205  }
206  auto weak = WeakFromThis();
207  using F = future<StatusOr<std::chrono::system_clock::time_point>>;
208  cq_.MakeRelativeTimer(rs.backoff_policy->OnCompletion())
209  .then([weak, rs, status](F f) mutable {
210  auto s = weak.lock();
211  if (!s) return;
212  if (f.get().ok()) {
213  s->OnBackoff(std::move(rs), std::move(status));
214  } else {
215  s->shutdown_manager_->FinishedOperation("stream");
216  }
217  });
218 }

220 void StreamingSubscriptionBatchSource::OnBackoff(RetryLoopState rs,
221  Status status) {
222  if (rs.retry_policy->IsExhausted()) {
223  OnRetryFailure(std::move(status));
224  return;
225  }
226  auto const scheduled =
227  shutdown_manager_->StartOperation(__func__, "retry", [&] {
228  StartStream(std::move(rs.retry_policy), std::move(rs.backoff_policy));
229  shutdown_manager_->FinishedOperation("retry");
230  });
231  if (!scheduled) {
232  shutdown_manager_->FinishedOperation("stream");
233  }
234 }

236 void StreamingSubscriptionBatchSource::OnRetryFailure(Status status) {
237  if (shutdown_manager_->FinishedOperation("stream")) return;
238  shutdown_manager_->MarkAsShutdown(__func__, status);
239  callback_(std::move(status));

242 void StreamingSubscriptionBatchSource::ReadLoop() {
243  std::unique_lock<std::mutex> lk(mu_);
244  if (stream_state_ != StreamState::kActive) return;
245  pending_read_ = true;
246  auto stream = stream_;
247  lk.unlock();
248  auto weak = WeakFromThis();
249  using ResponseType =
250  absl::optional<google::pubsub::v1::StreamingPullResponse>;
251  stream->Read().then([weak, stream](future<ResponseType> f) {
252  if (auto self = weak.lock()) self->OnRead(f.get());
253  });
254 }

256 void StreamingSubscriptionBatchSource::OnRead(
257  absl::optional<google::pubsub::v1::StreamingPullResponse> response) {
258  auto weak = WeakFromThis();
259  std::unique_lock<std::mutex> lk(mu_);
260  pending_read_ = false;
261  if (response && stream_state_ == StreamState::kActive && !shutdown_) {
262  lk.unlock();
263  callback_(*std::move(response));
264  cq_.RunAsync([weak] {
265  if (auto self = weak.lock()) self->ReadLoop();
266  });
267  return;
268  }
269  ShutdownStream(std::move(lk), response ? "state" : "read error");
270 }

272 void StreamingSubscriptionBatchSource::ShutdownStream(
273  std::unique_lock<std::mutex> lk, char const* reason) {
274  if (stream_state_ != StreamState::kActive &&
275  stream_state_ != StreamState::kDisconnecting) {
276  return;
277  }
278  ChangeState(lk, StreamState::kDisconnecting, __func__, reason);
279  if (pending_read_ || pending_write_) return;
280 
281  auto stream = stream_;
282  ChangeState(lk, StreamState::kFinishing, __func__, reason);
283  lk.unlock();
284  auto weak = WeakFromThis();
285  // There are no pending reads or writes, and something (probable a read or
286  // write error) recommends we shutdown the stream
287  stream->Finish().then([weak, stream](future<Status> f) {
288  if (auto self = weak.lock()) self->OnFinish(f.get());
289  });
290 }

292 void StreamingSubscriptionBatchSource::OnFinish(Status status) {
293  std::unique_lock<std::mutex> lk(mu_);
294  status_ = std::move(status);
295  stream_.reset();
296  ChangeState(lk, StreamState::kNull, __func__, "done");
297  if (shutdown_manager_->FinishedOperation("stream")) return;
298  lk.unlock();
299  shutdown_manager_->StartOperation(__func__, "stream", [this] {
300  StartStream(retry_policy_->clone(), backoff_policy_->clone());
301  });
302 }

304 void StreamingSubscriptionBatchSource::DrainQueues(
305  std::unique_lock<std::mutex> lk, bool force_flush) {
306  auto const threshold = force_flush ? 1 : ack_batching_config_.max_batch_size;
307  if (ack_queue_.size() < threshold && nack_queue_.size() < threshold &&
308  deadlines_queue_.empty()) {
309  return;
310  }
311  if (stream_state_ != StreamState::kActive || pending_write_) return;
312  auto stream = stream_;
313  pending_write_ = true;
314 
315  std::vector<std::string> acks;
316  acks.swap(ack_queue_);
317  std::vector<std::string> nacks;
318  nacks.swap(nack_queue_);
319  std::vector<std::pair<std::string, std::chrono::seconds>> deadlines;
320  deadlines.swap(deadlines_queue_);
321  lk.unlock();
322 
323  google::pubsub::v1::StreamingPullRequest request;
324  for (auto& a : acks) request.add_ack_ids(std::move(a));
325  for (auto& n : nacks) {
326  request.add_modify_deadline_ack_ids(std::move(n));
327  request.add_modify_deadline_seconds(0);
328  }
329  for (auto& d : deadlines) {
330  request.add_modify_deadline_ack_ids(std::move(d.first));
331  request.add_modify_deadline_seconds(
332  static_cast<std::int32_t>(d.second.count()));
333  }
334  // Note that we do not use `AsyncRetryLoop()` here. The ack/nack pipeline is
335  // best-effort anyway, there is no guarantee that the server will act on any
336  // of these.
337  auto weak = WeakFromThis();
338  stream->Write(request, grpc::WriteOptions{}.set_write_through())
339  .then([weak, stream](future<bool> f) {
340  if (auto self = weak.lock()) self->OnWrite(f.get());
341  });
342 }

348  DrainQueues(std::move(lk), false);
