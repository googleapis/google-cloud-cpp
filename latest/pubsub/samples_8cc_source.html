82  template <typename Predicate>
83  void Wait(Predicate&& predicate,
84  google::cloud::future<google::cloud::Status> session,
85  std::string const& name) {
86  std::unique_lock<std::mutex> lk(mu_);
87  cv_.wait_for(lk, kWaitTimeout,
88  [this, &predicate] { return predicate(counter_); });
89  lk.unlock();
90  session.cancel();
91  WaitForSession(std::move(session), name);
92  }
