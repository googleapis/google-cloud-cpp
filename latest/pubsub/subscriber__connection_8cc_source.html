120 std::shared_ptr<pubsub::SubscriberConnection> MakeSubscriberConnection(
121  pubsub::Subscription subscription, pubsub::SubscriberOptions options,
122  pubsub::ConnectionOptions connection_options,
123  std::vector<std::shared_ptr<SubscriberStub>> stubs,
124  std::unique_ptr<pubsub::RetryPolicy const> retry_policy,
125  std::unique_ptr<pubsub::BackoffPolicy const> backoff_policy) {
126  auto default_retry_policy = [] {
127  // Subscribers are special: by default we want to retry essentially forever
128  // because (a) the service will disconnect the streaming pull from time to
129  // time, but that is not a "failure", (b) applications can change this
130  // behavior if they need, and this is easier than some hard-coded
131  // "treat these disconnects as non-failures" code.
132  return pubsub::LimitedErrorCountRetryPolicy(std::numeric_limits<int>::max())
133  .clone();
134  };
135  if (stubs.empty()) return nullptr;
136  if (!retry_policy) retry_policy = default_retry_policy();
137  if (!backoff_policy) backoff_policy = DefaultBackoffPolicy();
138  std::shared_ptr<SubscriberStub> stub =
139  std::make_shared<SubscriberRoundRobin>(std::move(stubs));
140  stub = std::make_shared<SubscriberMetadata>(std::move(stub));
141  if (connection_options.tracing_enabled("rpc")) {
142  GCP_LOG(INFO) << "Enabled logging for gRPC calls";
143  stub = std::make_shared<pubsub_internal::SubscriberLogging>(
144  std::move(stub), connection_options.tracing_options(),
145  connection_options.tracing_enabled("rpc-streams"));
146  }
147  auto default_thread_pool_size = []() -> std::size_t {
148  auto constexpr kDefaultThreadPoolSize = 4;
149  auto const n = std::thread::hardware_concurrency();
150  return n == 0 ? kDefaultThreadPoolSize : n;
151  };
152  if (connection_options.background_thread_pool_size() == 0) {
153  connection_options.set_background_thread_pool_size(
154  default_thread_pool_size());
155  }
156  return std::make_shared<SubscriberConnectionImpl>(
157  std::move(subscription), std::move(options),
158  std::move(connection_options), std::move(stub), std::move(retry_policy),
159  std::move(backoff_policy));
160 }
