71 absl::optional<std::size_t> ParseSize(std::string const& str) {
72  std::size_t econv = -1;
73  auto const val = std::stol(str, &econv);
74  if (econv != str.size()) return {};
75  if (val <= 0) return {};

131 void LogSink::Log(LogRecord log_record) {
132  auto copy = CopyBackends();
133  if (copy.empty()) return;
134  // In general, we just give each backend a const-reference and the backends
135  // must make a copy if needed. But if there is only one backend we can give
136  // the backend an opportunity to optimize things by transferring ownership of
137  // the LogRecord to it.
138  if (copy.size() == 1) {
139  copy.begin()->second->ProcessWithOwnership(std::move(log_record));
140  return;
141  }
142  for (auto& kv : copy) {
143  kv.second->Process(log_record);
144  }
145 }

200 std::shared_ptr<LogBackend> DefaultLogBackend() {
201  auto clog_or_none = []() -> std::shared_ptr<LogBackend> {
202  if (internal::GetEnv("GOOGLE_CLOUD_CPP_ENABLE_CLOG").has_value()) {
203  return std::make_shared<StdClogBackend>();
204  }
205  return {};
206  };
207 
208  auto constexpr kVariableName = "GOOGLE_CLOUD_CPP_EXPERIMENTAL_LOG_CONFIG";
209  auto config = internal::GetEnv(kVariableName).value_or("");
210  std::vector<std::string> fields = absl::StrSplit(config, ',');
211  if (fields.empty()) return clog_or_none();
212  if (fields[0] == "lastN" && fields.size() == 3) {
213  auto size = ParseSize(fields[1]);
214  auto min_severity = ParseSeverity(fields[2]);
215  if (size.has_value() && min_severity.has_value()) {
216  return std::make_shared<CircularBufferBackend>(
217  *size, *min_severity, std::make_shared<StdClogBackend>());
218  }
219  }
220  if (fields[0] == "clog" && fields.size() == 1) {
221  return std::make_shared<StdClogBackend>();
222  }
223  return clog_or_none();
224 }
