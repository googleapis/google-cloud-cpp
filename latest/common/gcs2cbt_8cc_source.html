57 int main(int argc, char* argv[]) try {
58  auto const options = ParseArgs(argc, argv);
59 
60  // Create a connection to Cloud Bigtable and an object to manipulate the
61  // specific table used in this demo.
62  cbt::Table table(cbt::CreateDefaultDataClient(
63  options.project_id, options.instance_id,
64  cbt::ClientOptions().set_connection_pool_size(
65  std::thread::hardware_concurrency())),
66  options.table_id);
67  cbt::MutationBatcher batcher(table);
68 
69  // How often do we print a progress marker ('.') in the reader thread.
70  int const report_reader_progress_rate = 500000;
71  // How often do we print a progress marker ('+') in the worker threads.
72  int const report_worker_progress_rate = 500000;
73  // The size of the thread pool pushing data to Cloud Bigtable
74  std::size_t const thread_pool_size = []() -> std::size_t {
75  if (std::thread::hardware_concurrency() != 0) {
76  return std::thread::hardware_concurrency() - 1;
77  }
78  return 1;
79  }();
80 
81  std::cout << "Starting " << thread_pool_size << " workers ..." << std::flush;
82  google::cloud::bigtable::CompletionQueue cq;
83  std::vector<std::thread> thread_pool;
84  for (std::size_t i = 0; i != thread_pool_size; ++i) {
85  thread_pool.emplace_back([&cq] { cq.Run(); });
86  }
87  std::cout << " DONE\n";
88 
89  google::cloud::StatusOr<gcs::ClientOptions> opts =
90  gcs::ClientOptions::CreateDefaultClientOptions();
91  if (!opts) {
92  std::cerr << "Couldn't create gcs::ClientOptions, status=" << opts.status();
93  return 1;
94  }
95  gcs::Client client(opts->set_project_id(options.project_id));
96  // The main thread just reads the file one line at a time.
97  auto is = client.ReadObject(options.bucket, options.object);
98  std::string line;
99  std::getline(is, line, '\n');
100  int lineno = 0;
101  std::vector<std::string> const headers =
102  absl::StrSplit(line, options.separator);
103 
104  std::cout << "# HEADER " << line << "\n";
105 
106  std::cout << "Reading input file " << std::flush;
107  auto start = std::chrono::steady_clock::now();
108 
109  std::atomic<int> apply_finished_count(0);
110  auto report_progress_callback =
111  [=, &apply_finished_count](
112  ::google::cloud::future<::google::cloud::Status> status_future) {
113  if ((apply_finished_count.fetch_add(1) + 1) %
114  report_worker_progress_rate ==
115  0) {
116  std::cout << '+' << std::flush;
117  }
118  auto status = status_future.get();
119  if (!status.ok()) {
120  std::cerr << "Apply failed: " << status;
121  }
122  };
123  while (std::getline(is, line, '\n')) {
124  ++lineno;
125 
126  if (line.empty()) {
127  break;
128  }
129  std::vector<std::string> const parsed =
130  absl::StrSplit(line, options.separator);
131 
132  using std::chrono::milliseconds;
133  auto ts = std::chrono::duration_cast<milliseconds>(
134  std::chrono::system_clock::now().time_since_epoch());
135 
136  // Create the row key by concatenating the desired columns.
137  std::string row_key;
138  std::string row_key_separator;
139  for (auto&& index : options.keys) {
140  row_key += row_key_separator;
141  row_key_separator = options.keys_separator;
142  if (index == -1) {
143  // Magic index, use the line number
144  row_key += std::to_string(lineno);
145  } else {
146  row_key += parsed.at(index);
147  }
148  }
149 
150  // Create a mutation that inserts one column per field, the name of the
151  // column is derived from the header.
152  cbt::SingleRowMutation mutation(row_key);
153  auto field_count = std::min(headers.size(), parsed.size());
154  for (std::size_t i = 0; i != field_count; ++i) {
155  mutation.emplace_back(
156  cbt::SetCell(options.family, headers[i], ts, parsed[i]));
157  }
158  auto admission_and_completion = batcher.AsyncApply(cq, std::move(mutation));
159  admission_and_completion.second.then(report_progress_callback);
160 
161  if (lineno % report_reader_progress_rate == 0) {
162  std::cout << '.' << std::flush;
163  }
164 
165  // Wait until there is space in buffers.
166  admission_and_completion.first.get();
167  }
168  batcher.AsyncWaitForNoPendingRequests().get();
169  std::cout << " DONE\n";
170 
171  std::cout << "Waiting for worker threads " << std::flush;
172  // Let the workers know that they can exit.
173  cq.Shutdown();
174  for (auto& thread : thread_pool) {
175  thread.join();
176  }
177  std::cout << " DONE\n";
178 
179  auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
180  std::chrono::steady_clock::now() - start);
181  std::cout << "Total running time " << elapsed.count() << "s\n";
182 
183  return 0;
184 } catch (std::exception const& ex) {
185  std::cerr << "Standard exception raised: " << ex.what() << "\n";
186  return 1;
187 }

191 std::string ConsumeArg(Options& options, std::vector<std::string>& argv,
192  char const* arg_name) {
193  std::string const separator_option = "--separator=";
194  std::string const key_option = "--key=";
195  std::string const keys_separator_option = "--key-separator=";
196 
197  std::string const usage = R""(
198 [options] <project> <instance> <table> <family> <bucket> <object>
199 The options are:
200  --help: produce this help.
201  --separator=c: use the 'c' character instead of comma (',') to separate the
202  values in the CSV file.
203  --key=N: use field number N as part of the row key. The fields are numbered
204  starting at one. They are concatenated in the order provided.
205  --key-separator=sep: use 'sep' to separate the fields when forming the row
206  key.
207  project: the Google Cloud Platform project id for your table.
208  instance: the Cloud Bigtable instance hosting your table.
209  table: the table where you want to upload the CSV file.
210  family: the column family where you want to upload the CSV file.
211  bucket: the name of the GCS bucket that contains the data.
212  object: the name of the GCS object that contains the data.
213 )"";
214  while (argv.size() >= 2) {
215  std::string argument(argv[1]);
216  argv.erase(argv.begin() + 1);
217  if (argument == "--help") {
218  } else if (0 == argument.find(separator_option)) {
219  options.separator = argument.substr(separator_option.size())[0];
220  } else if (0 == argument.find(key_option)) {
221  options.keys.push_back(std::stoi(argument.substr(key_option.size())) - 1);
222  } else if (0 == argument.find(keys_separator_option)) {
223  options.keys_separator = argument.substr(keys_separator_option.size());
224  } else {
225  return argument;
226  }
227  }
228  std::string cmd = argv[0];
229  auto last_slash = std::string(cmd).find_last_of('/');
230  cmd = cmd.substr(last_slash);
231 
232  std::ostringstream os;
233  os << "Missing argument " << arg_name << "\n";
234  os << "Usage: " << cmd << usage << "\n";
235  throw std::runtime_error(os.str());
236 }

238 Options ParseArgsNoAutoRun(int argc, char const* const argv[]) {
239  // Parse the command-line arguments.
240  Options options;
241  std::vector<std::string> args(argv, argv + argc);
242  options.project_id = ConsumeArg(options, args, "project_id");
243  options.instance_id = ConsumeArg(options, args, "instance_id");
244  options.table_id = ConsumeArg(options, args, "table_id");
245  options.family = ConsumeArg(options, args, "family");
246  options.bucket = ConsumeArg(options, args, "bucket");
247  options.object = ConsumeArg(options, args, "object");
248  // If the user does not say, use the first column as the row key.
249  if (options.keys.empty()) {
250  options.keys.push_back(0);
251  }
252  return options;
253 }

259  using google::cloud::internal::Sample;

261  for (auto const& var :

271  auto const instance_id =

276  auto constexpr kObjectName = "gcs2cbt-sample-data.csv";

280 3,v1,v2,v3
281 )""";
282  gcs_client.InsertObject(bucket_name, kObjectName, kTestData).value();
283

285  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
286  instance_id);
287  auto schema = admin.CreateTable(
288  kTableId,
289  cbt::TableConfig({{"fam", cbt::GcRule::MaxNumVersions(2)}}, {}));
290  // Throw the error unless it is "already exists"
291  if (!schema &&

293  (void)schema.value();
294  }
295 
296  char const* argv[] = {
297  "auto-run", "--key=1", "--separator=,", project_id.c_str(),
298  instance_id.c_str(), kTableId, "fam", bucket_name.c_str(),
299  kObjectName};
300  int argc = sizeof(argv) / sizeof(argv[0]);
301  return ParseArgsNoAutoRun(argc, argv);
302 }
