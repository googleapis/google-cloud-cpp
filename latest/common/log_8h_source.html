390  LogRecord record;

392  record.function = function_;
393  record.filename = filename_;
394  record.lineno = lineno_;
395  record.timestamp = std::chrono::system_clock::now();
396  record.message = stream_->str();
397  sink.Log(std::move(record));
398  }
