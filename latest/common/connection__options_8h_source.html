263 template <typename ConnectionTraits>
264 Options MakeOptions(ConnectionOptions<ConnectionTraits> old) {
265  Options opts = std::move(old.opts_);
266  opts.set<UserAgentProductsOption>({std::move(old.user_agent_prefix_)});
267  opts.set<GrpcBackgroundThreadsFactoryOption>(
268  old.background_threads_factory());
269  if (!old.channel_pool_domain_.empty()) {
270  // To get a different channel pool one just needs to set any channel
271  // parameter to a different value. Newer versions of gRPC include a macro
272  // for this purpose (GRPC_ARG_CHANNEL_POOL_DOMAIN). As we are compiling
273  // against older versions in some cases, we use the actual value.
274  opts.set<GrpcChannelArgumentsOption>(
275  {{"grpc.channel_pooling_domain", std::move(old.channel_pool_domain_)}});
276  }
277  return opts;
278 }
