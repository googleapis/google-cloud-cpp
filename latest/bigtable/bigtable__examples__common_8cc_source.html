33 bool RunAdminIntegrationTests() {
34  // When using the emulator we can always run the admin integration tests
35  if (UsingEmulator()) return true;

109 Commands::value_type MakeCommandEntry(std::string const& name,
110  std::vector<std::string> const& args,
111  TableAsyncCommandType const& command) {
112  auto adapter = [=](std::vector<std::string> argv) {
113  std::vector<std::string> const common{"<project-id>", "<instance-id>",
114  "<table-id>"};
115  if ((argv.size() == 1 && argv[0] == "--help") ||
116  argv.size() != common.size() + args.size()) {
117  std::ostringstream os;
118  os << name;
119  for (auto const& a : common) {
120  os << " " << a;
121  }
122  for (auto const& a : args) {
123  os << " " << a;
124  }
125  throw Usage{std::move(os).str()};
126  }
127  google::cloud::bigtable::Table table(
128  google::cloud::bigtable::CreateDefaultDataClient(
129  argv[0], argv[1], google::cloud::bigtable::ClientOptions()),
130  argv[2]);
131  google::cloud::CompletionQueue cq;
132  std::thread t([&cq] { cq.Run(); });
133  AutoShutdownCQ shutdown(cq, std::move(t));
134  argv.erase(argv.begin(), argv.begin() + common.size());
135  command(std::move(table), cq, std::move(argv));
136  };
137  return Commands::value_type{name, std::move(adapter)};
138 }
