47  //! [async-apply]
48  [&](cbt::Table table, cbt::CompletionQueue cq, std::string table_id) {
49  // Write several rows with some trivial data.
50  for (int i = 0; i != 20; ++i) {
51  // Note: This example uses sequential numeric IDs for simplicity, but
52  // this can result in poor performance in a production application.
53  // Since rows are stored in sorted order by key, sequential keys can
54  // result in poor distribution of operations across nodes.
55  //
56  // For more information about how to design a Bigtable schema for the
57  // best performance, see the documentation:
58  //
59  // https://cloud.google.com/bigtable/docs/schema-design
60 
61  char buf[32];
62  snprintf(buf, sizeof(buf), "key-%06d", i);
63  google::cloud::bigtable::SingleRowMutation mutation(buf);
64  mutation.emplace_back(google::cloud::bigtable::SetCell(
65  "fam", "col0", "value0-" + std::to_string(i)));
66  mutation.emplace_back(google::cloud::bigtable::SetCell(
67  "fam", "col1", "value2-" + std::to_string(i)));
68  mutation.emplace_back(google::cloud::bigtable::SetCell(
69  "fam", "col2", "value3-" + std::to_string(i)));
70  mutation.emplace_back(google::cloud::bigtable::SetCell(
71  "fam", "col3", "value4-" + std::to_string(i)));
72 
73  google::cloud::future<google::cloud::Status> fut =
74  table.AsyncApply(std::move(mutation), cq);
75  google::cloud::Status status = fut.get();
76  if (!status.ok()) {
77  throw std::runtime_error(status.message());
78  }
79  }
80  }
81  //! [async-apply]
82  (std::move(table), std::move(cq), argv[1]);

128 void AsyncCheckAndMutate(cbt::Table table, cbt::CompletionQueue cq,
129  std::vector<std::string> argv) {
130  if (argv.size() != 2U) {

132  "async-check-and-mutate: <project-id> <instance-id> <table-id>"};

135  //! [async check and mutate]
136  [](cbt::Table table, cbt::CompletionQueue cq, std::string table_id) {
137  // Check if the latest value of the flip-flop column is "on".
138  auto predicate = google::cloud::bigtable::Filter::Chain(
139  google::cloud::bigtable::Filter::ColumnRangeClosed("fam", "flip-flop",
140  "flip-flop"),
141  google::cloud::bigtable::Filter::Latest(1),
142  google::cloud::bigtable::Filter::ValueRegex("on"));
143  google::cloud::future<google::cloud::StatusOr<
144  google::bigtable::v2::CheckAndMutateRowResponse>>
145  future = table.AsyncCheckAndMutateRow(
146  MAGIC_ROW_KEY, std::move(predicate),
147  {google::cloud::bigtable::SetCell("fam", "flip-flop", "off"),
148  google::cloud::bigtable::SetCell("fam", "flop-flip", "on")},
149  {google::cloud::bigtable::SetCell("fam", "flip-flop", "on"),
150  google::cloud::bigtable::SetCell("fam", "flop-flip", "off")},
151  cq);
152 
153  auto final =
154  future.then([](google::cloud::future<google::cloud::StatusOr<
155  google::bigtable::v2::CheckAndMutateRowResponse>>
156  f) {
157  auto row = f.get();
158  if (!row) {
159  throw std::runtime_error(row.status().message());
160  }
161 
162  return google::cloud::Status();
163  });
164  final.get();
165  }
166  //! [async check and mutate]
167  (std::move(table), std::move(cq), argv[1]);

204 int main(int argc, char* argv[]) try {
205  using CommandType = std::function<void(
206  google::cloud::bigtable::Table, google::cloud::bigtable::CompletionQueue,
207  std::vector<std::string>)>;
208 
209  std::map<std::string, CommandType> commands = {
210  {"async-apply", &AsyncApply},
211  {"async-bulk-apply", &AsyncBulkApply},
212  {"async-check-and-mutate", &AsyncCheckAndMutate},
213  {"async-read-modify-write", &AsyncReadModifyWrite}};
214 
215  google::cloud::bigtable::CompletionQueue cq;
216 
217  {
218  // Force each command to generate its Usage string, so we can provide a good
219  // usage string for the whole program. We need to create an Table
220  // object to do this, but that object is never used, it is passed to the
221  // commands, without any calls made to it.
222  google::cloud::bigtable::Table unused(
223  google::cloud::bigtable::CreateDefaultDataClient(
224  "unused-project", "Unused-instance",
225  google::cloud::bigtable::ClientOptions()),
226  "Unused-table");
227  for (auto&& kv : commands) {
228  try {
229  kv.second(unused, cq, {});
230  } catch (Usage const& u) {
231  command_usage += " ";
232  command_usage += u.msg;
233  command_usage += "\n";
234  }
235  }
236  }
237 
238  if (argc < 5) {
239  PrintUsage(argv[0],
240  "Missing command and/or project-id/ or instance-id or table-id");
241  return 1;
242  }
243 
244  std::vector<std::string> args;
245  args.emplace_back(argv[0]);
246  std::string const command_name = argv[1];
247  std::string const project_id = argv[2];
248  std::string const instance_id = argv[3];
249  std::string const table_id = argv[4];
250  std::transform(argv + 4, argv + argc, std::back_inserter(args),
251  [](char* x) { return std::string(x); });
252 
253  auto command = commands.find(command_name);
254  if (commands.end() == command) {
255  PrintUsage(argv[0], "Unknown command: " + command_name);
256  return 1;
257  }
258 
259  // Start a thread to run the completion queue event loop.
260  std::thread runner([&cq] { cq.Run(); });
261 
262  // Connect to the Cloud Bigtable endpoint.
263  //! [connect data]
264  google::cloud::bigtable::Table table(
265  google::cloud::bigtable::CreateDefaultDataClient(
266  project_id, instance_id, google::cloud::bigtable::ClientOptions()),
267  table_id);
268  //! [connect data]
269 
270  command->second(table, cq, args);
271 
272  // Shutdown the completion queue event loop and join the thread.
273  cq.Shutdown();
274  runner.join();
275 
276  return 0;
277 } catch (Usage const& ex) {
278  PrintUsage(argv[0], ex.msg);
279  return 1;
280 } catch (std::exception const& ex) {
281  std::cerr << "Standard C++ exception raised: " << ex.what() << "\n";
282  return 1;
283 }
