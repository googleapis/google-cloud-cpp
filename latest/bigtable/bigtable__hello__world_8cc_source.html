32 void BigtableHelloWorld(std::vector<std::string> const& argv) {
33  if (argv.size() != 3) {
34  throw Usage{"hello-world <project-id> <instance-id> <table-id>"};
35  }
36  std::string const project_id = argv[0];
37  std::string const instance_id = argv[1];
38  std::string const table_id = argv[2];
39 
40  // Create a namespace alias to make the code easier to read.
42  namespace cbt = google::cloud::bigtable;
43  using google::cloud::StatusOr;
45 
46  // Connect to the Cloud Bigtable Admin API.
48  cbt::TableAdmin table_admin(
49  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions()),
50  instance_id);
52 
54  // Define the desired schema for the Table.
55  cbt::GcRule gc_rule = cbt::GcRule::MaxNumVersions(1);
56  cbt::TableConfig schema({{"family", gc_rule}}, {});
57 
58  // Create a table.
59  StatusOr<google::bigtable::admin::v2::Table> returned_schema =
60  table_admin.CreateTable(table_id, schema);
62 
63  // Create an object to access the Cloud Bigtable Data API.
65  cbt::Table table(cbt::CreateDefaultDataClient(project_id, instance_id,
66  cbt::ClientOptions()),
67  table_id);
69 
70  // Modify (and create if necessary) a row.
72  std::vector<std::string> greetings{"Hello World!", "Hello Cloud Bigtable!",
73  "Hello C++!"};
74  int i = 0;
75  for (auto const& greeting : greetings) {
76  // Each row has a unique row key.
77  //
78  // Note: This example uses sequential numeric IDs for simplicity, but
79  // this can result in poor performance in a production application.
80  // Since rows are stored in sorted order by key, sequential keys can
81  // result in poor distribution of operations across nodes.
82  //
83  // For more information about how to design a Bigtable schema for the
84  // best performance, see the documentation:
85  //
86  // https://cloud.google.com/bigtable/docs/schema-design
87  std::string row_key = "key-" + std::to_string(i);
88  google::cloud::Status status = table.Apply(cbt::SingleRowMutation(
89  std::move(row_key), cbt::SetCell("family", "c0", greeting)));
90 
91  if (!status.ok()) throw std::runtime_error(status.message());
92  ++i;
93  }
95 
97  cbt::Filter filter = cbt::Filter::ColumnRangeClosed("family", "c0", "c0");
99 
100  // Read a single row.
102  StatusOr<std::pair<bool, cbt::Row>> result = table.ReadRow("key-0", filter);
103  if (!result) throw std::runtime_error(result.status().message());
104  if (!result->first) {
105  std::cout << "Cannot find row 'key-0' in the table: " << table.table_name()
106  << "\n";
107  return;
108  }
109  cbt::Cell const& cell = result->second.cells().front();
110  std::cout << cell.family_name() << ":" << cell.column_qualifier() << " @ "
111  << cell.timestamp().count() << "us\n"
112  << '"' << cell.value() << '"' << "\n";
114 
115  // Read all rows.
117  for (StatusOr<cbt::Row> const& row : table.ReadRows(
118  cbt::RowRange::InfiniteRange(), cbt::Filter::PassAllFilter())) {
119  if (!row) throw std::runtime_error(row.status().message());
120  std::cout << row->row_key() << ":\n";
121  for (cbt::Cell const& c : row->cells()) {
122  std::cout << "\t" << c.family_name() << ":" << c.column_qualifier()
123  << " @ " << c.timestamp().count() << "us\n"
124  << "\t\"" << c.value() << '"' << "\n";
125  }
126  }
128 
129  // Delete the table
131  google::cloud::Status status = table_admin.DeleteTable(table_id);
132  if (!status.ok()) throw std::runtime_error(status.message());
134 }
