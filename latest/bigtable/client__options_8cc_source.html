32 // As learned from experiments, idle gRPC connections enter IDLE state after 4m.
33 auto constexpr kDefaultMaxRefreshPeriod =
34  std::chrono::milliseconds(std::chrono::minutes(3));
35 
36 // Applications with hundreds of clients seem to work better with a longer
37 // delay for the initial refresh. As there is no particular rush, start with 1m.
38 auto constexpr kDefaultMinRefreshPeriod =
39  std::chrono::milliseconds(std::chrono::minutes(1));
40 
41 static_assert(kDefaultMinRefreshPeriod <= kDefaultMaxRefreshPeriod,
42  "The default period range must be valid");

102 inline std::size_t CalculateDefaultConnectionPoolSize() {
103  // For batter resource utilization and greater throughput, it is recommended
104  // to calculate the default pool size based on cores(CPU) available. However,
105  // as per C++11 documentation `std::thread::hardware_concurrency()` cannot be
106  // fully rely upon, it is only a hint and the value can be 0 if it is not
107  // well defined or not computable. Apart from CPU count, multiple channels
108  // can be opened for each CPU to increase throughput. The pool size is also
109  // capped so that servers with many cores do not create too many channels.
110  std::size_t cpu_count = std::thread::hardware_concurrency();
111  if (cpu_count == 0) return BIGTABLE_CLIENT_DEFAULT_CONNECTION_POOL_SIZE;
112  return (
113  std::min<std::size_t>)(BIGTABLE_CLIENT_DEFAULT_CONNECTION_POOL_SIZE_MAX,
114  cpu_count *
115  BIGTABLE_CLIENT_DEFAULT_CHANNELS_PER_CPU);
116 }

118 ClientOptions::ClientOptions(std::shared_ptr<grpc::ChannelCredentials> creds)
119  : credentials_(std::move(creds)),
120  connection_pool_size_(CalculateDefaultConnectionPoolSize()),
121  data_endpoint_("bigtable.googleapis.com"),
122  admin_endpoint_("bigtableadmin.googleapis.com"),
123  instance_admin_endpoint_("bigtableadmin.googleapis.com"),
124  tracing_components_(internal::DefaultTracingComponents()),
125  tracing_options_(internal::DefaultTracingOptions()),
126  // Refresh connections before the server has a chance to disconnect them
127  // due to being idle.
128  max_conn_refresh_period_(kDefaultMaxRefreshPeriod),
129  min_conn_refresh_period_(kDefaultMinRefreshPeriod) {
130  using std::chrono::duration_cast;
131  using std::chrono::milliseconds;
132 
133  static std::string const kUserAgentPrefix = UserAgentPrefix();
134  channel_arguments_.SetUserAgentPrefix(kUserAgentPrefix);
135  channel_arguments_.SetMaxSendMessageSize(
136  BIGTABLE_CLIENT_DEFAULT_MAX_MESSAGE_LENGTH);
137  channel_arguments_.SetMaxReceiveMessageSize(
138  BIGTABLE_CLIENT_DEFAULT_MAX_MESSAGE_LENGTH);
139  auto to_arg = [](std::chrono::milliseconds ms) {
140  auto const count = ms.count();
141  if (count >= (std::numeric_limits<int>::max)()) {
142  return std::numeric_limits<int>::max();
143  }
144  if (count <= (std::numeric_limits<int>::min)()) {
145  return std::numeric_limits<int>::min();
146  }
147  return static_cast<int>(ms.count());
148  };
149  channel_arguments_.SetInt(
150  GRPC_ARG_KEEPALIVE_TIME_MS,
151  to_arg(duration_cast<milliseconds>(kDefaultKeepaliveTime)));
152  channel_arguments_.SetInt(
153  GRPC_ARG_KEEPALIVE_TIMEOUT_MS,
154  to_arg(duration_cast<milliseconds>(kDefaultKeepaliveTimeout)));
155 }
