29 void CreateBackup(google::cloud::bigtable::TableAdmin const& admin,
30  std::vector<std::string> const& argv) {
32  namespace cbt = google::cloud::bigtable;
33  using google::cloud::StatusOr;
34  [](cbt::TableAdmin admin, std::string const& table_id,
35  std::string const& cluster_id, std::string const& backup_id,
36  std::string const& expire_time_string) {
37  absl::Time t;
38  std::string err;
39  if (!absl::ParseTime(absl::RFC3339_full, expire_time_string, &t, &err)) {
40  throw std::runtime_error("Unable to parse expire_time:" + err);
41  }
42  auto expire_time = absl::ToChronoTime(t);
43  StatusOr<google::bigtable::admin::v2::Backup> backup =
44  admin.CreateBackup(cbt::TableAdmin::CreateBackupParams(
45  cluster_id, backup_id, table_id, expire_time));
46  if (!backup) throw std::runtime_error(backup.status().message());
47  std::cout << "Backup successfully created: " << backup->DebugString()
48  << "\n";
49  }
51  (admin, argv.at(0), argv.at(1), argv.at(2), argv.at(3));
52 }

183  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
184  auto const instance_id = google::cloud::internal::GetEnv(
185  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
186  .value();
187  auto const service_account =
188  google::cloud::internal::GetEnv(
189  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_SERVICE_ACCOUNT")

193  .value();
194 
195  cbt::TableAdmin admin(
196  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
197  instance_id);
198 
199  // If a previous run of these samples crashes before cleaning up there may be
200  // old tables left over. As there are quotas on the total number of tables we
201  // remove stale tables after 48 hours.
202  std::cout << "\nCleaning up old tables" << std::endl;
203  std::string const prefix = "table-admin-snippets-";
204  google::cloud::bigtable::testing::CleanupStaleTables(admin);
205  std::string const backup_prefix = "table-admin-snippets-backup-";
206  google::cloud::bigtable::testing::CleanupStaleBackups(admin);

210  // GetTable() and "table does exist")
211  auto table_id_1 = google::cloud::bigtable::testing::RandomTableId(generator);
212 
213  auto table_1 = admin.CreateTable(
214  table_id_1, cbt::TableConfig(
