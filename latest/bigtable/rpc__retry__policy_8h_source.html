31 struct SafeGrpcRetry {
32  static inline bool IsTransientFailure(google::cloud::StatusCode code) {
33  return code == StatusCode::kAborted || code == StatusCode::kUnavailable ||
34  code == StatusCode::kDeadlineExceeded;

37  static inline bool IsOk(google::cloud::Status const& status) {
38  return status.ok();

41  return IsTransientFailure(status.code());

43  static inline bool IsPermanentFailure(google::cloud::Status const& status) {
