33 void BigtableHelloInstance(std::vector<std::string> const& argv) {
34  if (argv.size() != 4) {
35  throw Usage{
36  "bigtable-hello-instance <project-id> <instance-id> <cluster-id> "
37  "<zone>"};
38  }
39 
40  std::string const project_id = argv[0];
41  std::string const instance_id = argv[1];
42  std::string const cluster_id = argv[2];
43  std::string const zone = argv[3];
44 
46  namespace cbt = google::cloud::bigtable;
47  using google::cloud::future;
48  using google::cloud::StatusOr;
50 
51  // Connect to the Cloud Bigtable admin endpoint.
53  cbt::InstanceAdmin instance_admin(
54  cbt::CreateDefaultInstanceAdminClient(project_id, cbt::ClientOptions()));
56 
58  std::cout << "\nCheck Instance exists:\n";
59  StatusOr<cbt::InstanceList> instances = instance_admin.ListInstances();
60  if (!instances) throw std::runtime_error(instances.status().message());
61  if (!instances->failed_locations.empty()) {
62  std::cerr
63  << "The service tells us it has no information about these locations:";
64  for (auto const& failed_location : instances->failed_locations) {
65  std::cerr << " " << failed_location;
66  }
67  std::cerr << ". Continuing anyway\n";
68  }
69  auto instance_name =
70  instance_admin.project_name() + "/instances/" + instance_id;
71  bool instance_exists = google::cloud::internal::ContainsIf(
72  instances->instances,
73  [&instance_name](google::bigtable::admin::v2::Instance const& i) {
74  return i.name() == instance_name;
75  });
76  std::cout << "The instance " << instance_id
77  << (instance_exists ? " already exists" : " does not exist")
78  << "\n";
80 
81  // Create instance if does not exists
82  if (!instance_exists) {
84  std::cout << "\nCreating a PRODUCTION Instance: ";
85 
86  // production instance needs at least 3 nodes
87  auto cluster_config = cbt::ClusterConfig(zone, 3, cbt::ClusterConfig::HDD);
88  cbt::InstanceConfig config(instance_id, "Sample Instance",
89  {{cluster_id, cluster_config}});
90  config.set_type(cbt::InstanceConfig::PRODUCTION);
91 
92  google::cloud::future<void> creation_done =
93  instance_admin.CreateInstance(config).then(
94  [instance_id](
95  future<StatusOr<google::bigtable::admin::v2::Instance>> f) {
96  auto instance = f.get();
97  if (!instance) {
98  std::cerr << "Could not create instance " << instance_id
99  << "\n";
100  throw std::runtime_error(instance.status().message());
101  }
102  std::cout << "Successfully created instance: "
103  << instance->DebugString() << "\n";
104  });
105  // Note how this blocks until the instance is created, in production code
106  // you may want to perform this task asynchronously.
107  creation_done.get();
108  std::cout << "DONE\n";
110  }
111 
113  std::cout << "\nListing Instances:\n";
114  instances = instance_admin.ListInstances();
115  if (!instances) throw std::runtime_error(instances.status().message());
116  if (!instances->failed_locations.empty()) {
117  std::cerr
118  << "The service tells us it has no information about these locations:";
119  for (auto const& failed_location : instances->failed_locations) {
120  std::cerr << " " << failed_location;
121  }
122  std::cerr << ". Continuing anyway\n";
123  }
124  for (auto const& instance : instances->instances) {
125  std::cout << " " << instance.name() << "\n";
126  }
127  std::cout << "DONE\n";
129 
131  std::cout << "\nGet Instance:\n";
132  auto instance = instance_admin.GetInstance(instance_id);
133  if (!instance) throw std::runtime_error(instance.status().message());
134  std::cout << "Instance details :\n" << instance->DebugString() << "\n";
136 
138  std::cout << "\nListing Clusters:\n";
139  StatusOr<cbt::ClusterList> cluster_list =
140  instance_admin.ListClusters(instance_id);
141  if (!cluster_list) throw std::runtime_error(cluster_list.status().message());
142  if (!cluster_list->failed_locations.empty()) {
143  std::cout << "The Cloud Bigtable service reports that the following "
144  "locations are temporarily unavailable and no information "
145  "about clusters in these locations can be obtained:\n";
146  for (auto const& failed_location : cluster_list->failed_locations) {
147  std::cout << failed_location << "\n";
148  }
149  }
150  std::cout << "Cluster Name List:\n";
151  for (auto const& cluster : cluster_list->clusters) {
152  std::cout << "Cluster Name: " << cluster.name() << "\n";
153  }
154  std::cout << "DONE\n";
156 
158  std::cout << "Deleting instance " << instance_id << "\n";
159  google::cloud::Status delete_status =
160  instance_admin.DeleteInstance(instance_id);
161  if (!delete_status.ok()) throw std::runtime_error(delete_status.message());
162  std::cout << "DONE\n";
164 }

166 void RunAll(std::vector<std::string> const& argv) {
167  namespace examples = ::google::cloud::bigtable::examples;
168  namespace cbt = google::cloud::bigtable;
169 
170  if (!argv.empty()) throw Usage{"auto"};
171  if (!examples::RunAdminIntegrationTests()) return;
172  examples::CheckEnvironmentVariablesAreSet({
173  "GOOGLE_CLOUD_PROJECT",
174  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_ZONE_A",
175  });
176  auto const project_id =
177  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
178  auto const zone_a =
179  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_BIGTABLE_TEST_ZONE_A")
180  .value();
181 
182  cbt::InstanceAdmin admin(
183  cbt::CreateDefaultInstanceAdminClient(project_id, cbt::ClientOptions{}));
184 
185  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
186  google::cloud::bigtable::testing::CleanupStaleInstances(admin);
187  auto const instance_id =
188  google::cloud::bigtable::testing::RandomInstanceId(generator);
189  auto const cluster_id = instance_id + "-c1";
190 
191  std::cout << "\nRunning the BigtableHelloInstance() example" << std::endl;
192  BigtableHelloInstance({project_id, instance_id, cluster_id, zone_a});
193 }
