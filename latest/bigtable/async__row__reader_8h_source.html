166  void TryGiveRowToUser() {
167  // The user is likely to ask for more rows immediately after receiving a
168  // row, which means that this function will be called recursively. The depth
169  // of the recursion can be as deep as the size of ready_rows_, which might
170  // be significant and potentially lead to stack overflow. The way to
171  // overcome this is to always switch thread to a CompletionQueue thread.
172  // Switching thread for every row has a non-trivial cost, though. To find a
173  // good balance, we allow for recursion no deeper than 100 and achieve it by
174  // tracking the level in `recursion_level_`.
175  //
176  // The magic value 100 is arbitrary, but back-of-the-envelope calculation
177  // indicates it should cap this stack usage to below 100K. Default stack
178  // size is usually 1MB.
179  struct CountFrame {
180  explicit CountFrame(int& cntr) : cntr(++cntr){};
181  ~CountFrame() { --cntr; }
182  int& cntr;
183  };
184  CountFrame frame(recursion_level_);
185 
186  if (ready_rows_.empty()) {
187  if (whole_op_finished_) {
188  // The scan is finished for good, there will be no more rows.
189  on_finish_(status_);
190  return;
191  }
192  if (!continue_reading_) {
193  // TODO(#1402): replace with GCP_ASSERT(continue_reading_);
194  Terminate(
195  "No rows are ready and we can't continue reading. This is a bug, "
196  "please report it at "
197  "https://github.com/googleapis/google-cloud-cpp/issues/new");
198  }
199  // No rows, but we can fetch some.
200  auto continue_reading = std::move(continue_reading_);
201  continue_reading_.reset();
202  continue_reading->set_value(true);
203  return;
204  }
205 
206  // Yay! We have something to give to the user and they want it.
207  auto row = std::move(ready_rows_.front());
208  ready_rows_.pop();
209 
210  auto self = this->shared_from_this();
211  bool const break_recursion = recursion_level_ >= 100;
212  on_row_(std::move(row)).then([self, break_recursion](future<bool> fut) {
213  bool should_cancel;
214 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
215  try {
216  should_cancel = !fut.get();
217  } catch (std::exception& ex) {
218  self->Cancel(
219  std::string("future<> returned from the user callback threw an "
220  "exception: ") +
221  ex.what());
222  return;
223  } catch (...) {
224  self->Cancel(
225  "future<> returned from the user callback threw an unknown "
226  "exception");
227  return;
228  }
229 #else // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
230  should_cancel = !fut.get();
231 #endif // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
232  if (should_cancel) {
233  self->Cancel("User cancelled");
234  return;
235  }
236  if (break_recursion) {
237  self->cq_.RunAsync([self] { self->UserWantsRows(); });
238  return;
239  }
240  self->UserWantsRows();
241  });
242  }

245  future<bool> OnDataReceived(
246  google::bigtable::v2::ReadRowsResponse const& response) {
247  // assert(!whole_op_finished_);
248  // assert(!continue_reading_);
249  // assert(status_.ok());
250  status_ = ConsumeResponse(std::move(response));
251  // We've processed the response.
252  //
253  // If there were errors (e.g. malformed response from the server), we should
254  // interrupt this stream. Interrupting it will yield lower layers calling
255  // `OnStreamFinished` with a status unrelated to the real reason, so we
256  // store the actual reason in status_ and proceed exactly the
257  // same way as if the stream was broken for other reasons.
258  //
259  // Even if status_ is not OK, we might have consumed some rows,
260  // but, don't give them to the user yet. We want to keep the invariant that
261  // either the user doesn't hold a `future<>` when we're fetching more rows.
262  // Retries (successful or not) will do it. Improving this behavior makes
263  // little sense because parser errors are very unexpected and probably not
264  // retryable anyway.
265 
266  if (status_.ok()) {
267  continue_reading_.emplace(promise<bool>());
268  auto res = continue_reading_->get_future();
269  TryGiveRowToUser();
270  return res;
271  }
272  return make_ready_future<bool>(false);
273  }

276  // NOLINTNEXTLINE(performance-unnecessary-value-param)
277  void OnStreamFinished(Status status) {
278  // assert(!continue_reading_);
279  if (status_.ok()) {
280  status_ = std::move(status);
281  }
282  grpc::Status parser_status;
283  parser_->HandleEndOfStream(parser_status);
284  if (!parser_status.ok() && status_.ok()) {
285  // If there stream finished with an error ignore what the parser says.
286  status_ = MakeStatusFromRpcError(parser_status);
287  }
288 
289  // In the unlikely case when we have already reached the requested
290  // number of rows and still receive an error (the parser can throw
291  // an error at end of stream for example), there is no need to
292  // retry and we have no good value for rows_limit anyway.
293  if (rows_limit_ != NO_ROWS_LIMIT && rows_limit_ <= rows_count_) {
294  status_ = Status();
295  }
296 
297  if (!last_read_row_key_.empty()) {
298  // We've returned some rows and need to make sure we don't
299  // request them again.
300  row_set_ = row_set_.Intersect(RowRange::Open(last_read_row_key_, ""));
301  }
302 
303  // If we receive an error, but the retryable set is empty, consider it a
304  // success.
305  if (row_set_.IsEmpty()) {
306  status_ = Status();
307  }
308 
309  if (status_.ok()) {
310  // We've successfully finished the scan.
311  whole_op_finished_ = true;
312  TryGiveRowToUser();
313  return;
314  }
315 
316  if (!rpc_retry_policy_->OnFailure(status_)) {
317  // Can't retry.
318  whole_op_finished_ = true;
319  TryGiveRowToUser();
320  return;
321  }
322  auto self = this->shared_from_this();
323  cq_.MakeRelativeTimer(rpc_backoff_policy_->OnCompletion(status_))
324  .then([self](future<StatusOr<std::chrono::system_clock::time_point>>
325  result) {
326  if (auto tp = result.get()) {
327  self->MakeRequest();
328  } else {
329  self->whole_op_finished_ = true;
330  self->TryGiveRowToUser();
331  }
332  });
333  }

336  void Cancel(std::string const& reason) {
337  ready_rows_ = std::queue<Row>();
338  auto continue_reading = std::move(continue_reading_);
339  continue_reading_.reset();
340  Status status(StatusCode::kCancelled, reason);
341  if (!continue_reading) {
342  // If we're not in the middle of the stream fire some user callbacks, but
343  // also override the overall status.
344  // assert(whole_op_finished_);
345  status_ = std::move(status);
346  TryGiveRowToUser();
347  return;
348  }
349  // If we are in the middle of the stream, cancel the stream.
350  status_ = std::move(status);
351  continue_reading->set_value(false);
352  }

355  Status DrainParser() {
356  grpc::Status status;
357  while (parser_->HasNext()) {
358  Row parsed_row = parser_->Next(status);
359  if (!status.ok()) {
360  return MakeStatusFromRpcError(status);
361  }
362  ++rows_count_;
363  last_read_row_key_ = std::string(parsed_row.row_key());
364  ready_rows_.emplace(std::move(parsed_row));
365  }
366  return Status();
367  }

370  Status ConsumeResponse(google::bigtable::v2::ReadRowsResponse response) {
371  for (auto& chunk : *response.mutable_chunks()) {
372  grpc::Status status;
373  parser_->HandleChunk(std::move(chunk), status);
374  if (!status.ok()) {
375  return MakeStatusFromRpcError(status);
376  }
377  Status parser_status = DrainParser();
378  if (!parser_status.ok()) {
379  return parser_status;
380  }
381  }
382  return Status();
383  }
