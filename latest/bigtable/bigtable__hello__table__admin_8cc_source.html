32 void HelloWorldTableAdmin(std::vector<std::string> const& argv) {
33  if (argv.size() != 3) {
34  throw Usage{
35  "hello-world-table-admin <project-id> <instance-id> "
36  "<table-id>"};
37  }
38  std::string const project_id = argv[0];
39  std::string const instance_id = argv[1];
40  std::string const table_id = argv[2];
41 
43  namespace cbt = google::cloud::bigtable;
44  using google::cloud::StatusOr;
46 
47  // Connect to the Cloud Bigtable admin endpoint.
49  cbt::TableAdmin admin(
50  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions()),
51  instance_id);
53 
55  std::cout << "Creating a table:\n";
56  StatusOr<google::bigtable::admin::v2::Table> schema = admin.CreateTable(
57  table_id,
58  cbt::TableConfig({{"fam", cbt::GcRule::MaxNumVersions(10)},
59  {"foo", cbt::GcRule::MaxAge(std::chrono::hours(72))}},
60  {}));
61  std::cout << "DONE\n";
63 
65  std::cout << "Listing tables:\n";
66  auto tables = admin.ListTables(cbt::TableAdmin::NAME_ONLY);
67  if (!tables) throw std::runtime_error(tables.status().message());
68  for (auto const& table : *tables) {
69  std::cout << " " << table.name() << "\n";
70  }
71  std::cout << "DONE\n";
73 
75  std::cout << "Get table metadata:\n";
76  StatusOr<google::bigtable::admin::v2::Table> table =
77  admin.GetTable(table_id, cbt::TableAdmin::FULL);
78  if (!table) throw std::runtime_error(table.status().message());
79  std::cout << "Table name : " << table->name() << "\n";
80 
81  std::cout << "List table families and GC rules:\n";
82  for (auto const& kv : table->column_families()) {
83  std::string const& family_name = kv.first;
84  google::bigtable::admin::v2::ColumnFamily const& metadata = kv.second;
85  std::cout << "Column Family :" << family_name << "\t"
86  << metadata.DebugString() << "\n";
87  }
88  std::cout << "DONE\n";
90 
92  std::cout << "Update a column family GC rule:\n";
93  StatusOr<google::bigtable::admin::v2::Table> updated_schema =
94  admin.ModifyColumnFamilies(
95  table_id,
96  {cbt::ColumnFamilyModification::Update(
97  "fam", cbt::GcRule::Union(
98  cbt::GcRule::MaxNumVersions(5),
99  cbt::GcRule::MaxAge(std::chrono::hours(24 * 7))))});
100  if (!updated_schema) {
101  throw std::runtime_error(updated_schema.status().message());
102  }
103  std::cout << "Schema modified to: " << updated_schema->DebugString() << "\n";
105 
107  std::cout << "Deleting all the rows in " << table_id << "\n";
108  google::cloud::Status status = admin.DropAllRows(table_id);
109  if (!status.ok()) throw std::runtime_error(status.message());
110  std::cout << "DONE\n";
112 
114  std::cout << "Deleting table:\n";
115  google::cloud::Status delete_status = admin.DeleteTable(table_id);
116  if (!delete_status.ok()) throw std::runtime_error(delete_status.message());
117  std::cout << "DONE\n";
119 }

121 void RunAll(std::vector<std::string> const& argv) {
122  namespace examples = ::google::cloud::bigtable::examples;
123  namespace cbt = google::cloud::bigtable;
124 
125  if (!argv.empty()) throw google::cloud::bigtable::examples::Usage{"auto"};
126  if (!examples::RunAdminIntegrationTests()) return;
127  examples::CheckEnvironmentVariablesAreSet({
128  "GOOGLE_CLOUD_PROJECT",
129  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
130  });
131  auto const project_id =
132  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
133  auto const instance_id = google::cloud::internal::GetEnv(
134  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
135  .value();
136 
137  cbt::TableAdmin admin(
138  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
139  instance_id);
140 
141  google::cloud::bigtable::testing::CleanupStaleTables(admin);
142 
143  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
144  auto table_id = google::cloud::bigtable::testing::RandomTableId(generator);
145 
146  std::cout << "\nRunning the HelloWorldTableAdmin() example" << std::endl;
147  HelloWorldTableAdmin({project_id, instance_id, table_id});
148 }
