41 std::pair<future<void>, future<Status>> MutationBatcher::AsyncApply(
42  CompletionQueue& cq, SingleRowMutation mut) {
43  AdmissionPromise admission_promise;
44  CompletionPromise completion_promise;
45  auto res = std::make_pair(admission_promise.get_future(),
46  completion_promise.get_future());
47  PendingSingleRowMutation pending(std::move(mut),
48  std::move(completion_promise),
49  std::move(admission_promise));
50  std::unique_lock<std::mutex> lk(mu_);
51 
52  grpc::Status mutation_status = IsValid(pending);
53  if (!mutation_status.ok()) {
54  lk.unlock();
55  // Destroy the mutation before satisfying the admission promise so that we
56  // can limit the memory usage.
57  pending.mut.Clear();
58  pending.completion_promise.set_value(
59  MakeStatusFromRpcError(mutation_status));
60  // No need to consider no_more_pending_promises because this operation
61  // didn't lower the number of pending operations.
62  pending.admission_promise.set_value();
63  return res;
64  }
65  ++num_requests_pending_;
66 
67  if (!CanAppendToBatch(pending)) {
68  pending_mutations_.push(std::move(pending));
69  return res;
70  }
71  std::vector<AdmissionPromise> admission_promises_to_satisfy;
72  admission_promises_to_satisfy.emplace_back(
73  std::move(pending.admission_promise));
74  Admit(std::move(pending));
75  FlushIfPossible(cq);
76  SatisfyPromises(std::move(admission_promises_to_satisfy), lk);
77  return res;
78 }

103 grpc::Status MutationBatcher::IsValid(PendingSingleRowMutation& mut) const {
104  // Objects of this class need to be aware of the maximum allowed number of
105  // mutations in a batch because it should not pack more. If we have this
106  // knowledge, we might as well simplify everything and not admit larger
107  // mutations.
108  if (mut.num_mutations > options_.max_mutations_per_batch) {
109  std::stringstream stream;
110  stream << "Too many (" << mut.num_mutations
111  << ") mutations in a SingleRowMutations request. "
112  << options_.max_mutations_per_batch << " is the limit.";
113  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT, stream.str());
114  }
115  if (mut.num_mutations == 0) {
116  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT,
117  "Supplied SingleRowMutations has no entries");
118  }
119  if (mut.request_size > options_.max_size_per_batch) {
120  std::stringstream stream;
121  stream << "Too large (" << mut.request_size
122  << " bytes) mutation in a SingleRowMutations request. "
123  << options_.max_size_per_batch << " bytes is the limit.";
124  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT, stream.str());
125  }
126  return grpc::Status();
127 }

143 bool MutationBatcher::FlushIfPossible(CompletionQueue cq) {
144  if (cur_batch_->num_mutations > 0 &&
145  num_outstanding_batches_ < options_.max_batches) {
146  ++num_outstanding_batches_;
147 
148  auto batch = std::make_shared<Batch>();
149  cur_batch_.swap(batch);
150  AsyncBulkApplyImpl(table_, std::move(batch->requests))
151  .then([this, cq,
152  batch](future<std::vector<FailedMutation>> failed) mutable {
153  // Calling OnBulkApplyDone here might lead to a deadlock if the
154  // underlying operation completes very quickly, yielding the outer
155  // `.then()` call synchronous. The deadlock would occur because the
156  // mutex is held here and OnBulkApplyDone would try to reacquire it.
157  //
158  // We're not using a lambda here because in C++11 that would mean
159  // copying the `failed` vector.
160  struct Functor {
161  void operator()(CompletionQueue& cq) {
162  self->OnBulkApplyDone(cq, std::move(*batch), std::move(failed));
163  }
164 
165  MutationBatcher* self;
166  std::shared_ptr<Batch> batch;
167  std::vector<FailedMutation> failed;
168  };
169  cq.RunAsync(Functor{this, std::move(batch), failed.get()});
170  });
171  return true;
172  }
173  return false;
174 }

176 void MutationBatcher::OnBulkApplyDone(
177  CompletionQueue cq, MutationBatcher::Batch batch,
178  std::vector<FailedMutation> const& failed) {
179  // First process all the failures, marking the mutations as done after
180  // processing them.
181  for (auto const& f : failed) {
182  int const idx = f.original_index();
183  if (idx < 0 ||
184  static_cast<std::size_t>(idx) >= batch.mutation_data.size()) {
185  // This is a bug on the server or the client, either terminate (when
186  // -fno-exceptions is set) or throw an exception.
187  std::ostringstream os;
188  os << "Index " << idx << " is out of range [0,"
189  << batch.mutation_data.size() << ")";
190  google::cloud::internal::ThrowRuntimeError(std::move(os).str());
191  }
192  MutationData& data = batch.mutation_data[idx];
193  data.completion_promise.set_value(f.status());
194  data.done = true;
195  }
196  // Any remaining mutations are treated as successful.
197  for (auto& data : batch.mutation_data) {
198  if (!data.done) {
199  data.completion_promise.set_value(Status());
200  data.done = true;
201  }
202  }
203  auto const num_mutations = batch.mutation_data.size();
204  batch.mutation_data.clear();
205 
206  std::unique_lock<std::mutex> lk(mu_);
207  outstanding_size_ -= batch.requests_size;
208  num_requests_pending_ -= num_mutations;
209  num_outstanding_batches_--;
210  SatisfyPromises(TryAdmit(cq), lk); // unlocks the lock
211 }

213 std::vector<MutationBatcher::AdmissionPromise> MutationBatcher::TryAdmit(
214  CompletionQueue& cq) {
215  // Defer satisfying promises until we release the lock.
216  std::vector<AdmissionPromise> admission_promises;
217 
218  do {
219  while (!pending_mutations_.empty() &&
220  HasSpaceFor(pending_mutations_.front())) {
221  auto& mut = pending_mutations_.front();
222  admission_promises.emplace_back(std::move(mut.admission_promise));
223  Admit(std::move(mut));
224  pending_mutations_.pop();
225  }
226  } while (FlushIfPossible(cq));
227  return admission_promises;
228 }

230 void MutationBatcher::Admit(PendingSingleRowMutation mut) {
231  outstanding_size_ += mut.request_size;
232  cur_batch_->requests_size += mut.request_size;
233  cur_batch_->num_mutations += mut.num_mutations;
234  cur_batch_->requests.emplace_back(std::move(mut.mut));
235  cur_batch_->mutation_data.emplace_back(MutationData(std::move(mut)));
236 }

238 void MutationBatcher::SatisfyPromises(
239  std::vector<AdmissionPromise> admission_promises,
240  std::unique_lock<std::mutex>& lk) {
241  std::vector<NoMorePendingPromise> no_more_pending_promises;
242  if (num_requests_pending_ == 0 && num_outstanding_batches_ == 0) {
243  // We should wait not only on num_requests_pending_ being zero but also on
244  // num_outstanding_batches_ because we want to allow the user to kill the
245  // completion queue after this promise is fulfilled. Otherwise, the user can
246  // destroy the completion queue while the last batch is still being
247  // processed - we've had this bug (#2140).
248  no_more_pending_promises_.swap(no_more_pending_promises);
249  }
250  lk.unlock();
251 
252  // Inform the user that we've admitted these mutations and there might be some
253  // space in the buffer finally.
254  for (auto& promise : admission_promises) {
255  promise.set_value();
256  }
257  for (auto& promise : no_more_pending_promises) {
258  promise.set_value();
259  }
260 }
