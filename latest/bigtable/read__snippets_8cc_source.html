33 void PrintRow(google::cloud::bigtable::Row const& row) {
34  std::cout << "Reading data for " << row.row_key() << "\n";
35  std::string col_family;
36  for (auto const& cell : row.cells()) {
37  if (col_family != cell.family_name()) {
38  col_family = cell.family_name();
39  std::cout << "Column Family " << col_family << "\n";
40  }
41  std::cout << "\t" << cell.column_qualifier() << ": " << cell.value() << "@"
42  << google::cloud::internal::FormatRfc3339(
43  std::chrono::system_clock::time_point(cell.timestamp()))
44  << "\n";
45  }
46 }

109 void ReadRowsWithLimit(google::cloud::bigtable::Table table,
110  std::vector<std::string> const&) {
112  namespace cbt = google::cloud::bigtable;
113  using google::cloud::StatusOr;
114  [](cbt::Table table) {
115  // Create the range of rows to read.
116  auto range = cbt::RowRange::Range("phone#4c410523#20190501",
117  "phone#4c410523#20190502");
118  // Filter the results, only include values from the "connected_wifi" column
119  // in the "stats_summary" column family, and only get the latest value.
120  cbt::Filter filter = cbt::Filter::Chain(
121  cbt::Filter::ColumnRangeClosed("stats_summary", "connected_wifi",
122  "connected_wifi"),
123  cbt::Filter::Latest(1));
124  // Read and print the first 5 rows in the range.
125  for (StatusOr<cbt::Row> const& row : table.ReadRows(range, 5, filter)) {
126  if (!row) throw std::runtime_error(row.status().message());
127  if (row->cells().size() != 1) {
128  std::ostringstream os;
129  os << "Unexpected number of cells in " << row->row_key();
130  throw std::runtime_error(os.str());
131  }
132  auto const& cell = row->cells().at(0);
133  std::cout << cell.row_key() << " = [" << cell.value() << "]\n";
134  }
135  }
137  (std::move(table));
138 }

140 void ReadKeysSet(std::vector<std::string> argv) {
141  if (argv.size() < 4) {
142  throw Usage{
143  "read-keys-set <project-id> <instance-id> <table-id>"
144  " key1 [key2 ...]"};
145  }
146 
147  google::cloud::bigtable::Table table(
148  google::cloud::bigtable::CreateDefaultDataClient(
149  argv[0], argv[1], google::cloud::bigtable::ClientOptions()),
150  argv[2]);
151  argv.erase(argv.begin(), argv.begin() + 3);
152 
153  // [START bigtable_read_keys_set]
154  namespace cbt = google::cloud::bigtable;
155  using google::cloud::StatusOr;
156  [](cbt::Table table, std::vector<std::string> const& row_keys) {
157  auto row_set = cbt::RowSet();
158 
159  for (auto const& row_key : row_keys) {
160  row_set.Append(row_key);
161  }
162 
163  cbt::Filter filter = cbt::Filter::Latest(1);
164  for (StatusOr<cbt::Row>& row : table.ReadRows(std::move(row_set), filter)) {
165  if (!row) throw std::runtime_error(row.status().message());
166  std::cout << row->row_key() << ":\n";
167  for (auto const& cell : row->cells()) {
168  std::cout << "\t" << cell.family_name() << ":"
169  << cell.column_qualifier() << " @ "
170  << cell.timestamp().count() << "us\n"
171  << "\t\"" << cell.value() << '"' << "\n";
172  }
173  }
174  }
175  // [END bigtable_read_keys_set]
176  (std::move(table), argv);
177 }

338 void RunAll(std::vector<std::string> const& argv) {
339  namespace examples = ::google::cloud::bigtable::examples;
340  namespace cbt = google::cloud::bigtable;
341 
342  if (!argv.empty()) throw google::cloud::bigtable::examples::Usage{"auto"};
343  examples::CheckEnvironmentVariablesAreSet({
344  "GOOGLE_CLOUD_PROJECT",
345  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
346  });
347  auto const project_id =
348  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
349  auto const instance_id = google::cloud::internal::GetEnv(
350  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
351  .value();
352 
353  cbt::TableAdmin admin(
354  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
355  instance_id);
356 
357  // If a previous run of these samples crashes before cleaning up there may be
358  // old tables left over. As there are quotas on the total number of tables we
359  // remove stale tables after 48 hours.
360  google::cloud::bigtable::testing::CleanupStaleTables(admin);
361  google::cloud::bigtable::testing::CleanupStaleTables(admin);
362 
363  // Initialize a generator with some amount of entropy.
364  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
365 
366  auto table_id = google::cloud::bigtable::testing::RandomTableId(generator);
367  std::cout << "Creating table " << table_id << std::endl;
368  auto schema = admin.CreateTable(
369  table_id, cbt::TableConfig(
370  {{"stats_summary", cbt::GcRule::MaxNumVersions(10)}}, {}));
371  if (!schema) throw std::runtime_error(schema.status().message());
372 
373  google::cloud::bigtable::Table table(
374  google::cloud::bigtable::CreateDefaultDataClient(
375  admin.project(), admin.instance_id(),
376  google::cloud::bigtable::ClientOptions()),
377  table_id);
378 
379  std::cout << "Preparing data for read examples" << std::endl;
380  PrepareReadSamples(table);
381  std::cout << "Running ReadRow" << std::endl;
382  ReadRow(table, {"phone#4c410523#20190501"});
383  std::cout << "Running ReadRowPartial" << std::endl;
384  ReadRowPartial(table, {"phone#4c410523#20190501"});
385  std::cout << "Running ReadRows" << std::endl;
386  ReadRows(table, {});
387  std::cout << "Running ReadRowRange" << std::endl;
388  ReadRowRange(table, {});
389  std::cout << "Running ReadRowRanges" << std::endl;
390  ReadRowRanges(table, {});
391  std::cout << "Running ReadRowPrefix" << std::endl;
392  ReadRowPrefix(table, {});
393  std::cout << "Running ReadFilter" << std::endl;
394  ReadFilter(table, {});
395  std::cout << "Running ReadRowsWithLimit() example" << std::endl;
396  ReadRowsWithLimit(table, {});
397 
398  std::cout << "Running ReadKeySet() example" << std::endl;
399  ReadKeysSet({table.project_id(), table.instance_id(), table.table_id(),
400  "root/0/0/1", "root/0/1/0"});
401  std::cout << "Running ReadPrefixList() example" << std::endl;
402  ReadPrefixList(table, {"root/0/1/", "root/2/1/"});
403 
404  admin.DeleteTable(table_id);
405 }
