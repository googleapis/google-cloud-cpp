93 void BulkApply(google::cloud::bigtable::Table table,
94  std::vector<std::string> const&) {
96  namespace cbt = google::cloud::bigtable;
97  [](cbt::Table table) {
98  // Write several rows in a single operation, each row has some trivial data.
99  cbt::BulkMutation bulk;
100  for (int i = 0; i != 5000; ++i) {
101  // Note: This example uses sequential numeric IDs for simplicity, but
102  // this can result in poor performance in a production application.
103  // Since rows are stored in sorted order by key, sequential keys can
104  // result in poor distribution of operations across nodes.
105  //
106  // For more information about how to design a Bigtable schema for the
107  // best performance, see the documentation:
108  //
109  // https://cloud.google.com/bigtable/docs/schema-design
110  char buf[32];
111  snprintf(buf, sizeof(buf), "key-%06d", i);
112  cbt::SingleRowMutation mutation(buf);
113  mutation.emplace_back(
114  cbt::SetCell("fam", "col0", "value0-" + std::to_string(i)));
115  mutation.emplace_back(
116  cbt::SetCell("fam", "col1", "value1-" + std::to_string(i)));
117  bulk.emplace_back(std::move(mutation));
118  }
119  std::vector<cbt::FailedMutation> failures =
120  table.BulkApply(std::move(bulk));
121  if (failures.empty()) {
122  std::cout << "All mutations applied successfully\n";
123  return;
124  }
125  // By default, the `table` object uses the `SafeIdempotentMutationPolicy`
126  // which does not retry if any of the mutations fails and are
127  // not-idempotent. In this example we simply print such failures, if any,
128  // and ignore them otherwise.
129  std::cerr << "The following mutations failed and were not retried:\n";
130  for (auto const& f : failures) {
131  std::cerr << "index[" << f.original_index() << "]=" << f.status() << "\n";
132  }
133  }
135  (std::move(table));
136 }

138 void ReadRow(google::cloud::bigtable::Table table,
139  std::vector<std::string> const& argv) {
140  namespace cbt = google::cloud::bigtable;
141  using google::cloud::StatusOr;
142  [](google::cloud::bigtable::Table table, std::string const& row_key) {
143  // Filter the results, only include the latest value on each cell.
144  cbt::Filter filter = cbt::Filter::Latest(1);
145  // Read a row, this returns a tuple (bool, row)
146  StatusOr<std::pair<bool, cbt::Row>> tuple =
147  table.ReadRow(row_key, std::move(filter));
148  if (!tuple) throw std::runtime_error(tuple.status().message());
149  if (!tuple->first) {
150  std::cout << "Row " << row_key << " not found\n";
151  return;
152  }
153  std::cout << "key: " << tuple->second.row_key() << "\n";
154  for (auto const& cell : tuple->second.cells()) {
155  std::cout << " " << cell.family_name() << ":"
156  << cell.column_qualifier() << " = <";
157  if (cell.column_qualifier() == "counter") {
158  // This example uses "counter" to store 64-bit numbers in big-endian
159  // format, extract them as follows:
160  std::cout << cell.decode_big_endian_integer<std::int64_t>().value();
161  } else {
162  std::cout << cell.value();
163  }
164  std::cout << ">\n";
165  }
166  }(std::move(table), argv.at(0));
167 }

169 void CheckAndMutate(google::cloud::bigtable::Table table,
170  std::vector<std::string> const& argv) {
172  namespace cbt = google::cloud::bigtable;
173  using google::cloud::StatusOr;
174  [](cbt::Table table, std::string const& row_key) {
175  // Check if the latest value of the flip-flop column is "on".
176  cbt::Filter predicate = cbt::Filter::Chain(
177  cbt::Filter::ColumnRangeClosed("fam", "flip-flop", "flip-flop"),
178  cbt::Filter::Latest(1), cbt::Filter::ValueRegex("on"));
179  // If the predicate matches, change the latest value to "off", otherwise,
180  // change the latest value to "on". Modify the "flop-flip" column at the
181  // same time.
182  StatusOr<cbt::MutationBranch> branch =
183  table.CheckAndMutateRow(row_key, std::move(predicate),
184  {cbt::SetCell("fam", "flip-flop", "off"),
185  cbt::SetCell("fam", "flop-flip", "on")},
186  {cbt::SetCell("fam", "flip-flop", "on"),
187  cbt::SetCell("fam", "flop-flip", "off")});
188 
189  if (!branch) throw std::runtime_error(branch.status().message());
190  if (*branch == cbt::MutationBranch::kPredicateMatched) {
191  std::cout << "The predicate was matched\n";
192  } else {
193  std::cout << "The predicate was not matched\n";
194  }
195  }
197  (std::move(table), argv.at(0));
198 }

429 void MutateInsertUpdateRows(google::cloud::bigtable::Table table,
430  std::vector<std::string> const& argv) {
431  // Fortunately region tags can appear more than once, the segments are merged
432  // by the region tag processing tools.
433 
434  // [START bigtable_insert_update_rows]
435  struct InsertOrUpdate {
436  std::string column_family;
437  std::string column;
438  std::string value;
439  };
440  // [END bigtable_insert_update_rows]
441 
442  // A simple, though probably not very efficient, parser for mutations.
443  auto parse = [](std::string const& mut) {
444  std::vector<std::string> const tokens =
445  absl::StrSplit(mut, absl::ByAnyChar(":="));
446  return InsertOrUpdate{tokens[0], tokens[1], tokens[2]};
447  };
448 
449  auto it = argv.cbegin();
450  auto const key = *it++;
451  std::vector<InsertOrUpdate> mutations;
452  for (; it != argv.cend(); ++it) {
453  mutations.emplace_back(parse(*it));
454  }

458  [](cbt::Table table, std::string const& key,
459  std::vector<InsertOrUpdate> const& inserts) {
460  cbt::SingleRowMutation mutation(key);
461  for (auto const& mut : inserts) {
462  mutation.emplace_back(
463  cbt::SetCell(mut.column_family, mut.column, mut.value));
464  }

494 void RenameColumn(google::cloud::bigtable::Table table,
495  std::vector<std::string> const& argv) {
496  // [START bigtable_mutate_mix_match]
497  namespace cbt = google::cloud::bigtable;
498  using google::cloud::Status;
499  using google::cloud::StatusOr;
500  [](cbt::Table table, std::string const& key, std::string const& family,
501  std::string const& old_name, std::string const& new_name) {
502  StatusOr<std::pair<bool, cbt::Row>> row =
503  table.ReadRow(key, cbt::Filter::ColumnName(family, old_name));
504 
505  if (!row) throw std::runtime_error(row.status().message());
506  if (!row->first) throw std::runtime_error("Cannot find row " + key);
507 
508  cbt::SingleRowMutation mutation(key);
509  for (auto const& cell : row->second.cells()) {
510  // Create a new cell
511  auto timestamp_in_milliseconds =
512  std::chrono::duration_cast<std::chrono::milliseconds>(
513  cell.timestamp());
514  mutation.emplace_back(cbt::SetCell(family, new_name,
515  timestamp_in_milliseconds,
516  std::move(cell).value()));
517  }
518  mutation.emplace_back(cbt::DeleteFromColumn("fam", old_name));
519 
520  google::cloud::Status status = table.Apply(std::move(mutation));
521  if (!status.ok()) throw std::runtime_error(status.message());
522  std::cout << "Row successfully updated\n";
523  }
524  // [END bigtable_mutate_mix_match]
525  (std::move(table), argv.at(0), argv.at(1), argv.at(2), argv.at(3));
526 }
