32 void FilterLimitRowSample(google::cloud::bigtable::Table table,
33  std::vector<std::string> const&) {
35  namespace cbt = google::cloud::bigtable;
36  using google::cloud::StatusOr;
37  [](cbt::Table table) {
38  // Filter the results, only include rows with a given probability
39  cbt::Filter filter = cbt::Filter::RowSample(0.75);
40 
41  // Read and print the rows.
42  for (StatusOr<cbt::Row> const& row :
43  table.ReadRows(cbt::RowSet(cbt::RowRange::InfiniteRange()), filter)) {
44  if (!row) throw std::runtime_error(row.status().message());
45  std::cout << row->row_key() << " = ";
46  for (auto const& cell : row->cells()) {
47  std::cout << "[" << cell.family_name() << ", "
48  << cell.column_qualifier() << ", " << cell.value() << "],";
49  }
50  std::cout << "\n";
51  }
52  }
54  (std::move(table));
55 }

487 // This command just generates data suitable for other examples to run. This

489 void InsertTestData(google::cloud::bigtable::Table table,
490  std::vector<std::string> const&) {
491  // Write several rows in a single operation, each row has some trivial data.
492  // This is not a code sample in the normal sense, we do not display this code
493  // in the documentation. We use it to populate data in the table used to run
494  // the actual examples during the CI builds.
495  namespace cbt = google::cloud::bigtable;
496  cbt::BulkMutation bulk;
497  for (int i = 0; i != 50; ++i) {
498  // Note: This example uses sequential numeric IDs for simplicity, but
499  // this can result in poor performance in a production application.
500  // Since rows are stored in sorted order by key, sequential keys can
501  // result in poor distribution of operations across nodes.
502  //
503  // For more information about how to design a Bigtable schema for the
504  // best performance, see the documentation:
505  //
506  // https://cloud.google.com/bigtable/docs/schema-design
507  char buf[32];
508  snprintf(buf, sizeof(buf), "key-%06d", i);
509  cbt::SingleRowMutation mutation(buf);
510  mutation.emplace_back(cbt::SetCell("fam-0", "col-a", milliseconds(1),
511  "value-0-" + std::to_string(i)));
512  mutation.emplace_back(cbt::SetCell("fam-0", "col-a", milliseconds(2),
513  "value-1-" + std::to_string(i)));
514  mutation.emplace_back(cbt::SetCell("fam-0", "col-b", milliseconds(1),
515  "value-0-" + std::to_string(i)));
516  mutation.emplace_back(cbt::SetCell("fam-0", "col-c", milliseconds(1),
517  "value-0-" + std::to_string(i)));
518  mutation.emplace_back(cbt::SetCell("fam-0", "col-c", milliseconds(2),
519  "value-1-" + std::to_string(i)));
520  mutation.emplace_back(cbt::SetCell("fam-0", "col-c", milliseconds(3),
521  "value-2-" + std::to_string(i)));
522  mutation.emplace_back(cbt::SetCell("fam-1", "col-a", milliseconds(1),
523  "value-0-" + std::to_string(i)));
524  bulk.emplace_back(std::move(mutation));
525  }
526  auto failures = table.BulkApply(std::move(bulk));
527  if (failures.empty()) {
528  return;
529  }
530  std::cerr << "The following mutations failed:\n";
531  for (auto const& f : failures) {
532  std::cerr << "index[" << f.original_index() << "]=" << f.status() << "\n";
533  }
534  throw std::runtime_error(failures.front().status().message());
535 }

537 void RunAll(std::vector<std::string> const& argv) {
538  namespace examples = ::google::cloud::bigtable::examples;
539  namespace cbt = google::cloud::bigtable;
540 
541  if (!argv.empty()) throw google::cloud::bigtable::examples::Usage{"auto"};
542  examples::CheckEnvironmentVariablesAreSet({
543  "GOOGLE_CLOUD_PROJECT",
544  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
545  });
546  auto const project_id =
547  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
548  auto const instance_id = google::cloud::internal::GetEnv(
549  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
550  .value();
551 
552  cbt::TableAdmin admin(
553  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
554  instance_id);
555 
556  // If a previous run of these samples crashes before cleaning up there may be
557  // old tables left over. As there are quotas on the total number of tables we
558  // remove stale tables after 48 hours.
559  google::cloud::bigtable::testing::CleanupStaleTables(admin);
560 
561  // Initialize a generator with some amount of entropy.
562  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
563 
564  auto table_id = google::cloud::bigtable::testing::RandomTableId(generator);
565  auto schema = admin.CreateTable(
566  table_id, cbt::TableConfig({{"fam-0", cbt::GcRule::MaxNumVersions(10)},
567  {"fam-1", cbt::GcRule::MaxNumVersions(10)}},
568  {}));
569  if (!schema) throw std::runtime_error(schema.status().message());
570 
571  google::cloud::bigtable::Table table(
572  google::cloud::bigtable::CreateDefaultDataClient(
573  admin.project(), admin.instance_id(),
574  google::cloud::bigtable::ClientOptions()),
575  table_id);
576 
577  std::cout << "\nPreparing data for multiple examples" << std::endl;
578  InsertTestData(table, {});
579  std::cout << "Running FilterLimitRowSample() example [1]" << std::endl;
580  FilterLimitRowSample(table, {});
581  std::cout << "Running FilterLimitRowRegex() example [2]" << std::endl;
582  FilterLimitRowRegex(table, {});
