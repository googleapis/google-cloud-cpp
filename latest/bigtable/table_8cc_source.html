41 template <typename Response>
42 Row TransformReadModifyWriteRowResponse(Response& response) {
43  std::vector<bigtable::Cell> cells;
44  auto& row = *response.mutable_row();
45  for (auto& family : *row.mutable_families()) {
46  for (auto& column : *family.mutable_columns()) {
47  for (auto& cell : *column.mutable_cells()) {
48  std::vector<std::string> labels;
49  std::move(cell.mutable_labels()->begin(), cell.mutable_labels()->end(),
50  std::back_inserter(labels));
51  bigtable::Cell new_cell(row.key(), family.name(), column.qualifier(),
52  cell.timestamp_micros(),
53  std::move(*cell.mutable_value()),
54  std::move(labels));
55 
56  cells.emplace_back(std::move(new_cell));
57  }
58  }
59  }
60 
61  return Row(std::move(*row.mutable_key()), std::move(cells));
62 }

71 Status Table::Apply(SingleRowMutation mut) {
72  // Copy the policies in effect for this operation. Many policy classes change
73  // their state as the operation makes progress (or fails to make progress), so
74  // we need fresh instances.
75  auto rpc_policy = clone_rpc_retry_policy();
76  auto backoff_policy = clone_rpc_backoff_policy();
77  auto idempotent_policy = clone_idempotent_mutation_policy();
78 
79  // Build the RPC request, try to minimize copying.
80  btproto::MutateRowRequest request;
81  SetCommonTableOperationRequest<btproto::MutateRowRequest>(
82  request, app_profile_id_, table_name_);
83  mut.MoveTo(request);
84 
85  bool const is_idempotent =
86  std::all_of(request.mutations().begin(), request.mutations().end(),
87  [&idempotent_policy](btproto::Mutation const& m) {
88  return idempotent_policy->is_idempotent(m);
89  });
90 
91  btproto::MutateRowResponse response;
92  grpc::Status status;
93  while (true) {
94  grpc::ClientContext client_context;
95  rpc_policy->Setup(client_context);
96  backoff_policy->Setup(client_context);
97  metadata_update_policy_.Setup(client_context);
98  status = client_->MutateRow(&client_context, request, &response);
99 
100  if (status.ok()) {
101  return google::cloud::Status{};
102  }
103  // It is up to the policy to terminate this loop, it could run
104  // forever, but that would be a bad policy (pun intended).
105  if (!rpc_policy->OnFailure(status) || !is_idempotent) {
106  return MakeStatusFromRpcError(status);
107  }
108  auto delay = backoff_policy->OnCompletion(status);
109  std::this_thread::sleep_for(delay);
110  }
111 }

114  CompletionQueue& cq) {

116  SetCommonTableOperationRequest<google::bigtable::v2::MutateRowRequest>(

119  auto context = absl::make_unique<grpc::ClientContext>();

121  // Determine if all the mutations are idempotent. The idempotency of the
122  // mutations won't change as the retry loop executes, so we can just compute
123  // it once and use a constant value for the loop.
124  auto idempotent_mutation_policy = clone_idempotent_mutation_policy();
125  auto const is_idempotent = std::all_of(
126  request.mutations().begin(), request.mutations().end(),

128  return idempotent_mutation_policy->is_idempotent(m);

130 
131  auto const idempotency =

224 StatusOr<std::pair<bool, Row>> Table::ReadRow(std::string row_key,
225  Filter filter) {
226  RowSet row_set(std::move(row_key));
227  std::int64_t const rows_limit = 1;
228  RowReader reader =
229  ReadRows(std::move(row_set), rows_limit, std::move(filter));
230 
231  auto it = reader.begin();
232  if (it == reader.end()) {
233  return std::make_pair(false, Row("", {}));
234  }
235  if (!*it) {
236  return it->status();
237  }
238  auto result = std::make_pair(true, std::move(**it));
239  if (++it != reader.end()) {
240  return Status(StatusCode::kInternal,
241  "internal error - RowReader returned 2 rows in ReadRow()");
242  }
243  return result;
244 }

336 // Call the `google.bigtable.v2.Bigtable.SampleRowKeys` RPC until
337 // successful. When RPC is finished, this function returns the SampleRowKeys
338 // as a std::vector<>. If the RPC fails, it will keep retrying until the
339 // policies in effect tell us to stop. Note that each retry must clear the
340 // samples otherwise the result is an inconsistent set of samples row keys.
341 StatusOr<std::vector<bigtable::RowKeySample>> Table::SampleRows() {
342  // Copy the policies in effect for this operation.
343  auto backoff_policy = clone_rpc_backoff_policy();
344  auto retry_policy = clone_rpc_retry_policy();
345  std::vector<bigtable::RowKeySample> samples;
346 
347  // Build the RPC request for SampleRowKeys
348  btproto::SampleRowKeysRequest request;
349  btproto::SampleRowKeysResponse response;
350  SetCommonTableOperationRequest<btproto::SampleRowKeysRequest>(
351  request, app_profile_id_, table_name_);
352 
353  while (true) {
354  grpc::ClientContext client_context;
355  backoff_policy->Setup(client_context);
356  retry_policy->Setup(client_context);
357  clone_metadata_update_policy().Setup(client_context);
358 
359  auto stream = client_->SampleRowKeys(&client_context, request);
360  while (stream->Read(&response)) {
361  bigtable::RowKeySample row_sample;
362  row_sample.offset_bytes = response.offset_bytes();
363  row_sample.row_key = std::move(*response.mutable_row_key());
364  samples.emplace_back(std::move(row_sample));
365  }
366  auto status = stream->Finish();
367  if (status.ok()) {
368  break;
369  }
370  if (!retry_policy->OnFailure(status)) {
371  return MakeStatusFromRpcError(
372  status.error_code(),
373  "Retry policy exhausted: " + status.error_message());
374  }
375  samples.clear();
376  auto delay = backoff_policy->OnCompletion(status);
377  std::this_thread::sleep_for(delay);
378  }
379  return samples;
380 }

431 future<StatusOr<std::pair<bool, Row>>> Table::AsyncReadRowImpl(
432  CompletionQueue& cq, std::string row_key, Filter filter) {
433  class AsyncReadRowHandler {
434  public:
435  AsyncReadRowHandler() : row_("", {}) {}
436 
437  future<StatusOr<std::pair<bool, Row>>> GetFuture() {
438  return row_promise_.get_future();
439  }
440 
441  future<bool> OnRow(Row row) {
442  // assert(!row_received_);
443  row_ = std::move(row);
444  row_received_ = true;
445  // Don't satisfy the promise before `OnStreamFinished`.
446  //
447  // The `CompletionQueue`, which this object holds a reference to, should
448  // not be shut down before `OnStreamFinished` is called. In order to make
449  // sure of that, satisying the `promise<>` is deferred until then - the
450  // user shouldn't shutown the `CompleetionQue` before this whole
451  // operations is done.
452  return make_ready_future(false);
453  }
454 
455  void OnStreamFinished(Status status) {
456  if (row_received_) {
457  // If we got a row we don't need to care about the stream status.
458  row_promise_.set_value(std::make_pair(true, std::move(row_)));
459  return;
460  }
461  if (status.ok()) {
462  row_promise_.set_value(std::make_pair(false, Row("", {})));
463  } else {
464  row_promise_.set_value(std::move(status));
465  }
466  }
467 
468  private:
469  Row row_;
470  bool row_received_{};
471  promise<StatusOr<std::pair<bool, Row>>> row_promise_;
472  };
473 
474  RowSet row_set(std::move(row_key));
475  std::int64_t const rows_limit = 1;
476  auto handler = std::make_shared<AsyncReadRowHandler>();
477  AsyncReadRowsImpl(
478  cq, [handler](Row row) { return handler->OnRow(std::move(row)); },
479  [handler](Status status) {
480  handler->OnStreamFinished(std::move(status));
481  },
482  std::move(row_set), rows_limit, std::move(filter));
483  return handler->GetFuture();
484 }
