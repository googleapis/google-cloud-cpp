45 void SetIamPolicy(google::cloud::bigtable::TableAdmin const& admin,
46  std::vector<std::string> const& argv) {
48  namespace cbt = google::cloud::bigtable;
49  using google::cloud::StatusOr;
50  [](cbt::TableAdmin admin, std::string const& table_id,
51  std::string const& role, std::string const& member) {
52  StatusOr<google::iam::v1::Policy> current = admin.GetIamPolicy(table_id);
53  if (!current) throw std::runtime_error(current.status().message());
54  // This example adds the member to all existing bindings for that role. If
55  // there are no such bindgs, it adds a new one. This might not be what the
56  // user wants, e.g. in case of conditional bindings.
57  size_t num_added = 0;
58  for (auto& binding : *current->mutable_bindings()) {
59  if (binding.role() == role) {
60  binding.add_members(member);
61  ++num_added;
62  }
63  }
64  if (num_added == 0) {
65  *current->add_bindings() = cbt::IamBinding(role, {member});
66  }
67  StatusOr<google::iam::v1::Policy> policy =
68  admin.SetIamPolicy(table_id, *current);
69  if (!policy) throw std::runtime_error(policy.status().message());
70  std::cout << "The IAM Policy for " << table_id << " is\n"
71  << policy->DebugString() << "\n";
72  }
74  (std::move(admin), argv.at(0), argv.at(1), argv.at(2));
75 }

141  // If a previous run of these samples crashes before cleaning up there may be

144  std::cout << "\nCleaning up old tables" << std::endl;
145  google::cloud::bigtable::testing::CleanupStaleTables(admin);

152  auto table = admin.CreateTable(
