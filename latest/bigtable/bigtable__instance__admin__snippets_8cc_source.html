299  using google::cloud::StatusOr;
300  [](cbt::InstanceAdmin instance_admin, std::string const& instance_id,
301  std::string const& cluster_id) {
302  // GetCluster first and then modify it.
303  StatusOr<google::bigtable::admin::v2::Cluster> cluster =
304  instance_admin.GetCluster(instance_id, cluster_id);
305  if (!cluster) throw std::runtime_error(cluster.status().message());
306 
307  // The state cannot be sent on updates, so clear it first.
308  cluster->clear_state();
309  // Set the desired cluster configuration.

311  auto modified_config = cbt::ClusterConfig(std::move(*cluster));
312 
313  StatusOr<google::bigtable::admin::v2::Cluster> modified_cluster =
314  instance_admin.UpdateCluster(modified_config).get();
315  if (!modified_cluster) {
316  throw std::runtime_error(modified_cluster.status().message());
317  }
318  std::cout << "cluster details : " << cluster->DebugString() << "\n";
319  }
321  (std::move(instance_admin), argv.at(0), argv.at(1));

554 void DeleteAppProfile(std::vector<std::string> const& argv) {
555  std::string basic_usage =
556  "delete-app-profile <project-id> <instance-id> <profile-id>"
557  " [ignore-warnings (default: true)]";
558  if (argv.size() != 3 && argv.size() != 4) {
559  throw Usage{basic_usage};
560  }
561 
562  bool ignore_warnings = true;
563  if (argv.size() == 4) {
564  std::string arg = argv[3];
565  if (arg == "true") {
566  ignore_warnings = true;
567  } else if (arg == "false") {
568  ignore_warnings = false;
569  } else {
570  auto msg = basic_usage;
571  msg +=
572  "\ndelete-app-profile: ignore-warnings parameter must be either"
573  " 'true' or 'false'";
574  throw Usage{msg};
575  }

610 void SetNativeIamPolicy(google::cloud::bigtable::InstanceAdmin instance_admin,
611  std::vector<std::string> const& argv) {
613  namespace cbt = google::cloud::bigtable;
614  using google::cloud::StatusOr;
615  [](cbt::InstanceAdmin instance_admin, std::string const& instance_id,
616  std::string const& role, std::string const& member) {
617  StatusOr<google::iam::v1::Policy> current =
618  instance_admin.GetNativeIamPolicy(instance_id);
619  if (!current) throw std::runtime_error(current.status().message());
620  // This example adds the member to all existing bindings for that role. If
621  // there are no such bindgs, it adds a new one. This might not be what the
622  // user wants, e.g. in case of conditional bindings.
623  size_t num_added = 0;
624  for (auto& binding : *current->mutable_bindings()) {
625  if (binding.role() == role) {
626  binding.add_members(member);
627  ++num_added;
628  }
629  }
630  if (num_added == 0) {
631  *current->add_bindings() = cbt::IamBinding(role, {member});
632  }
633  StatusOr<google::iam::v1::Policy> policy =
634  instance_admin.SetIamPolicy(instance_id, *current);
635  if (!policy) throw std::runtime_error(policy.status().message());
636  std::cout << "The IAM Policy for " << instance_id << " is\n"
637  << policy->DebugString() << "\n";
638  }
640  (std::move(instance_admin), argv.at(0), argv.at(1), argv.at(2));
