134  StatusOr<storage::internal::HttpResponse> DoMetadataServerGetRequest(
135  std::string const& path, bool recursive) const {
136  // Allows mocking the metadata server hostname for testing.

140  HttpRequestBuilderType request_builder(

157  Status RetrieveServiceAccountInfo() const {
158  auto response = DoMetadataServerGetRequest(
159  "/computeMetadata/v1/instance/service-accounts/" +
160  service_account_email_ + "/",
161  true);
162  if (!response) {
163  return std::move(response).status();
164  }
165  if (response->status_code >= 300) {
166  return AsStatus(*response);
167  }
168 
169  auto metadata = ParseMetadataServerResponse(*response);
170  if (!metadata) {
171  return metadata.status();
172  }
173  service_account_email_ = std::move(metadata->email);
174  scopes_ = std::move(metadata->scopes);
175  return Status();
176  }

178  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() const {
179  auto status = RetrieveServiceAccountInfo();
180  if (!status.ok()) {
181  return status;
182  }
183 
184  auto response = DoMetadataServerGetRequest(
185  "/computeMetadata/v1/instance/service-accounts/" +
186  service_account_email_ + "/token",
187  false);
188  if (!response) {
189  return std::move(response).status();
190  }
191  if (response->status_code >= 300) {
192  return AsStatus(*response);
193  }
194 
195  return ParseComputeEngineRefreshResponse(*response, clock_.now());
196  }
