261  // Create a client to communicate with Google Cloud Storage.
262  google::cloud::StatusOr<google::cloud::storage::Client> client =
263  google::cloud::storage::Client::CreateDefaultClient();

270  // Build the list of commands and the usage string from that list.
271  using CommandType =
272  std::function<void(google::cloud::storage::Client, int&, char* [])>;
273  std::map<std::string, CommandType> commands = {
274  {"list-default-object-acl", &ListDefaultObjectAcl},
275  {"create-default-object-acl", &CreateDefaultObjectAcl},
276  {"delete-default-object-acl", &DeleteDefaultObjectAcl},
277  {"get-default-object-acl", &GetDefaultObjectAcl},
278  {"update-default-object-acl", &UpdateDefaultObjectAcl},
279  {"patch-default-object-acl", &PatchDefaultObjectAcl},
280  {"patch-default-object-acl-no-read", &PatchDefaultObjectAclNoRead},
281  };
282  for (auto&& kv : commands) {
283  try {
284  int fake_argc = 1;
285  kv.second(*client, fake_argc, argv);
286  } catch (Usage const& u) {
287  command_usage += " ";
288  command_usage += u.msg;
289  command_usage += "\n";
290  }
291  }
292 
293  if (argc < 2) {
294  PrintUsage(argc, argv, "Missing command");
295  return 1;
296  }
297 
298  std::string const command = ConsumeArg(argc, argv);
299  auto it = commands.find(command);
300  if (commands.end() == it) {
301  PrintUsage(argc, argv, "Unknown command: " + command);
302  return 1;
303  }
304 
305  // Call the command with that client.
306  it->second(*client, argc, argv);
307 
308  return 0;
309 } catch (Usage const& ex) {
310  PrintUsage(argc, argv, ex.msg);
311  return 1;
312 } catch (std::exception const& ex) {
313  std::cerr << "Standard C++ exception raised: " << ex.what() << "\n";
314  return 1;
