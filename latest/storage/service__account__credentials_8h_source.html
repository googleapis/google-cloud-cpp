134 template <typename HttpRequestBuilderType =
135  storage::internal::CurlRequestBuilder,
136  typename ClockType = std::chrono::system_clock>
137 class ServiceAccountCredentials : public Credentials {
138  public:
139  // NOLINTNEXTLINE(performance-unnecessary-value-param) TODO(#4112)
140  explicit ServiceAccountCredentials(ServiceAccountCredentialsInfo info)
141  : ServiceAccountCredentials(std::move(info), {}) {}
142  ServiceAccountCredentials(ServiceAccountCredentialsInfo info,
143  ChannelOptions const& options)
144  : info_(std::move(info)), clock_() {
145  HttpRequestBuilderType request_builder(
146  info_.token_uri,
147  storage::internal::GetDefaultCurlHandleFactory(options));
148  request_builder.AddHeader(
149  "Content-Type: application/x-www-form-urlencoded");
150  // This is the value of grant_type for JSON-formatted service account
151  // keyfiles downloaded from Cloud Console.
152  std::string grant_type("grant_type=");
153  grant_type +=
154  request_builder
155  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
156  .get();
157  grant_type_ = std::move(grant_type);
158  request_ = request_builder.BuildRequest();
159  }
160 
161  StatusOr<std::string> AuthorizationHeader() override {
162  std::unique_lock<std::mutex> lock(mu_);
163  return refreshing_creds_.AuthorizationHeader(clock_.now(),
164  [this] { return Refresh(); });
165  }
166 
178  StatusOr<std::vector<std::uint8_t>> SignBlob(
179  SigningAccount const& signing_account,
180  std::string const& blob) const override {
181  if (signing_account.has_value() &&
182  signing_account.value() != info_.client_email) {
183  return Status(StatusCode::kInvalidArgument,
184  "The current_credentials cannot sign blobs for " +
185  signing_account.value());
186  }
187  return internal::SignStringWithPem(blob, info_.private_key,
188  JwtSigningAlgorithms::RS256);
189  }
190 
191  std::string AccountEmail() const override { return info_.client_email; }
192  std::string KeyId() const override { return info_.private_key_id; }
193 
194  private:
195  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() {
196  auto payload =
197  CreateServiceAccountRefreshPayload(info_, grant_type_, clock_.now());
198 
199  auto response = request_.MakeRequest(payload);
200  if (!response) {
201  return std::move(response).status();
202  }
203  if (response->status_code >= 300) {
204  return AsStatus(*response);
205  }
206 
207  return ParseServiceAccountRefreshResponse(*response, clock_.now());
208  }
209 
210  typename HttpRequestBuilderType::RequestType request_;
211  std::string grant_type_;
212  ServiceAccountCredentialsInfo info_;
213  mutable std::mutex mu_;
214  RefreshingCredentialsWrapper refreshing_creds_;
215  ClockType clock_;
216 };
