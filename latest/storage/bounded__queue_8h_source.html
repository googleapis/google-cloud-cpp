27 template <typename T>
28 class BoundedQueue {
29  public:
30  BoundedQueue() : BoundedQueue(512, 1024) {}
31  explicit BoundedQueue(std::size_t lwm, std::size_t hwm)
32  : lwm_(lwm), hwm_(hwm) {}
33 
34  void Shutdown() {
35  std::unique_lock<std::mutex> lk(mu_);
36  is_shutdown_ = true;
37  lk.unlock();
38  cv_read_.notify_all();
39  cv_write_.notify_all();
40  }
41 
42  absl::optional<T> Pop() {
43  std::unique_lock<std::mutex> lk(mu_);
44  cv_read_.wait(lk, [this]() { return is_shutdown_ || !empty(); });
45  if (empty()) return {};
46  auto next = std::move(buffer_.front());
47  buffer_.pop_front();
48  if (below_lwm()) {
49  cv_write_.notify_all();
50  }
51  return next;
52  }
53 
54  void Push(T data) {
55  std::unique_lock<std::mutex> lk(mu_);
56  cv_write_.wait(lk, [this]() { return is_shutdown_ || below_hwm(); });
57  if (is_shutdown_) return;
58  buffer_.push_back(std::move(data));
59  cv_read_.notify_all();
60  }
61 
62  private:
63  bool empty() const { return buffer_.empty(); }
64  bool below_hwm() const { return buffer_.size() <= hwm_; }
65  bool below_lwm() const { return buffer_.size() <= lwm_; }
66 
67  std::size_t const lwm_;
68  std::size_t const hwm_;
69  std::mutex mu_;
70  std::condition_variable cv_read_;
71  std::condition_variable cv_write_;
72  std::deque<T> buffer_;
73  bool is_shutdown_ = false;
74 };
