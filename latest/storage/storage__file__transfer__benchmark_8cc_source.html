31 char const kDescription[] = R"""(
32 A throughput benchmark for the Google Cloud Storage C++ client library.
33 
34 This program benchmarks the Google Cloud Storage (GCS) C++ client library when
35 used to upload and download files. The program creates a file of a prescribed
36 size, and then repeatedly uploads that file to a GCS object, and then downloads
37 the GCS object to a separate file. The program reports the time taken to perform
38 the each operation, as well as the effective bandwidth (in Gbps and MiB/s). The
39 program deletes the target GCS object after each iteration.
40 
41 To perform this benchmark the program creates a new standard bucket, in a region
42 configured via the command line. Other test parameters, such as the project id,
43 the file size, and the buffer sizes are configurable via the command line too.
44 
45 The bucket name, the local file names, and the object names are all randomly
46 generated, so multiple instances of the program can run simultaneously. The
47 output of this program is an annotated CSV file, that can be analyzed by an
48 external script. The annotation lines start with a '#', analysis scripts should
49 skip these lines.
50 )""";

65 int main(int argc, char* argv[]) {
66  google::cloud::StatusOr<Options> options = ParseArgs(argc, argv);
67  if (!options) {
68  std::cerr << options.status() << "\n";
69  return 1;
70  }
71 
72  google::cloud::StatusOr<gcs::ClientOptions> client_options =
73  gcs::ClientOptions::CreateDefaultClientOptions();
74  if (!client_options) {
75  std::cerr << "Could not create ClientOptions, status="
76  << client_options.status() << "\n";
77  return 1;
78  }
79  client_options->SetUploadBufferSize(options->upload_buffer_size);
80  client_options->SetDownloadBufferSize(options->download_buffer_size);
81  client_options->set_project_id(options->project_id);
82 
83  gcs::Client client(*std::move(client_options));
84 
85  std::cout << "# Cleaning up stale benchmark buckets\n";
86  google::cloud::storage::testing::RemoveStaleBuckets(
87  client, gcs_bm::RandomBucketPrefix(),
88  std::chrono::system_clock::now() - std::chrono::hours(48));
89 
90  google::cloud::internal::DefaultPRNG generator =
91  google::cloud::internal::MakeDefaultPRNG();
92 
93  auto bucket_name = gcs_bm::MakeRandomBucketName(generator);
94  auto meta =
95  client
96  .CreateBucket(bucket_name,
97  gcs::BucketMetadata()
98  .set_storage_class(gcs::storage_class::Standard())
99  .set_location(options->region),
100  gcs::PredefinedAcl("private"),
101  gcs::PredefinedDefaultObjectAcl("projectPrivate"),
102  gcs::Projection("full"))
103  .value();
104  std::cout << "# Running test on bucket: " << meta.name() << "\n";
105  std::string notes = google::cloud::storage::version_string() + ";" +
106  google::cloud::internal::compiler() + ";" +
107  google::cloud::internal::compiler_flags();
108  std::transform(notes.begin(), notes.end(), notes.begin(),
109  [](char c) { return c == '\n' ? ';' : c; });
110  std::cout << "# Start time: "
111  << google::cloud::internal::FormatRfc3339(
112  std::chrono::system_clock::now())
113  << "\n# Region: " << options->region
114  << "\n# Duration: " << options->duration.count() << "s"
115  << "\n# File Size: " << options->file_size
116  << "\n# File Size (MiB): " << options->file_size / gcs_bm::kMiB
117  << "\n# Download buffer size (KiB): "
118  << options->download_buffer_size / gcs_bm::kKiB
119  << "\n# Upload buffer size (KiB): "
120  << options->upload_buffer_size / gcs_bm::kKiB
121  << "\n# Build info: " << notes << "\n";
122 
123  std::cout << "# Creating file to upload ..." << std::flush;
124  auto filename = gcs_bm::MakeRandomFileName(generator);
125  {
126  auto const filler = gcs_bm::MakeRandomData(generator, 4 * gcs_bm::kMiB);
127  std::ofstream os(filename, std::ios::binary);
128  std::int64_t current_size = 0;
129  while (current_size < options->file_size) {
130  auto n = filler.size();
131  if (static_cast<std::int64_t>(n) > (options->file_size - current_size)) {
132  n = static_cast<std::size_t>(options->file_size - current_size);
133  }
134  os.write(filler.data(), n);
135  current_size += n;
136  }
137  }
138  std::cout << " DONE\n"
139  << "# File: " << filename << "\n";
140 
141  auto deadline = std::chrono::system_clock::now() + options->duration;
142  for (auto now = std::chrono::system_clock::now(); now < deadline;
143  now = std::chrono::system_clock::now()) {
144  auto object_name = gcs_bm::MakeRandomObjectName(generator);
145 
146  auto upload_start = std::chrono::steady_clock::now();
147  auto object_metadata =
148  client.UploadFile(filename, bucket_name, object_name);
149  auto upload_elapsed = std::chrono::duration_cast<std::chrono::nanoseconds>(
150  std::chrono::steady_clock::now() - upload_start);
151  // Convert to Gbps
152  auto to_gbps = [](std::int64_t size, std::chrono::nanoseconds elapsed) {
153  return 8.0 * static_cast<double>(size) /
154  static_cast<double>(elapsed.count());
155  };
156  // Convert to MiB/s
157  auto to_mibs = [](std::int64_t size, std::chrono::milliseconds elapsed) {
158  return (static_cast<double>(size) / gcs_bm::kMiB) /
159  (static_cast<double>(elapsed.count()) / 1000.0);
160  };
161  double gbps = to_gbps(options->file_size, upload_elapsed);
162  auto ms =
163  std::chrono::duration_cast<std::chrono::milliseconds>(upload_elapsed);
164  auto mi_bs = to_mibs(options->file_size, ms);
165  std::cout << "FileUpload," << options->file_size << ','
166  << upload_elapsed.count() << ',' << gbps << ',' << ms.count()
167  << ',' << mi_bs << ',' << object_metadata.status().code() << "\n";
168  if (!object_metadata) {
169  std::cout << "# Error in FileUpload: " << object_metadata.status()
170  << "\n";
171  continue;
172  }
173 
174  auto destination_filename = gcs_bm::MakeRandomFileName(generator);
175  auto download_start = std::chrono::steady_clock::now();
176  client.DownloadToFile(object_metadata->bucket(), object_metadata->name(),
177  destination_filename);
178  auto download_elapsed =
179  std::chrono::duration_cast<std::chrono::nanoseconds>(
180  std::chrono::steady_clock::now() - download_start);
181  gbps = to_gbps(options->file_size, download_elapsed);
182  ms =
183  std::chrono::duration_cast<std::chrono::milliseconds>(download_elapsed);
184  mi_bs = to_mibs(options->file_size, ms);
185  std::cout << "FileDownload," << options->file_size << ','
186  << download_elapsed.count() << ',' << gbps << ',' << ms.count()
187  << ',' << mi_bs << ',' << object_metadata.status().code() << "\n";
188 
189  (void)client.DeleteObject(object_metadata->bucket(),
190  object_metadata->name(),
191  gcs::Generation(object_metadata->generation()));
192  std::remove(destination_filename.c_str());
193  }
194 
195  std::remove(filename.c_str());
196 
197  std::cout << "# Deleting " << bucket_name << "\n";
198  auto status = client.DeleteBucket(bucket_name);
199  if (!status.ok()) {
200  std::cerr << "# Error deleting bucket, status=" << status << "\n";
201  return 1;
202  }
203 
204  return 0;
205 }

211 google::cloud::StatusOr<Options> ParseArgsDefault(
212  std::vector<std::string> const& argv) {
213  Options options;
214 
215  bool wants_help = false;
216  bool wants_description = false;
217  std::vector<OptionDescriptor> descriptors{
218  {"--help", "print the usage message",
219  [&wants_help](std::string const&) { wants_help = true; }},
220  {"--description", "print a description of the benchmark",
221  [&wants_description](std::string const&) { wants_description = true; }},
222  {"--project-id", "the GCP project to create the bucket",
223  [&options](std::string const& val) { options.project_id = val; }},
224  {"--duration", "how long should the benchmark run (in seconds).",
225  [&options](std::string const& val) {
226  options.duration = gcs_bm::ParseDuration(val);
227  }},
228  {"--file-size", "the size of the file to upload",
229  [&options](std::string const& val) {
230  options.file_size = gcs_bm::ParseSize(val);
231  }},
232  {"--upload-buffer-size", "configure gcs::Client upload buffer size",
233  [&options](std::string const& val) {
234  options.upload_buffer_size = gcs_bm::ParseBufferSize(val);
235  }},
236  {"--download-buffer-size", "configure gcs::Client download buffer size",
237  [&options](std::string const& val) {
238  options.download_buffer_size = gcs_bm::ParseBufferSize(val);
239  }},
240  {"--region", "The GCS region used for the benchmark",
241  [&options](std::string const& val) { options.region = val; }},
242  };
243  auto usage = BuildUsage(descriptors, argv[0]);
244 
245  auto unparsed = OptionsParse(descriptors, argv);
246  if (wants_help) {
247  std::cout << usage << "\n";
248  }
249 
250  if (wants_description) {
251  std::cout << kDescription << "\n";
252  }
253 
254  if (unparsed.size() > 2) {
255  std::ostringstream os;
256  os << "Unknown arguments or options\n" << usage << "\n";
257  return google::cloud::Status{google::cloud::StatusCode::kInvalidArgument,
258  std::move(os).str()};
259  }
260  if (unparsed.size() == 2) {
261  options.region = unparsed[1];
262  }
263  if (options.region.empty()) {
264  std::ostringstream os;
265  os << "Missing value for --region option" << usage << "\n";
266  return google::cloud::Status{google::cloud::StatusCode::kInvalidArgument,
267  std::move(os).str()};
268  }
269 
270  return options;
271 }
