122  namespace gcs = google::cloud::storage;
123  using ::google::cloud::StatusOr;
124  [](gcs::Client client, std::string const& bucket_name,
125  std::string const& entity, std::string const& role) {
126  StatusOr<gcs::BucketAccessControl> original_acl =
127  client.GetBucketAcl(bucket_name, entity);
128 
129  if (!original_acl) {
130  throw std::runtime_error(original_acl.status().message());
131  }
132 
133  auto new_acl = *original_acl;
134  new_acl.set_role(role);
135 
136  StatusOr<gcs::BucketAccessControl> patched_acl =
137  client.PatchBucketAcl(bucket_name, entity, *original_acl, new_acl);

139  if (!patched_acl) throw std::runtime_error(patched_acl.status().message());
140  std::cout << "ACL entry for " << patched_acl->entity() << " in bucket "
141  << patched_acl->bucket() << " is now " << *patched_acl << "\n";
142  }

186 void RemoveBucketOwner(google::cloud::storage::Client client,
187  std::vector<std::string> const& argv) {
189  namespace gcs = google::cloud::storage;
190  using ::google::cloud::StatusOr;
191  [](gcs::Client client, std::string const& bucket_name,
192  std::string const& entity) {
193  StatusOr<gcs::BucketMetadata> original_metadata =
194  client.GetBucketMetadata(bucket_name, gcs::Projection::Full());
195 
196  if (!original_metadata) {
197  throw std::runtime_error(original_metadata.status().message());
198  }
199 
200  std::vector<gcs::BucketAccessControl> original_acl =
201  original_metadata->acl();
202  auto it = std::find_if(original_acl.begin(), original_acl.end(),
203  [entity](const gcs::BucketAccessControl& entry) {
204  return entry.entity() == entity &&
205  entry.role() ==
206  gcs::BucketAccessControl::ROLE_OWNER();
207  });
208 
209  if (it == original_acl.end()) {
210  std::cout << "Could not find entity " << entity
211  << " with role OWNER in bucket " << bucket_name << "\n";
212  return;
213  }
214 
215  gcs::BucketAccessControl owner = *it;
216  google::cloud::Status status =
217  client.DeleteBucketAcl(bucket_name, owner.entity());
218 
219  if (!status.ok()) throw std::runtime_error(status.message());
220  std::cout << "Deleted ACL entry for " << owner.entity() << " in bucket "
221  << bucket_name << "\n";
222  }
224  (std::move(client), argv.at(0), argv.at(1));
225 }

227 void RunAll(std::vector<std::string> const& argv) {
228  namespace examples = ::google::cloud::storage::examples;
229  namespace gcs = ::google::cloud::storage;
230 
231  if (!argv.empty()) throw examples::Usage{"auto"};
232  examples::CheckEnvironmentVariablesAreSet({
233  "GOOGLE_CLOUD_PROJECT",
234  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT",
235  });
236  auto const project_id =
237  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
238  auto const service_account =
239  google::cloud::internal::GetEnv(
240  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT")
241  .value();
242  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
243  auto const bucket_name = examples::MakeRandomBucketName(generator);
244  auto const entity = "user-" + service_account;
245  auto client = gcs::Client::CreateDefaultClient().value();
246  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
247  << std::endl;
248  (void)client
249  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
250  .value();
251  // In GCS a single project cannot create or delete buckets more often than
252  // once every two seconds. We will pause until that time before deleting the
253  // bucket.
254  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
255 
256  auto const reader = gcs::BucketAccessControl::ROLE_READER();
257  auto const owner = gcs::BucketAccessControl::ROLE_OWNER();
258 
259  std::cout << "\nRunning ListBucketAcl() example" << std::endl;
260  ListBucketAcl(client, {bucket_name});
261 
262  std::cout << "\nRunning CreateBucketAcl() example" << std::endl;
263  CreateBucketAcl(client, {bucket_name, entity, reader});
264 
265  std::cout << "\nRunning GetBucketAcl() example" << std::endl;
266  GetBucketAcl(client, {bucket_name, entity});
267 
268  std::cout << "\nRunning UpdateBucketAcl() example" << std::endl;
269  UpdateBucketAcl(client, {bucket_name, entity, owner});
270 
271  std::cout << "\nRunning PatchBucketAcl() example" << std::endl;
272  PatchBucketAcl(client, {bucket_name, entity, reader});
273 
274  std::cout << "\nRunning PatchBucketAcl() example" << std::endl;
275  PatchBucketAclNoRead(client, {bucket_name, entity, owner});
276 
277  std::cout << "\nRunning DeleteBucketAcl() example" << std::endl;
278  DeleteBucketAcl(client, {bucket_name, entity});
279 
280  std::cout << "\nRunning AddBucketOwner() example" << std::endl;
281  AddBucketOwner(client, {bucket_name, entity});
282 
283  std::cout << "\nRunning RemoveBucketOwner() example" << std::endl;
284  RemoveBucketOwner(client, {bucket_name, entity});
285 
286  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
287  (void)examples::RemoveBucketAndContents(client, bucket_name);
288 }
