167  /// Delete a metadata entry. This is a no-op if the key does not exist.
168  ObjectMetadata& delete_metadata(std::string const& key) {
169  auto i = metadata_.find(key);
170  if (i == metadata_.end()) {
171  return *this;
172  }
173  metadata_.erase(i);
174  return *this;
175  }

177  /// Insert or update the metadata entry.
178  ObjectMetadata& upsert_metadata(std::string key, std::string value) {
179  auto i = metadata_.lower_bound(key);
180  if (i == metadata_.end() || i->first != key) {
181  metadata_.emplace_hint(i, std::move(key), std::move(value));
182  } else {
183  i->second = std::move(value);
184  }
185  return *this;
186  }
