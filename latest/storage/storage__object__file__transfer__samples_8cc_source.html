25 void UploadFile(google::cloud::storage::Client client,
26  std::vector<std::string> const& argv) {
28  namespace gcs = google::cloud::storage;
29  using ::google::cloud::StatusOr;
30  [](gcs::Client client, std::string const& file_name,
31  std::string const& bucket_name, std::string const& object_name) {
32  // Note that the client library automatically computes a hash on the
33  // client-side to verify data integrity during transmission.
34  StatusOr<gcs::ObjectMetadata> metadata = client.UploadFile(
35  file_name, bucket_name, object_name, gcs::IfGenerationMatch(0));
36  if (!metadata) throw std::runtime_error(metadata.status().message());
37 
38  std::cout << "Uploaded " << file_name << " to object " << metadata->name()
39  << " in bucket " << metadata->bucket()
40  << "\nFull metadata: " << *metadata << "\n";
41  }
43  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
44 }
