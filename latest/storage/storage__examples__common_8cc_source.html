44 Commands::value_type CreateCommandEntry(
45  std::string const& name, std::vector<std::string> const& arg_names,
46  ClientCommand const& command) {
47  bool allow_varargs =
48  !arg_names.empty() && arg_names.back().find("...") != std::string::npos;
49  auto adapter = [=](std::vector<std::string> const& argv) {
50  if ((argv.size() == 1 && argv[0] == "--help") ||
51  (allow_varargs ? argv.size() < (arg_names.size() - 1)
52  : argv.size() != arg_names.size())) {
53  std::ostringstream os;
54  os << name;
55  for (auto const& a : arg_names) {
56  os << " " << a;
57  }
58  throw Usage{std::move(os).str()};
59  }
60  auto client = google::cloud::storage::Client::CreateDefaultClient().value();
61  command(std::move(client), std::move(argv));
62  };
63  return {name, std::move(adapter)};
64 }
