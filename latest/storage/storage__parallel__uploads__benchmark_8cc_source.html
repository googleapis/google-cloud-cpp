32 char const kDescription[] = R"""(
33 A parallel uploads benchmark for the Google Cloud Storage C++ client library.
34 
35 This program measures the throughput of uploading objects via
36 ParallelUploadFile() API from Google Cloud Storage C++ client library. The
37 program repeats the "experiment" of uploading a file many times, using a
38 randomly selected size and parallelism in each iteration. An
39 external script may perform statistical analysis on the results to estimate a
40 function for the proper number of shards.
41 
42 The program first creates a GCS bucket that will contain all the objects used
43 by that run of the program. The name of this bucket is selected at random, so
44 multiple copies of the program can run simultaneously. The bucket is deleted at
45 the end of the run of this program. The bucket uses the `STANDARD` storage
46 class, in a region set via the command line. Choosing regions close to where the
47 program is running can be used to estimate the throughput without any wide-area
48 network effects. Choosing regions far from where the program is running can be
49 used to evaluate the performance of the library when the WAN is taken into
50 account.
51 
52 After creating this bucket the program creates a number of threads, configurable
53 via the command line, to obtain more samples in parallel. Configure this value
54 with a small enough number of threads such that you do not saturate the CPU or
55 memory. Each thread creates a separate copy of the `storage::Client` object
56 repeats this loop:
57 
58 - Select a random size, between two values configured in the command line of the
59  object to upload.
60 - Select a random number of shards, between two values configured in the command
61  line, this is the level of parallelism of each upload.
62 - Upload an object of the selected size, choosing the name of the object at
63  random.
64 - Once the object is fully uploaded, the program captures the object size,
65  the elapsed time (in microseconds), used during the upload, and the status
66  code for the upload.
67 - The program then deletes this object and starts another iteration.
68 
69 The loop stops when any of the following conditions are met:
70 
71 - The test has obtained more than a prescribed "maximum number of samples"
72 - The test has obtained at least a prescribed "minimum number of samples" *and*
73  the test has been running for more than a prescribed "duration".
74 
75 Once the threads finish running their loops the program prints the captured
76 performance data. The bucket is deleted after the program terminates.
77 )""";

100 StatusOr<std::string> CreateTempFile(
101  std::string const& directory,
102  google::cloud::internal::DefaultPRNG& generator, std::uintmax_t size_left) {
103  std::size_t const single_buf_size = 4 * 1024 * 1024;
104  auto const file_name = directory + (directory.empty() ? "" : "/") +
105  gcs_bm::MakeRandomFileName(generator);
106 
107  std::string random_data = gcs_bm::MakeRandomData(generator, single_buf_size);
108 
109  std::ofstream file(file_name, std::ios::binary | std::ios::trunc);
110  if (!file.good()) {
111  return Status(
112  StatusCode::kInternal,
113  "Failed to create a temporary file (file_name=" + file_name + ")");
114  }
115  while (size_left > 0) {
116  auto const to_write = [size_left, &random_data] {
117  if (size_left > random_data.size()) return random_data.size();
118  return static_cast<std::size_t>(size_left);
119  }();
120  size_left -= to_write;
121  file.write(random_data.c_str(), to_write);
122  if (!file.good()) {
123  std::remove(file_name.c_str());
124  return Status(StatusCode::kInternal,
125  "Failed to write to file " + file_name);
126  }
127  }
128  return file_name;
129 }

131 Status PerformUpload(gcs::Client& client, std::string const& file_name,
132  std::string const& bucket_name, std::string const& prefix,
133  std::size_t num_shards) {
134  if (num_shards == 1) {
135  auto res = client.UploadFile(file_name, bucket_name, prefix + ".dest");
136  if (!res) {
137  return res.status();
138  }
139  return Status();
140  }
141  auto object_metadata = gcs::ParallelUploadFile(
142  client, file_name, bucket_name, prefix + ".dest", prefix, false,
143  gcs::MinStreamSize(0), gcs::MaxStreams(num_shards));
144  if (!object_metadata) {
145  return object_metadata.status();
146  }
147 
148  return Status();
149 }

151 StatusOr<std::chrono::milliseconds> TimeSingleUpload(
152  gcs::Client& client, std::string const& global_prefix,
153  std::string const& bucket_name, std::size_t num_shards,
154  std::string const& file_name) {
155  using std::chrono::milliseconds;
156 
157  std::string const& prefix = gcs::CreateRandomPrefixName(global_prefix);
158 
159  auto start = std::chrono::steady_clock::now();
160  auto status =
161  PerformUpload(client, file_name, bucket_name, prefix, num_shards);
162  auto elapsed = std::chrono::steady_clock::now() - start;
163  if (status.ok()) {
164  return std::chrono::duration_cast<std::chrono::milliseconds>(elapsed);
165  }
166  return status;
167 }

208  std::cout << "# Running test on bucket: " << meta.name() << "\n";

224  << "\n# Max Object Size (MiB): "
225  << options->maximum_object_size / gcs_bm::kMiB
226  << "\n# Min Number of Shards: " << options->minimum_num_shards
227  << "\n# Max Number of Shards: " << options->maximum_num_shards
228  << "\n# Build info: " << notes
229  << "\nFileSize,ShardCount,UploadTimeMs\n";
230 
231  std::vector<std::thread> threads;
232  std::atomic<int> iteration_count(0);
233  std::mutex cout_mutex;
234  for (int i = 0; i != options->thread_count; ++i) {
235  threads.emplace_back([&iteration_count, &cout_mutex, options, bucket_name] {
236  google::cloud::internal::DefaultPRNG generator =
237  google::cloud::internal::MakeDefaultPRNG();
238 
239  google::cloud::StatusOr<gcs::ClientOptions> client_options =
240  gcs::ClientOptions::CreateDefaultClientOptions();
241  if (!client_options) {
242  std::lock_guard<std::mutex> lk(cout_mutex);
243  std::cout << "# Could not create ClientOptions, status="
244  << client_options.status() << "\n";
245  return;
246  }
247  gcs::Client client(*std::move(client_options));
248 
249  std::uniform_int_distribution<std::uint64_t> size_generator(
250  options->minimum_object_size, options->maximum_object_size);
251  std::uniform_int_distribution<std::size_t> num_shards_generator(
252  options->minimum_num_shards, options->maximum_num_shards);
253 
254  auto deadline = std::chrono::steady_clock::now() + options->duration;
255 
256  for (auto start = std::chrono::steady_clock::now();
257  iteration_count < options->maximum_sample_count &&
258  (iteration_count < options->minimum_sample_count ||
259  start < deadline);
260  start = std::chrono::steady_clock::now(), ++iteration_count) {
261  auto const file_size = size_generator(generator);
262  auto const num_shards = num_shards_generator(generator);
263  auto file_name =
264  CreateTempFile(options->directory, generator, file_size);
265  if (!file_name) {
266  std::lock_guard<std::mutex> lk(cout_mutex);
267  std::cout << "# Could not prepare file to upload, size=" << file_size
268  << ", status=" << file_name.status() << "\n";
269  return;
270  }
271  auto time = TimeSingleUpload(client, options->object_prefix,
272  bucket_name, num_shards, *file_name);
273  std::remove(file_name->c_str());
274  if (!time) {
275  std::lock_guard<std::mutex> lk(cout_mutex);
276  std::cout << "# Could not create upload sample file, status="
277  << time.status() << "\n";
278  return;
279  }
280  std::lock_guard<std::mutex> lk(cout_mutex);
281  std::cout << file_size << ',' << num_shards << ',' << time->count()
282  << '\n';
283  }
284  });
285  };
286  for (auto& thread : threads) {
287  thread.join();
288  }
289 
290  auto status = DeleteByPrefix(client, bucket_name, "", gcs::Versions());
291  if (!status.ok()) {
292  std::cerr << "# Error deleting bucket leftovers status=" << status << "\n";
293  return 1;
294  }
295  status = client.DeleteBucket(bucket_name);
296  if (!status.ok()) {
297  std::cerr << "# Error deleting bucket, status=" << status << "\n";
298  return 1;
299  }
300  return 0;
301 }
