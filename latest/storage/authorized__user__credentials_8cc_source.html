23 StatusOr<AuthorizedUserCredentialsInfo> ParseAuthorizedUserCredentials(
24  std::string const& content, std::string const& source,
25  std::string const& default_token_uri) {
26  auto credentials = nlohmann::json::parse(content, nullptr, false);
27  if (credentials.is_discarded()) {
28  return Status(
29  StatusCode::kInvalidArgument,
30  "Invalid AuthorizedUserCredentials, parsing failed on data from " +
31  source);
32  }
33 
34  std::string const client_id_key = "client_id";
35  std::string const client_secret_key = "client_secret";
36  std::string const refresh_token_key = "refresh_token";
37  for (auto const& key :
38  {client_id_key, client_secret_key, refresh_token_key}) {
39  if (credentials.count(key) == 0) {
40  return Status(StatusCode::kInvalidArgument,
41  "Invalid AuthorizedUserCredentials, the " +
42  std::string(key) +
43  " field is missing on data loaded from " + source);
44  }
45  if (credentials.value(key, "").empty()) {
46  return Status(StatusCode::kInvalidArgument,
47  "Invalid AuthorizedUserCredentials, the " +
48  std::string(key) +
49  " field is empty on data loaded from " + source);
50  }
51  }
52  return AuthorizedUserCredentialsInfo{
53  credentials.value(client_id_key, ""),
54  credentials.value(client_secret_key, ""),
55  credentials.value(refresh_token_key, ""),
56  // Some credential formats (e.g. gcloud's ADC file) don't contain a
57  // "token_uri" attribute in the JSON object. In this case, we try using
58  // the default value.
59  credentials.value("token_uri", default_token_uri)};
60 }

62 StatusOr<RefreshingCredentialsWrapper::TemporaryToken>
63 ParseAuthorizedUserRefreshResponse(
64  storage::internal::HttpResponse const& response,
65  std::chrono::system_clock::time_point now) {
66  auto access_token = nlohmann::json::parse(response.payload, nullptr, false);
67  if (access_token.is_discarded() || access_token.count("access_token") == 0 ||
68  access_token.count("expires_in") == 0 ||
69  access_token.count("id_token") == 0 ||
70  access_token.count("token_type") == 0) {
71  auto payload =
72  response.payload +
73  "Could not find all required fields in response (access_token,"
74  " id_token, expires_in, token_type).";
75  return AsStatus(storage::internal::HttpResponse{response.status_code,
76  payload, response.headers});
77  }
78  std::string header = "Authorization: ";
79  header += access_token.value("token_type", "");
80  header += ' ';
81  header += access_token.value("access_token", "");
82  std::string new_id = access_token.value("id_token", "");
83  auto expires_in =
84  std::chrono::seconds(access_token.value("expires_in", int(0)));
85  auto new_expiration = now + expires_in;
86  return RefreshingCredentialsWrapper::TemporaryToken{std::move(header),
87  new_expiration};
88 }
