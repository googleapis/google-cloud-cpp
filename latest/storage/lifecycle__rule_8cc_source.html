112  LifecycleRuleCondition const& rhs) {
113  if (rhs.age.has_value()) {
114  if (result.age.has_value()) {
115  result.age = std::min(*result.age, *rhs.age);

117  auto tmp = *rhs.age;

119  }
120  }
121  if (rhs.created_before.has_value()) {
