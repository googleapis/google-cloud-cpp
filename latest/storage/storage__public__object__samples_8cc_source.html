25 void MakeObjectPublic(google::cloud::storage::Client client,
26  std::vector<std::string> const& argv) {
28  namespace gcs = google::cloud::storage;
29  using ::google::cloud::StatusOr;
30  [](gcs::Client client, std::string const& bucket_name,
31  std::string const& object_name) {
32  StatusOr<gcs::ObjectMetadata> updated = client.PatchObject(
33  bucket_name, object_name, gcs::ObjectMetadataPatchBuilder(),
34  gcs::PredefinedAcl::PublicRead());
35 
36  if (!updated) throw std::runtime_error(updated.status().message());
37  std::cout << "Object updated. The full metadata after the update is: "
38  << *updated << "\n";
39  }
41  (std::move(client), argv.at(0), argv.at(1));
42 }
