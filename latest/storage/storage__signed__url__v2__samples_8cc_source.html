22 void CreateGetSignedUrlV2(google::cloud::storage::Client client,
23  std::vector<std::string> const& argv) {
25  namespace gcs = google::cloud::storage;
26  using ::google::cloud::StatusOr;
27  [](gcs::Client client, std::string const& bucket_name,
28  std::string const& object_name, std::string const& signing_account) {
29  StatusOr<std::string> signed_url = client.CreateV2SignedUrl(
30  "GET", std::move(bucket_name), std::move(object_name),
31  gcs::ExpirationTime(std::chrono::system_clock::now() +
32  std::chrono::minutes(15)),
33  gcs::SigningAccount(signing_account));
34 
35  if (!signed_url) throw std::runtime_error(signed_url.status().message());
36  std::cout << "The signed url is: " << *signed_url << "\n\n"
37  << "You can use this URL with any user agent, for example:\n"
38  << "curl '" << *signed_url << "'\n";
39  }
41  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
42 }
