50 inline AddExtensionHeaderOption AddExtensionHeader(std::string header,
51  std::string value) {

54  [](char c) -> char { return static_cast<char>(std::tolower(c)); });
55  return AddExtensionHeaderOption(
56  std::make_pair(std::move(header), std::move(value)));
