36 class UploadObject : public ThroughputExperiment {
37  public:
38  explicit UploadObject(google::cloud::storage::Client client, ApiName api,
39  std::string random_data, bool prefer_insert)
40  : client_(std::move(client)),
41  api_(api),
42  random_data_(std::move(random_data)),
43  prefer_insert_(prefer_insert) {}
44  ~UploadObject() override = default;
45 
46  ThroughputResult Run(std::string const& bucket_name,
47  std::string const& object_name,
48  ThroughputExperimentConfig const& config) override {
49  auto api_selector = gcs::Fields();
50  if (api_ == ApiName::kApiXml) {
51  // The default API is JSON, we force XML by not using features that XML
52  // does not implement:
53  api_selector = gcs::Fields("");
54  }
55 
56  std::vector<char> buffer(config.app_buffer_size);
57 
58  // When the object is relatively small using `ObjectInsert` might be more
59  // efficient. Randomly select about 1/2 of the small writes to use
60  // ObjectInsert()
61  if (static_cast<std::size_t>(config.object_size) < random_data_.size() &&
62  prefer_insert_) {
63  Timer timer;
64  timer.Start();
65  std::string data =
66  random_data_.substr(0, static_cast<std::size_t>(config.object_size));
67  auto object_metadata = client_.InsertObject(
68  bucket_name, object_name, std::move(data),
69  gcs::DisableCrc32cChecksum(!config.enable_crc32c),
70  gcs::DisableMD5Hash(!config.enable_md5), api_selector);
71  timer.Stop();
72  return ThroughputResult{kOpInsert,
73  config.object_size,
74  config.app_buffer_size,
75  config.lib_buffer_size,
76  config.enable_crc32c,
77  config.enable_md5,
78  api_,
79  timer.elapsed_time(),
80  timer.cpu_time(),
81  object_metadata.status()};
82  }
83  Timer timer;
84  timer.Start();
85  auto writer = client_.WriteObject(
86  bucket_name, object_name,
87  gcs::DisableCrc32cChecksum(!config.enable_crc32c),
88  gcs::DisableMD5Hash(!config.enable_md5), api_selector);
89  for (std::int64_t offset = 0; offset < config.object_size;
90  offset += config.app_buffer_size) {
91  auto len = config.app_buffer_size;
92  if (offset + static_cast<std::int64_t>(len) > config.object_size) {
93  len = static_cast<std::size_t>(config.object_size - offset);
94  }
95  writer.write(random_data_.data(), len);
96  }
97  writer.Close();
98  timer.Stop();
99 
100  return ThroughputResult{kOpWrite,
101  config.object_size,
102  config.app_buffer_size,
103  config.lib_buffer_size,
104  config.enable_crc32c,
105  config.enable_md5,
106  api_,
107  timer.elapsed_time(),
108  timer.cpu_time(),
109  writer.metadata().status()};
110  }
111 
112  private:
113  google::cloud::storage::Client client_;
114  ApiName api_;
115  std::string random_data_;
116  bool prefer_insert_;
117 };

125  : client_(std::move(client)), api_(api) {}
126  ~DownloadObject() override = default;
127 
128  ThroughputResult Run(std::string const& bucket_name,
129  std::string const& object_name,
130  ThroughputExperimentConfig const& config) override {
131  auto api_selector = gcs::IfGenerationNotMatch();
132  if (api_ == ApiName::kApiJson) {
133  // The default API is XML, we force JSON by using a feature not available
134  // in XML.
135  api_selector = gcs::IfGenerationNotMatch(0);
136  }
137 
138  std::vector<char> buffer(config.app_buffer_size);
139 
140  Timer timer;
141  timer.Start();
142  auto reader = client_.ReadObject(
143  bucket_name, object_name,
144  gcs::DisableCrc32cChecksum(!config.enable_crc32c),
145  gcs::DisableMD5Hash(!config.enable_md5), api_selector);
146  for (std::uint64_t num_read = 0; reader.read(buffer.data(), buffer.size());
147  num_read += reader.gcount()) {
148  }
149  timer.Stop();
150  return ThroughputResult{config.op,
151  config.object_size,
152  config.app_buffer_size,
153  config.lib_buffer_size,
154  config.enable_crc32c,
155  config.enable_md5,
156  api_,
157  timer.elapsed_time(),
158  timer.cpu_time(),
159  reader.status()};
160  }
161 
162  private:
163  google::cloud::storage::Client client_;
164  ApiName api_;

178 class DownloadObjectLibcurl : public ThroughputExperiment {
179  public:
180  explicit DownloadObjectLibcurl(ApiName api)
181  : creds_(
182  google::cloud::storage::oauth2::GoogleDefaultCredentials().value()),
183  api_(api) {}
184  ~DownloadObjectLibcurl() override = default;
185 
186  ThroughputResult Run(std::string const& bucket_name,
187  std::string const& object_name,
188  ThroughputExperimentConfig const& config) override {
189  auto header = creds_->AuthorizationHeader();
190  if (!header) return {};
191 
192  Timer timer;
193  timer.Start();
194  struct curl_slist* slist1 = nullptr;
195  slist1 = curl_slist_append(slist1, header->c_str());
196 
197  auto* hnd = curl_easy_init();
198  curl_easy_setopt(hnd, CURLOPT_BUFFERSIZE, 102400L);
199  std::string url;
200  if (api_ == ApiName::kApiRawXml) {
201  url = "https://storage.googleapis.com/" + bucket_name + "/" + object_name;
202  } else {
203  // For this benchmark it is not necessary to URL escape the object name.
204  url = "https://storage.googleapis.com/storage/v1/b/" + bucket_name +
205  "/o/" + object_name + "?alt=media";
206  }
207  curl_easy_setopt(hnd, CURLOPT_URL, url.c_str());
208  curl_easy_setopt(hnd, CURLOPT_HTTPHEADER, slist1);
209  curl_easy_setopt(hnd, CURLOPT_USERAGENT, "curl/7.65.3");
210  curl_easy_setopt(hnd, CURLOPT_MAXREDIRS, 50L);
211  curl_easy_setopt(hnd, CURLOPT_HTTP09_ALLOWED, 1L);
212  curl_easy_setopt(hnd, CURLOPT_TCP_KEEPALIVE, 1L);
213 
214  std::vector<char> buffer(CURL_MAX_WRITE_SIZE);
215  curl_easy_setopt(hnd, CURLOPT_WRITEDATA, &buffer);
216  curl_easy_setopt(hnd, CURLOPT_WRITEFUNCTION, &OnWrite);
217  curl_easy_setopt(hnd, CURLOPT_HEADERDATA, nullptr);
218  curl_easy_setopt(hnd, CURLOPT_HEADERFUNCTION, &OnHeader);
219 
220  std::vector<char> error_buffer(2 * CURL_ERROR_SIZE);
221  curl_easy_setopt(hnd, CURLOPT_ERRORBUFFER, error_buffer.data());
222 
223  curl_easy_setopt(hnd, CURLOPT_STDERR, stderr);
224  CURLcode ret = curl_easy_perform(hnd);
225  Status status = ret == CURLE_OK
226  ? Status{}
227  : Status{StatusCode::kUnknown, "curl failed"};
228 
229  curl_easy_cleanup(hnd);
230  curl_slist_free_all(slist1);
231  timer.Stop();
232  return ThroughputResult{config.op,
233  config.object_size,
234  config.app_buffer_size,
235  config.lib_buffer_size,
236  config.enable_crc32c,
237  config.enable_md5,
238  api_,
239  timer.elapsed_time(),
240  timer.cpu_time(),
241  status};
242  }
243 
244  private:
245  std::shared_ptr<google::cloud::storage::oauth2::Credentials> creds_;
246  ApiName api_;
247 };

261 class DownloadObjectRawGrpc : public ThroughputExperiment {
262  public:
263  explicit DownloadObjectRawGrpc(int thread_id)
264  : stub_(google::storage::v1::Storage::NewStub(
265  CreateGcsChannel(thread_id))) {}
266  ~DownloadObjectRawGrpc() override = default;
267 
268  ThroughputResult Run(std::string const& bucket_name,
269  std::string const& object_name,
270  ThroughputExperimentConfig const& config) override {
271  Timer timer;
272 
273  timer.Start();
274  google::storage::v1::GetObjectMediaRequest request;
275  request.set_bucket(bucket_name);
276  request.set_object(object_name);
277  grpc::ClientContext context;
278  auto stream = stub_->GetObjectMedia(&context, request);
279  google::storage::v1::GetObjectMediaResponse response;
280  std::int64_t bytes_received = 0;
281  while (stream->Read(&response)) {
282  if (response.has_checksummed_data()) {
283  bytes_received += response.checksummed_data().content().size();
284  }
285  }
286  auto const status =
287  ::google::cloud::MakeStatusFromRpcError(stream->Finish());
288  timer.Stop();
289 
290  return ThroughputResult{config.op,
291  config.object_size,
292  config.app_buffer_size,
293  /*lib_buffer_size=*/0,
294  /*crc_enabled=*/false,
295  /*md5_enabled=*/false,
296  ApiName::kApiRawGrpc,
297  timer.elapsed_time(),
298  timer.cpu_time(),
299  status};
300  }
301 
302  private:
303  std::unique_ptr<google::storage::v1::Storage::StubInterface> stub_;
304 };

308 std::vector<std::unique_ptr<ThroughputExperiment>> CreateUploadExperiments(
309  ThroughputOptions const& options,
310  google::cloud::storage::ClientOptions const& client_options,
311  int thread_id) {
312  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
313  auto contents = MakeRandomData(generator, options.maximum_write_size);
314  gcs::Client rest_client(client_options);
315 
316  std::vector<std::unique_ptr<ThroughputExperiment>> result;
317  for (auto a : options.enabled_apis) {
318  switch (a) {
319 #if GOOGLE_CLOUD_CPP_STORAGE_HAVE_GRPC
320  case ApiName::kApiGrpc:
321  case ApiName::kApiRawGrpc: {
322  gcs::Client grpc_client =
323  google::cloud::storage_experimental::DefaultGrpcClient(
324  client_options, thread_id);
325  result.push_back(
326  absl::make_unique<UploadObject>(grpc_client, a, contents, false));
327  result.push_back(
328  absl::make_unique<UploadObject>(grpc_client, a, contents, true));
329  }
330  // this comment keeps clang-format from merging to previous line
331  // methinks `} break;` is just weird.
332  break;
333 #else
334  case ApiName::kApiGrpc:
335  case ApiName::kApiRawGrpc:
336  (void)thread_id;
337  break;
338 #endif // GOOGLE_CLOUD_CPP_STORAGE_HAVE_GRPC
339  case ApiName::kApiXml:
340  case ApiName::kApiJson:
341  case ApiName::kApiRawJson:
342  case ApiName::kApiRawXml:
343  result.push_back(
344  absl::make_unique<UploadObject>(rest_client, a, contents, false));
345  result.push_back(
346  absl::make_unique<UploadObject>(rest_client, a, contents, true));

353 std::vector<std::unique_ptr<ThroughputExperiment>> CreateDownloadExperiments(
354  ThroughputOptions const& options,
355  google::cloud::storage::ClientOptions const& client_options,
356  int thread_id) {
357  gcs::Client rest_client(client_options);
358 
359  std::vector<std::unique_ptr<ThroughputExperiment>> result;
360  for (auto a : options.enabled_apis) {
361  switch (a) {
362 #if GOOGLE_CLOUD_CPP_STORAGE_HAVE_GRPC
363  case ApiName::kApiGrpc:
364  result.push_back(absl::make_unique<DownloadObject>(
365  google::cloud::storage_experimental::DefaultGrpcClient(
366  client_options),
367  a));
368  break;
369 #else
370  case ApiName::kApiGrpc:
371  break;
372 #endif // GOOGLE_CLOUD_CPP_STORAGE_HAVE_GRPC
373  case ApiName::kApiXml:
374  case ApiName::kApiJson:
375  result.push_back(absl::make_unique<DownloadObject>(rest_client, a));
376  break;
377  case ApiName::kApiRawXml:
378  case ApiName::kApiRawJson:
379  result.push_back(absl::make_unique<DownloadObjectLibcurl>(a));
380  break;
381  case ApiName::kApiRawGrpc:
382  result.push_back(absl::make_unique<DownloadObjectRawGrpc>(thread_id));
383  break;
384  }
385  }
386  return result;
