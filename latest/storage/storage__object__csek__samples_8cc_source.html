29 std::string GenerateEncryptionKey() {
31  // Create a pseudo-random number generator (PRNG), this is included for
32  // demonstration purposes only. You should consult your security team about
33  // best practices to initialize PRNG. In particular, you should verify that
34  // the C++ library and operating system provide enough entropy to meet the
35  // security policies in your organization.
36 
37  // Use the Mersenne-Twister Engine in this example:
38  // https://en.cppreference.com/w/cpp/numeric/random/mersenne_twister_engine
39  // Any C++ PRNG can be used below, the choice is arbitrary.
40  using GeneratorType = std::mt19937_64;
41 
42  // Create the default random device to fetch entropy.
43  std::random_device rd;
44 
45  // Compute how much entropy we need to initialize the GeneratorType:
46  constexpr auto kRequiredEntropyWords =
47  GeneratorType::state_size *
48  (GeneratorType::word_size / std::numeric_limits<unsigned int>::digits);
49 
50  // Capture the entropy bits into a vector.
51  std::vector<std::uint64_t> entropy(kRequiredEntropyWords);
52  std::generate(entropy.begin(), entropy.end(), [&rd] { return rd(); });
53 
54  // Create the PRNG with the fetched entropy.
55  std::seed_seq seed(entropy.begin(), entropy.end());
56 
57  // initialized with enough entropy such that the encryption keys are not
58  // predictable. Note that the default constructor for all the generators in
59  // the C++ standard library produce predictable keys.
60  std::mt19937_64 gen(seed);
61 
62  namespace gcs = google::cloud::storage;
63  gcs::EncryptionKeyData data = gcs::CreateKeyFromGenerator(gen);
64 
65  std::cout << "Base64 encoded key = " << data.key << "\n"
66  << "Base64 encoded SHA256 of key = " << data.sha256 << "\n";
68  return data.key;
69 }

118 void ComposeObjectFromEncryptedObjects(google::cloud::storage::Client client,
119  std::vector<std::string> const& argv) {
120  auto it = argv.cbegin();
121  auto bucket_name = *it++;
122  auto destination_object_name = *it++;
123  auto base64_aes256_key = *it++;
124  std::vector<google::cloud::storage::ComposeSourceObject> compose_objects;
125  do {
126  compose_objects.push_back({*it++, {}, {}});
127  } while (it != argv.cend());
128 
130  namespace gcs = google::cloud::storage;
131  using ::google::cloud::StatusOr;

133  std::string const& destination_object_name,

135  std::vector<gcs::ComposeSourceObject> const& compose_objects) {
136  StatusOr<gcs::ObjectMetadata> composed_object = client.ComposeObject(
137  bucket_name, compose_objects, destination_object_name,

140  if (!composed_object) {

142  }
143 
144  std::cout << "Composed new object " << composed_object->name()
145  << " in bucket " << composed_object->bucket()
146  << "\nFull metadata: " << *composed_object << "\n";

182 void RotateEncryptionKey(google::cloud::storage::Client client,
183  std::vector<std::string> const& argv) {
185  namespace gcs = google::cloud::storage;
186  using ::google::cloud::StatusOr;
187  [](gcs::Client client, std::string const& bucket_name,
188  std::string const& object_name, std::string const& old_key_base64,
189  std::string const& new_key_base64) {
190  StatusOr<gcs::ObjectMetadata> object_metadata =
191  client.RewriteObjectBlocking(
192  bucket_name, object_name, bucket_name, object_name,
193  gcs::SourceEncryptionKey::FromBase64Key(old_key_base64),
194  gcs::EncryptionKey::FromBase64Key(new_key_base64));
195 
196  if (!object_metadata) {
197  throw std::runtime_error(object_metadata.status().message());
198  }
199 
200  std::cout << "Rotated key on object " << object_metadata->name()
201  << " in bucket " << object_metadata->bucket()
202  << "\nFull Metadata: " << *object_metadata << "\n";
203  }
205  (std::move(client), argv.at(0), argv.at(1), argv.at(2), argv.at(3));
