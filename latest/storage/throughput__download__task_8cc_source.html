26 ThroughputResult ClientDownloadTask::PerformDownload(
27  std::string const& bucket_name, std::string const& object_name,
28  DownloadConfig const& config) {
29  auto json_read_selector = gcs::IfGenerationNotMatch();
30  if (api_ == ApiName::kApiJson) {
31  // The default API is XML, we force JSON by using a feature not available
32  // in XML.
33  json_read_selector = gcs::IfGenerationNotMatch(0);
34  }
35 
36  std::vector<char> buffer(config.read_size);
37 
38  SimpleTimer timer;
39  timer.Start();
40  auto reader = client_.ReadObject(
41  bucket_name, object_name,
42  gcs::DisableCrc32cChecksum(!config.enable_crc32c),
43  gcs::DisableMD5Hash(!config.enable_md5), json_read_selector);
44  for (size_t num_read = 0; reader.read(buffer.data(), buffer.size());
45  num_read += reader.gcount()) {
46  }
47  timer.Stop();
48  return ThroughputResult{config.op,
49  config.object_size,
50  config.read_size,
51  config.download_buffer_size,
52  config.enable_crc32c,
53  config.enable_md5,
54  api_,
55  timer.elapsed_time(),
56  timer.cpu_time(),
57  reader.status().code()};
58 }
