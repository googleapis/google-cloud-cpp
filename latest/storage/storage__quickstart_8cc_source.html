28 void StorageQuickstart(std::string const& bucket_name) {
29  // Create an aliases to make the code easier to read.
30  namespace gcs = google::cloud::storage;
31 
32  // Create a client to communicate with Google Cloud Storage. This client
33  // uses the default configuration for authentication and project id.
34  google::cloud::StatusOr<gcs::Client> client =
35  gcs::Client::CreateDefaultClient();
36  if (!client) throw std::runtime_error(client.status().message());
37 
38  // Create a bucket
39  google::cloud::StatusOr<gcs::BucketMetadata> metadata = client->CreateBucket(
40  bucket_name, gcs::BucketMetadata().set_location("US").set_storage_class(
41  gcs::storage_class::Standard()));
42  if (!metadata) throw std::runtime_error(metadata.status().message());
43 
44  std::cout << "Created bucket " << metadata->name() << "\n";
45 }
46 // [END storage_quickstart]
