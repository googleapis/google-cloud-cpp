23 StatusOr<ServiceAccountMetadata> ParseMetadataServerResponse(
24  storage::internal::HttpResponse const& response) {
25  auto response_body = nlohmann::json::parse(response.payload, nullptr, false);
26  // Note that the "scopes" attribute will always be present and contain a
27  // JSON array. At minimum, for the request to succeed, the instance must
28  // have been granted the scope that allows it to retrieve info from the
29  // metadata server.
30  if (response_body.is_discarded() || response_body.count("email") == 0 ||
31  response_body.count("scopes") == 0) {
32  auto payload =
33  response.payload +
34  "Could not find all required fields in response (email, scopes).";
35  return AsStatus(storage::internal::HttpResponse{
36  storage::internal::HttpStatusCode::kMinInvalidCode, payload,
37  response.headers});
38  }
39  ServiceAccountMetadata metadata;
40  // Do not update any state until all potential errors are handled.
41  metadata.email = response_body.value("email", "");
42  // We need to call the .get<>() helper because the conversion is ambiguous
43  // otherwise.
44  metadata.scopes = response_body["scopes"].get<std::set<std::string>>();
45  return metadata;
46 }

48 StatusOr<RefreshingCredentialsWrapper::TemporaryToken>
49 ParseComputeEngineRefreshResponse(
50  storage::internal::HttpResponse const& response,
51  std::chrono::system_clock::time_point now) {
52  // Response should have the attributes "access_token", "expires_in", and
53  // "token_type".
54  auto access_token = nlohmann::json::parse(response.payload, nullptr, false);
55  if (access_token.is_discarded() || access_token.count("access_token") == 0 or
56  access_token.count("expires_in") == 0 or
57  access_token.count("token_type") == 0) {
58  auto payload =
59  response.payload +
60  "Could not find all required fields in response (access_token,"
61  " expires_in, token_type).";
62  return AsStatus(storage::internal::HttpResponse{response.status_code,
63  payload, response.headers});
64  }
65  std::string header = "Authorization: ";
66  header += access_token.value("token_type", "");
67  header += ' ';
68  header += access_token.value("access_token", "");
69  auto expires_in =
70  std::chrono::seconds(access_token.value("expires_in", int(0)));
71  auto new_expiration = now + expires_in;
72 
73  return RefreshingCredentialsWrapper::TemporaryToken{std::move(header),
74  new_expiration};
75 }
