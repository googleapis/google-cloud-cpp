29 void DeleteAllObjects(google::cloud::storage::Client client,
30  std::string const& bucket_name, int thread_count) {
31  using WorkQueue = BoundedQueue<google::cloud::storage::ObjectMetadata>;
32  using std::chrono::duration_cast;
33  using std::chrono::milliseconds;
34  namespace gcs = google::cloud::storage;
35 
36  std::cout << "# Deleting test objects [" << thread_count << "]\n";
37  auto start = std::chrono::steady_clock::now();
38  WorkQueue work_queue;
39  std::vector<std::future<google::cloud::Status>> workers;
40  std::generate_n(
41  std::back_inserter(workers), thread_count, [&client, &work_queue] {
42  auto worker_task = [](gcs::Client client, WorkQueue& wq) {
43  google::cloud::Status status{};
44  for (auto object = wq.Pop(); object.has_value(); object = wq.Pop()) {
45  auto s = client.DeleteObject(object->bucket(), object->name(),
46  gcs::Generation(object->generation()));
47  if (!s.ok()) status = s;
48  }
49  return status;
50  };
51  return std::async(std::launch::async, worker_task, client,
52  std::ref(work_queue));
53  });
54 
55  for (auto& o : client.ListObjects(bucket_name, gcs::Versions(true))) {
56  if (!o) break;
57  work_queue.Push(*std::move(o));
58  }
59  work_queue.Shutdown();
60  int count = 0;
61  for (auto& t : workers) {
62  auto status = t.get();
63  if (!status.ok()) {
64  std::cerr << "Error return task[" << count << "]: " << status << "\n";
65  }
66  ++count;
67  }
68  auto elapsed = std::chrono::steady_clock::now() - start;
69  std::cout << "# Deleted in " << duration_cast<milliseconds>(elapsed).count()
70  << "ms\n";
71 }
