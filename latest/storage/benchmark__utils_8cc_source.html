124 std::vector<std::string> OptionsParse(std::vector<OptionDescriptor> const& desc,
125  std::vector<std::string> argv) {
126  if (argv.empty()) {
127  return argv;
128  }
129  std::string const usage = BuildUsage(desc, argv[0]);
130 
131  auto next_arg = argv.begin() + 1;
132  while (next_arg != argv.end()) {
133  std::string const& argument = *next_arg;
134 
135  // Try to match `argument` against the options in `desc`
136  bool matched = false;
137  for (auto const& d : desc) {
138  if (argument.rfind(d.option, 0) != 0) {
139  // Not a match, keep searching
140  continue;
141  }
142  std::string val = argument.substr(d.option.size());
143  if (!val.empty() && val[0] != '=') {
144  // Matched a prefix of an option, keep searching.
145  continue;
146  }
147  if (!val.empty()) {
148  // The first character must be '=', remove it too.
149  val.erase(val.begin());
150  }
151  d.parser(val);
152  // This is a match, consume the argument and stop the search.
153  matched = true;
154  break;
155  }
156  // If next_arg is matched against any option erase it, otherwise skip it.
157  next_arg = matched ? argv.erase(next_arg) : next_arg + 1;
158  }
159  return argv;
160 }

181 void SimpleTimer::Stop() {
182  elapsed_time_ = std::chrono::duration_cast<std::chrono::microseconds>(
183  std::chrono::steady_clock::now() - start_);
184 
185 #if GOOGLE_CLOUD_CPP_HAVE_GETRUSAGE
186  auto as_usec = [](timeval const& tv) {
187  return std::chrono::microseconds(std::chrono::seconds(tv.tv_sec)) +
188  std::chrono::microseconds(tv.tv_usec);
189  };
190 
191  struct rusage now {};
192  (void)getrusage(rusage_who(), &now);
193  auto utime = as_usec(now.ru_utime) - as_usec(start_usage_.ru_utime);
194  auto stime = as_usec(now.ru_stime) - as_usec(start_usage_.ru_stime);
195  cpu_time_ = utime + stime;
196  double cpu_fraction = 0;
197  if (elapsed_time_.count() != 0) {
198  cpu_fraction = static_cast<double>(cpu_time_.count()) /
199  static_cast<double>(elapsed_time_.count());
200  }
201  now.ru_minflt -= start_usage_.ru_minflt;
202  now.ru_majflt -= start_usage_.ru_majflt;
203  now.ru_nswap -= start_usage_.ru_nswap;
204  now.ru_inblock -= start_usage_.ru_inblock;
205  now.ru_oublock -= start_usage_.ru_oublock;
206  now.ru_msgsnd -= start_usage_.ru_msgsnd;
207  now.ru_msgrcv -= start_usage_.ru_msgrcv;
208  now.ru_nsignals -= start_usage_.ru_nsignals;
209  now.ru_nvcsw -= start_usage_.ru_nvcsw;
210  now.ru_nivcsw -= start_usage_.ru_nivcsw;
211 
212  std::ostringstream os;
213  os << "# user time =" << utime.count() << " us\n"
214  << "# system time =" << stime.count() << " us\n"
215  << "# CPU fraction =" << cpu_fraction << "\n"
216  << "# maximum resident set size =" << now.ru_maxrss << " KiB\n"
217  << "# integral shared memory size =" << now.ru_ixrss << " KiB\n"
218  << "# integral unshared data size =" << now.ru_idrss << " KiB\n"
219  << "# integral unshared stack size =" << now.ru_isrss << " KiB\n"
220  << "# soft page faults =" << now.ru_minflt << "\n"
221  << "# hard page faults =" << now.ru_majflt << "\n"
222  << "# swaps =" << now.ru_nswap << "\n"
223  << "# block input operations =" << now.ru_inblock << "\n"
224  << "# block output operations =" << now.ru_oublock << "\n"
225  << "# IPC messages sent =" << now.ru_msgsnd << "\n"
226  << "# IPC messages received =" << now.ru_msgrcv << "\n"
227  << "# signals received =" << now.ru_nsignals << "\n"
228  << "# voluntary context switches =" << now.ru_nvcsw << "\n"
229  << "# involuntary context switches =" << now.ru_nivcsw << "\n";
230  annotations_ = std::move(os).str();
231 #endif // GOOGLE_CLOUD_CPP_HAVE_GETRUSAGE
232 }

242 std::string FormatSize(std::uintmax_t size) {
243  struct {
244  std::uintmax_t limit;
245  std::uintmax_t resolution;
246  char const* name;
247  } ranges[] = {
248  {kKiB, 1, "B"},
249  {kMiB, kKiB, "KiB"},
250  {kGiB, kMiB, "MiB"},
251  {kTiB, kGiB, "GiB"},
252  };
253  std::uintmax_t resolution = kTiB;
254  char const* name = "TiB";
255  for (auto const& r : ranges) {
256  if (size < r.limit) {
257  resolution = r.resolution;
258  name = r.name;
259  break;
260  }
261  }
262  std::ostringstream os;
263  os.setf(std::ios::fixed);
264  os.precision(1);
265  os << (static_cast<double>(size) / static_cast<double>(resolution)) << name;
266  return os.str();
267 }

269 void DeleteAllObjects(google::cloud::storage::Client client,
270  std::string const& bucket_name, int thread_count) {
271  using WorkQueue = BoundedQueue<google::cloud::storage::ObjectMetadata>;
272  using std::chrono::duration_cast;
273  using std::chrono::milliseconds;
274  namespace gcs = google::cloud::storage;
275 
276  std::cout << "# Deleting test objects [" << thread_count << "]\n";
277  auto start = std::chrono::steady_clock::now();
278  WorkQueue work_queue;
279  std::vector<std::future<google::cloud::Status>> workers;
280  std::generate_n(
281  std::back_inserter(workers), thread_count, [&client, &work_queue] {
282  auto worker_task = [](gcs::Client client, WorkQueue& wq) {
283  google::cloud::Status status{};
284  for (auto object = wq.Pop(); object.has_value(); object = wq.Pop()) {
285  auto s = client.DeleteObject(object->bucket(), object->name(),
286  gcs::Generation(object->generation()));
287  if (!s.ok()) status = s;
288  }
289  return status;
290  };
291  return std::async(std::launch::async, worker_task, client,
292  std::ref(work_queue));
293  });
294 
295  for (auto& o : client.ListObjects(bucket_name, gcs::Versions(true))) {
296  if (!o) break;
297  work_queue.Push(*std::move(o));
298  }
299  work_queue.Shutdown();
300  int count = 0;
301  for (auto& t : workers) {
302  auto status = t.get();
303  if (!status.ok()) {
304  std::cerr << "Error return task[" << count << "]: " << status << "\n";
305  }
306  ++count;
307  }
308  auto elapsed = std::chrono::steady_clock::now() - start;
309  std::cout << "# Deleted in " << duration_cast<milliseconds>(elapsed).count()
310  << "ms\n";
311 }
