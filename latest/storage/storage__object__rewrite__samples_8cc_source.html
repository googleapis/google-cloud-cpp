86 
87  StatusOr<gcs::RewriteProgress> progress = rewriter.Iterate();

90  if (progress->done) {
91  std::cout
92  << "The rewrite completed immediately, no token to resume later\n";
93  return;
94  }
95  std::cout << "Rewrite in progress, token " << rewriter.token() << "\n";
96  }

133 void RenameObject(google::cloud::storage::Client client,
134  std::vector<std::string> const& argv) {
136  namespace gcs = google::cloud::storage;
137  using ::google::cloud::StatusOr;
138  [](gcs::Client client, std::string const& bucket_name,
139  std::string const& old_object_name, std::string const& new_object_name) {
140  StatusOr<gcs::ObjectMetadata> metadata = client.RewriteObjectBlocking(
141  bucket_name, old_object_name, bucket_name, new_object_name);
142  if (!metadata) throw std::runtime_error(metadata.status().message());
143 
144  google::cloud::Status status =
145  client.DeleteObject(bucket_name, old_object_name);
146  if (!status.ok()) throw std::runtime_error(status.message());
147 
148  std::cout << "Renamed " << old_object_name << " to " << new_object_name
149  << " in bucket " << bucket_name << "\n";
150  }
152  (std::move(client), argv.at(0), argv.at(1), argv.at(2));

155 void RunAll(std::vector<std::string> const& argv) {
156  namespace examples = ::google::cloud::storage::examples;
157  namespace gcs = ::google::cloud::storage;
158 
159  if (!argv.empty()) throw examples::Usage{"auto"};
160  examples::CheckEnvironmentVariablesAreSet({
161  "GOOGLE_CLOUD_PROJECT",
162  "GOOGLE_CLOUD_CPP_STORAGE_TEST_BUCKET_NAME",
163  "GOOGLE_CLOUD_CPP_STORAGE_TEST_DESTINATION_BUCKET_NAME",
164  });
165  auto const project_id =
166  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
167  auto const bucket_name = google::cloud::internal::GetEnv(
168  "GOOGLE_CLOUD_CPP_STORAGE_TEST_BUCKET_NAME")
169  .value();
170  auto const destination_bucket_name =
171  google::cloud::internal::GetEnv(
172  "GOOGLE_CLOUD_CPP_STORAGE_TEST_DESTINATION_BUCKET_NAME")
173  .value();
174 
175  auto client = gcs::Client::CreateDefaultClient().value();
176  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
177  auto const src_object_name =
178  examples::MakeRandomObjectName(generator, "object-") + ".txt";
179  auto const dst_object_name =
180  examples::MakeRandomObjectName(generator, "object-") + ".txt";
181  auto const old_object_name =
182  examples::MakeRandomObjectName(generator, "old-name-") + ".txt";
183  auto const new_object_name =
184  examples::MakeRandomObjectName(generator, "new-name-") + ".txt";
185  auto constexpr kText = R"""(Some text to insert in the test objects.)""";
186 
187  std::cout << "\nCreating an object to run the RenameObject() example"
188  << std::endl;
189  (void)client.InsertObject(bucket_name, old_object_name, kText).value();
190 
191  std::cout << "\nRunning the RenameObject() example" << std::endl;
192  RenameObject(client, {bucket_name, old_object_name, new_object_name});
193 
194  std::cout << "\nCleanup" << std::endl;
195  (void)client.DeleteObject(bucket_name, new_object_name);
196 
197  (void)client.InsertObject(bucket_name, src_object_name, kText).value();
198 
199  std::cout << "\nRunning the RewriteObject() example" << std::endl;
200  RewriteObject(client, {bucket_name, src_object_name, destination_bucket_name,
201  dst_object_name});
202 
203  std::cout << "\nRunning the RewriteObjectNonBlocking() example" << std::endl;
204  RewriteObjectNonBlocking(client, {bucket_name, src_object_name,
205  destination_bucket_name, dst_object_name});
206 
207  std::cout << "\nRunning the RewriteObjectToken() example [1]" << std::endl;
208  RewriteObjectToken(client, {bucket_name, src_object_name,
209  destination_bucket_name, dst_object_name});
210 
211  // Create a large object and obtain a token to rewrite to it.
212  std::cout << "\nCreating large object to test rewrites" << std::endl;
213  auto constexpr kRewriteBlock = 1024 * 1024L;
214  auto constexpr kDesiredSize = 16 * kRewriteBlock;
215  auto constexpr kLineSize = 256;
216  static_assert(kDesiredSize % kLineSize == 0,
217  "Desired size should be a multiple of the line size");
218  auto constexpr kLineCount = kDesiredSize / kLineSize;
219  auto const line =
220  google::cloud::internal::Sample(generator, kLineSize - 1,
221  "abcdefghijklmnopqrstuvwxyz0123456789") +
222  "\n";
223  auto writer = client.WriteObject(bucket_name, src_object_name);
224  for (int i = 0; i != kLineCount; ++i) {
225  writer.write(line.data(), line.size());
226  }
227  writer.Close();
228 
229  auto src = writer.metadata().value();
230  std::cout << "\nStarting large object (" << src.size() << " rewrite"
231  << std::endl;
232  auto rewriter = client.RewriteObject(
233  bucket_name, src_object_name, destination_bucket_name, dst_object_name,
234  gcs::MaxBytesRewrittenPerCall(kRewriteBlock));
235  auto progress = rewriter.Iterate().value();
236  if (progress.done) throw std::runtime_error("Rewrite completed unexpectedly");
237 
238  std::cout << "\nRunning the RewriteObjectResume() example" << std::endl;
239  RewriteObjectResume(client,
240  {bucket_name, src_object_name, destination_bucket_name,
241  dst_object_name, rewriter.token()});
242 
243  std::cout << "\nRunning the RewriteObjectToken() example [2]" << std::endl;
244  RewriteObjectToken(client, {bucket_name, src_object_name,
245  destination_bucket_name, dst_object_name});
246 
247  (void)client.DeleteObject(destination_bucket_name, dst_object_name);
248  (void)client.DeleteObject(bucket_name, src_object_name);
249 }
