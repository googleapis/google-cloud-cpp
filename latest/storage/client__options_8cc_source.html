74 std::size_t DefaultConnectionPoolSize() {
75  std::size_t nthreads = std::thread::hardware_concurrency();
76  if (nthreads == 0) {
77  return 4;
78  }
79  return 4 * nthreads;
80 }

141 void ClientOptions::SetupFromEnvironment() {
142  auto enable_clog =
143  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_CLOG");
144  if (enable_clog.has_value()) {
145  google::cloud::LogSink::EnableStdClog();
146  }
147  // This is overkill right now, eventually we will have different components
148  // that can be traced (http being the first), so we parse the environment
149  // variable.
150  auto tracing =
151  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_TRACING");
152  if (tracing.has_value()) {
153  std::set<std::string> const enabled = absl::StrSplit(*tracing, ',');
154  if (enabled.end() != enabled.find("http")) {
155  GCP_LOG(INFO) << "Enabling logging for http";
156  set_enable_http_tracing(true);
157  }
158  if (enabled.end() != enabled.find("raw-client")) {
159  GCP_LOG(INFO) << "Enabling logging for RawClient functions";
160  set_enable_raw_client_tracing(true);
161  }
162  }
163 
164  auto project_id = google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT");
165  if (project_id.has_value()) {
166  project_id_ = std::move(*project_id);
167  }
168 }
