32 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountCredentials(
33  std::string const& content, std::string const& source,
34  std::string const& default_token_uri) {
35  auto credentials = nlohmann::json::parse(content, nullptr, false);
36  if (credentials.is_discarded()) {
37  return Status(StatusCode::kInvalidArgument,
38  "Invalid ServiceAccountCredentials,"
39  "parsing failed on data loaded from " +
40  source);
41  }
42  std::string const private_key_id_key = "private_key_id";
43  std::string const private_key_key = "private_key";
44  std::string const token_uri_key = "token_uri";
45  std::string const client_email_key = "client_email";
46  for (auto const& key : {private_key_key, client_email_key}) {
47  if (credentials.count(key) == 0) {
48  return Status(StatusCode::kInvalidArgument,
49  "Invalid ServiceAccountCredentials, the " +
50  std::string(key) +
51  " field is missing on data loaded from " + source);
52  }
53  if (credentials.value(key, "").empty()) {
54  return Status(StatusCode::kInvalidArgument,
55  "Invalid ServiceAccountCredentials, the " +
56  std::string(key) +
57  " field is empty on data loaded from " + source);
58  }
59  }
60  // The token_uri field may be missing, but may not be empty:
61  if (credentials.count(token_uri_key) != 0 &&
62  credentials.value(token_uri_key, "").empty()) {
63  return Status(StatusCode::kInvalidArgument,
64  "Invalid ServiceAccountCredentials, the " +
65  std::string(token_uri_key) +
66  " field is empty on data loaded from " + source);
67  }
68  return ServiceAccountCredentialsInfo{
69  credentials.value(client_email_key, ""),
70  credentials.value(private_key_id_key, ""),
71  credentials.value(private_key_key, ""),
72  // Some credential formats (e.g. gcloud's ADC file) don't contain a
73  // "token_uri" attribute in the JSON object. In this case, we try using
74  // the default value.
75  credentials.value(token_uri_key, default_token_uri),
76  /*scopes*/ {},
77  /*subject*/ {}};
78 }

81 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountP12File(
82  std::string const& source, std::string const& default_token_uri) {
83  OpenSSL_add_all_algorithms();
84 
85  PKCS12* p12_raw = [](std::string const& source) {
86  FILE* fp = std::fopen(source.c_str(), "rb");
87  if (fp == nullptr) {
88  return static_cast<PKCS12*>(nullptr);
89  }
90  auto* result = d2i_PKCS12_fp(fp, nullptr);
91  fclose(fp);
92  return result;
93  }(source);
94 
95  std::unique_ptr<PKCS12, decltype(&PKCS12_free)> p12(p12_raw, &PKCS12_free);
96 
97  auto capture_openssl_errors = []() {
98  std::string msg;
99  while (auto code = ERR_get_error()) {
100  // OpenSSL guarantees that 256 bytes is enough:
101  // https://www.openssl.org/docs/man1.1.1/man3/ERR_error_string_n.html
102  // https://www.openssl.org/docs/man1.0.2/man3/ERR_error_string_n.html
103  // we could not find a macro or constant to replace the 256 literal.
104  auto constexpr kMaxOpenSslErrorLength = 256;
105  std::array<char, kMaxOpenSslErrorLength> buf{};
106  ERR_error_string_n(code, buf.data(), buf.size());
107  msg += buf.data();
108  }
109  return msg;
110  };
111 
112  if (p12 == nullptr) {
113  std::string msg = "Cannot open PKCS#12 file (" + source + "): ";
114  msg += capture_openssl_errors();
115  return Status(StatusCode::kInvalidArgument, msg);
116  }
117 
118  EVP_PKEY* pkey_raw;
119  X509* cert_raw;
120  if (PKCS12_parse(p12.get(), "notasecret", &pkey_raw, &cert_raw, nullptr) !=
121  1) {
122  std::string msg = "Cannot parse PKCS#12 file (" + source + "): ";
123  msg += capture_openssl_errors();
124  return Status(StatusCode::kInvalidArgument, msg);
125  }
126 
127  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> pkey(pkey_raw,
128  &EVP_PKEY_free);
129  std::unique_ptr<X509, decltype(&X509_free)> cert(cert_raw, &X509_free);
130 
131  if (pkey_raw == nullptr) {
132  return Status(StatusCode::kInvalidArgument,
133  "No private key found in PKCS#12 file (" + source + ")");
134  }
135  if (cert_raw == nullptr) {
136  return Status(StatusCode::kInvalidArgument,
137  "No private key found in PKCS#12 file (" + source + ")");
138  }
139 
140  // This is automatically deleted by `cert`.
141  X509_NAME* name = X509_get_subject_name(cert.get());
142 
143  std::string service_account_id = [&name]() -> std::string {
144  auto openssl_free = [](void* addr) { OPENSSL_free(addr); };
145  std::unique_ptr<char, decltype(openssl_free)> oneline(
146  X509_NAME_oneline(name, nullptr, 0), openssl_free);
147  // We expect the name to be simply CN/ followed by a (small) number of
148  // digits.
149  if (strncmp("/CN=", oneline.get(), 4) != 0) {
150  return "";
151  }
152  return oneline.get() + 4;
153  }();
154 
155  if (service_account_id.find_first_not_of("0123456789") != std::string::npos ||
156  service_account_id.empty()) {
157  return Status(
158  StatusCode::kInvalidArgument,
159  "Invalid PKCS#12 file (" + source +
160  "): service account id missing or not not formatted correctly");
161  }
162 
163  std::unique_ptr<BIO, decltype(&BIO_free)> mem_io(BIO_new(BIO_s_mem()),
164  &BIO_free);
165 
166  if (PEM_write_bio_PKCS8PrivateKey(mem_io.get(), pkey.get(), nullptr, nullptr,
167  0, nullptr, nullptr) == 0) {
168  std::string msg =
169  "Cannot print private key in PKCS#12 file (" + source + "): ";
170  msg += capture_openssl_errors();
171  return Status(StatusCode::kUnknown, msg);
172  }
173 
174  // This buffer belongs to the BIO chain and is freed upon its destruction.
175  BUF_MEM* buf_mem;
176  BIO_get_mem_ptr(mem_io.get(), &buf_mem);
177 
178  std::string private_key(buf_mem->data, buf_mem->length);
179 
180  return ServiceAccountCredentialsInfo{std::move(service_account_id),
181  "--unknown--",
182  std::move(private_key),
183  default_token_uri,
184  /*scopes*/ {},
185  /*subject*/ {}};
186 }

189 std::pair<std::string, std::string> AssertionComponentsFromInfo(
190  ServiceAccountCredentialsInfo const& info,
191  std::chrono::system_clock::time_point now) {
192  nlohmann::json assertion_header = {{"alg", "RS256"}, {"typ", "JWT"}};
193  if (!info.private_key_id.empty()) {
194  assertion_header["kid"] = info.private_key_id;
195  }
196 
197  // Scopes must be specified in a comma-delimited string.
198  std::string scope_str;
199  if (!info.scopes) {
200  scope_str = GoogleOAuthScopeCloudPlatform();
201  } else {
202  absl::StrAppend(&scope_str, absl::StrJoin(*(info.scopes), ","));
203  }
204 
205  auto expiration = now + GoogleOAuthAccessTokenLifetime();
206  // As much as possible, do the time arithmetic using the std::chrono types.
207  // Convert to an integer only when we are dealing with timestamps since the
208  // epoch. Note that we cannot use `time_t` directly because that might be a
209  // floating point.
210  auto const now_from_epoch =
211  static_cast<std::intmax_t>(std::chrono::system_clock::to_time_t(now));
212  auto const expiration_from_epoch = static_cast<std::intmax_t>(
213  std::chrono::system_clock::to_time_t(expiration));
214  nlohmann::json assertion_payload = {
215  {"iss", info.client_email},
216  {"scope", scope_str},
217  {"aud", info.token_uri},
218  {"iat", now_from_epoch},
219  // Resulting access token should be expire after one hour.
220  {"exp", expiration_from_epoch}};
221  if (info.subject) {
222  assertion_payload["sub"] = *(info.subject);
223  }
224 
225  // Note: we don't move here as it would prevent copy elision.
226  return std::make_pair(assertion_header.dump(), assertion_payload.dump());
227 }

236 std::string CreateServiceAccountRefreshPayload(
237  ServiceAccountCredentialsInfo const& info, std::string const& grant_type,
238  std::chrono::system_clock::time_point now) {
239  auto assertion_components = AssertionComponentsFromInfo(info, now);
240  std::string payload = grant_type;
241  payload += "&assertion=";
242  payload += MakeJWTAssertion(assertion_components.first,
243  assertion_components.second, info.private_key);

247 StatusOr<RefreshingCredentialsWrapper::TemporaryToken>
248 ParseServiceAccountRefreshResponse(
249  storage::internal::HttpResponse const& response,
250  std::chrono::system_clock::time_point now) {
251  auto access_token = nlohmann::json::parse(response.payload, nullptr, false);
252  if (access_token.is_discarded() || access_token.count("access_token") == 0 or
253  access_token.count("expires_in") == 0 or
254  access_token.count("token_type") == 0) {
255  auto payload =
256  response.payload +
257  "Could not find all required fields in response (access_token,"
258  " expires_in, token_type).";
259  return AsStatus(storage::internal::HttpResponse{response.status_code,
260  payload, response.headers});
261  }
262  // Response should have the attributes "access_token", "expires_in", and
263  // "token_type".
264  std::string header =
265  "Authorization: " + access_token.value("token_type", "") + " " +
266  access_token.value("access_token", "");
267  auto expires_in =
268  std::chrono::seconds(access_token.value("expires_in", int(0)));
269  auto new_expiration = now + expires_in;
270 
271  return RefreshingCredentialsWrapper::TemporaryToken{std::move(header),
272  new_expiration};
273 }
