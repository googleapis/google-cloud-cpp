561  google::longrunning::Operation operation) {
562  promise<StatusOr<gcsa::Database>> pr;
563  auto f = pr.get_future();
564 
565  // TODO(#4038) - use background_threads_->cq() to run this loop.
566  std::thread t(
567  [](std::shared_ptr<spanner_internal::DatabaseAdminStub> stub,
568  google::longrunning::Operation operation,
569  std::unique_ptr<PollingPolicy> polling_policy,
570  google::cloud::promise<StatusOr<gcsa::Database>> promise,
571  char const* location) mutable {
572  auto result =
573  PollingLoop<PollingLoopResponseExtractor<gcsa::Database>>(
574  std::move(polling_policy),
575  [stub](
576  grpc::ClientContext& context,
577  google::longrunning::GetOperationRequest const& request) {
578  return stub->GetOperation(context, request);
579  },
580  std::move(operation), location);
581 
582  // Drop our reference to stub; ideally we'd have std::moved into the
583  // lambda. Doing this also prevents a false leak from being reported
584  // when using googlemock.
585  stub.reset();
586  promise.set_value(std::move(result));
587  },
588  stub_, std::move(operation), polling_policy_prototype_->clone(),
589  std::move(pr), __func__);
590  t.detach();
591 
592  return f;
593  }
