82 StatusOr<google::iam::v1::Policy> DatabaseAdminClient::SetIamPolicy(
83  Database const& db, IamUpdater const& updater) {
84  auto const rerun_maximum_duration = std::chrono::minutes(15);
85  auto default_rerun_policy =
86  LimitedTimeTransactionRerunPolicy(rerun_maximum_duration).clone();
87 
88  auto const backoff_initial_delay = std::chrono::milliseconds(1000);
89  auto const backoff_maximum_delay = std::chrono::minutes(5);
90  auto const backoff_scaling = 2.0;
91  auto default_backoff_policy =
92  ExponentialBackoffPolicy(backoff_initial_delay, backoff_maximum_delay,
93  backoff_scaling)
94  .clone();
95 
96  return SetIamPolicy(db, updater, std::move(default_rerun_policy),
97  std::move(default_backoff_policy));
98 }

100 StatusOr<google::iam::v1::Policy> DatabaseAdminClient::SetIamPolicy(
101  Database const& db, IamUpdater const& updater,
102  std::unique_ptr<TransactionRerunPolicy> rerun_policy,
103  std::unique_ptr<BackoffPolicy> backoff_policy) {
104  using RerunnablePolicy = spanner_internal::SafeTransactionRerun;
105 
106  Status last_status;
107  do {
108  auto current_policy = GetIamPolicy(db);
109  if (!current_policy) {
110  last_status = std::move(current_policy).status();
111  } else {
112  auto etag = current_policy->etag();
113  auto desired = updater(*current_policy);
114  if (!desired.has_value()) {
115  return current_policy;
116  }
117  desired->set_etag(std::move(etag));
118  auto result = SetIamPolicy(db, *std::move(desired));
119  if (RerunnablePolicy::IsOk(result.status())) {
120  return result;
121  }
122  last_status = std::move(result).status();
123  }
124  if (!rerun_policy->OnFailure(last_status)) break;
125  std::this_thread::sleep_for(backoff_policy->OnCompletion());
126  } while (!rerun_policy->IsExhausted());
127  return last_status;
128 }
