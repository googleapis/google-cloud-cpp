102 template <typename Op>
103 class WriteMutationBuilder {
104  public:
105  WriteMutationBuilder(std::string table_name,
106  std::vector<std::string> column_names) {
107  auto& field = Op::mutable_field(m_.proto());
108  field.set_table(std::move(table_name));
109  field.mutable_columns()->Reserve(static_cast<int>(column_names.size()));
110  for (auto& name : column_names) {
111  field.add_columns(std::move(name));
112  }
113  }
114 
115  Mutation Build() const& { return m_; }
116  Mutation&& Build() && { return std::move(m_); }
117 
118  WriteMutationBuilder& AddRow(std::vector<Value> values) & {
119  auto& lv = *Op::mutable_field(m_.proto()).add_values();
120  for (auto& v : values) {
121  std::tie(std::ignore, *lv.add_values()) = internal::ToProto(std::move(v));
122  }
123  return *this;
124  }
125 
126  WriteMutationBuilder&& AddRow(std::vector<Value> values) && {
127  return std::move(AddRow(std::move(values)));
128  }
129 
130  template <typename... Ts>
131  WriteMutationBuilder& EmplaceRow(Ts&&... values) & {
132  return AddRow({Value(std::forward<Ts>(values))...});
133  }
134 
135  template <typename... Ts>
136  WriteMutationBuilder&& EmplaceRow(Ts&&... values) && {
137  return std::move(EmplaceRow(std::forward<Ts>(values)...));
138  }
139 
140  private:
141  Mutation m_;
142 };
