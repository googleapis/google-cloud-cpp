33 // Compares two sets of Type and Value protos for equality. This method calls
34 // itself recursively to compare subtypes and subvalues.
35 bool Equal(google::spanner::v1::Type const& pt1, // NOLINT(misc-no-recursion)
36  google::protobuf::Value const& pv1,
37  google::spanner::v1::Type const& pt2,
38  google::protobuf::Value const& pv2) {
39  if (pt1.code() != pt2.code()) return false;
40  if (pv1.kind_case() != pv2.kind_case()) return false;
41  switch (pt1.code()) {
42  case google::spanner::v1::TypeCode::BOOL:
43  return pv1.bool_value() == pv2.bool_value();
44  case google::spanner::v1::TypeCode::INT64:
45  return pv1.string_value() == pv2.string_value();
46  case google::spanner::v1::TypeCode::FLOAT64:
47  // NaN should always compare not equal, even to itself.
48  if (pv1.string_value() == "NaN" || pv2.string_value() == "NaN") {
49  return false;
50  }
51  return pv1.string_value() == pv2.string_value() &&
52  pv1.number_value() == pv2.number_value();
53  case google::spanner::v1::TypeCode::STRING:
54  case google::spanner::v1::TypeCode::BYTES:
55  case google::spanner::v1::TypeCode::DATE:
56  case google::spanner::v1::TypeCode::TIMESTAMP:
57  case google::spanner::v1::TypeCode::NUMERIC:
58  return pv1.string_value() == pv2.string_value();
59  case google::spanner::v1::TypeCode::ARRAY: {
60  auto const& etype1 = pt1.array_element_type();
61  auto const& etype2 = pt2.array_element_type();
62  if (etype1.code() != etype2.code()) return false;
63  auto const& v1 = pv1.list_value().values();
64  auto const& v2 = pv2.list_value().values();
65  if (v1.size() != v2.size()) return false;
66  for (int i = 0; i < v1.size(); ++i) {
67  if (!Equal(etype1, v1.Get(i), etype1, v2.Get(i))) {
68  return false;
69  }
70  }
71  return true;
72  }
73  case google::spanner::v1::TypeCode::STRUCT: {
74  auto const& fields1 = pt1.struct_type().fields();
75  auto const& fields2 = pt2.struct_type().fields();
76  if (fields1.size() != fields2.size()) return false;
77  auto const& v1 = pv1.list_value().values();
78  auto const& v2 = pv2.list_value().values();
79  if (fields1.size() != v1.size() || v1.size() != v2.size()) return false;
80  for (int i = 0; i < fields1.size(); ++i) {
81  auto const& f1 = fields1.Get(i);
82  auto const& f2 = fields2.Get(i);
83  if (f1.name() != f2.name()) return false;
84  if (!Equal(f1.type(), v1.Get(i), f2.type(), v2.Get(i))) {
85  return false;
86  }
87  }
88  return true;
89  }
90  default:
91  return true;
92  }

106 // An enum to tell StreamHelper() whether a value is being printed as a scalar
107 // or as part of an aggregate type (i.e., a vector or tuple). Some types may
108 // format themselves differently in each case.
109 enum class StreamMode { kScalar, kAggregate };
110 
111 std::ostream& StreamHelper(std::ostream& os, // NOLINT(misc-no-recursion)
112  google::protobuf::Value const& v,
113  google::spanner::v1::Type const& t,
114  StreamMode mode) {
115  if (v.kind_case() == google::protobuf::Value::kNullValue) {
116  return os << "NULL";
117  }
118 
119  switch (t.code()) {
120  case google::spanner::v1::TypeCode::BOOL:
121  return os << v.bool_value();
122 
123  case google::spanner::v1::TypeCode::INT64:
124  return os
125  << spanner_internal::FromProto(t, v).get<std::int64_t>().value();
126 
127  case google::spanner::v1::TypeCode::FLOAT64:
128  return os << spanner_internal::FromProto(t, v).get<double>().value();
129 
130  case google::spanner::v1::TypeCode::STRING:
131  switch (mode) {
132  case StreamMode::kScalar:
133  return os << v.string_value();
134  case StreamMode::kAggregate:
135  os << '"';
136  EscapeQuotes(os, v.string_value());
137  return os << '"';
138  }
139  return os; // Unreachable, but quiets warning.
140 
141  case google::spanner::v1::TypeCode::BYTES:
142  return os << spanner_internal::BytesFromBase64(v.string_value()).value();
143 
144  case google::spanner::v1::TypeCode::TIMESTAMP:
145  case google::spanner::v1::TypeCode::NUMERIC:
146  return os << v.string_value();
147 
148  case google::spanner::v1::TypeCode::DATE:
149  return os
150  << spanner_internal::FromProto(t, v).get<absl::CivilDay>().value();
151 
152  case google::spanner::v1::TypeCode::ARRAY: {
153  const char* delimiter = "";
154  os << '[';
155  for (auto const& e : v.list_value().values()) {
156  os << delimiter;
157  StreamHelper(os, e, t.array_element_type(), StreamMode::kAggregate);
158  delimiter = ", ";
159  }
160  return os << ']';
161  }
162 
163  case google::spanner::v1::TypeCode::STRUCT: {
164  const char* delimiter = "";
165  os << '(';
166  for (int i = 0; i < v.list_value().values_size(); ++i) {
167  os << delimiter;
168  if (!t.struct_type().fields(i).name().empty()) {
169  os << '"';
170  EscapeQuotes(os, t.struct_type().fields(i).name());
171  os << '"' << ": ";
172  }
173  StreamHelper(os, v.list_value().values(i),
174  t.struct_type().fields(i).type(), StreamMode::kAggregate);
175  delimiter = ", ";
176  }
177  return os << ')';
178  }
179 
180  default:
181  return os << "Error: unknown value type code " << t.code();
182  }
183  return os;
184 }
