263  future<StatusOr<gcsa::Instance>> AwaitCreateOrUpdateInstance(
264  google::longrunning::Operation operation) {
265  promise<StatusOr<gcsa::Instance>> pr;
266  auto f = pr.get_future();
267 
268  // TODO(#127) - use the (implicit) completion queue to run this loop.
269  std::thread t(
270  [](std::shared_ptr<internal::InstanceAdminStub> stub,
271  google::longrunning::Operation operation,
272  std::unique_ptr<PollingPolicy> polling_policy,
273  google::cloud::promise<StatusOr<gcsa::Instance>> promise,
274  char const* location) mutable {
275  auto result = internal::PollingLoop<
276  internal::PollingLoopResponseExtractor<gcsa::Instance>>(
277  std::move(polling_policy),
278  [stub](grpc::ClientContext& context,
279  google::longrunning::GetOperationRequest const& request) {
280  return stub->GetOperation(context, request);
281  },
282  std::move(operation), location);
283 
284  // Drop our reference to stub; ideally we'd have std::moved into the
285  // lambda. Doing this also prevents a false leak from being reported
286  // when using googlemock.
287  stub.reset();
288  promise.set_value(std::move(result));
289  },
290  stub_, std::move(operation), polling_policy_prototype_->clone(),
291  std::move(pr), __func__);
292  t.detach();
293 
294  return f;
295  }
