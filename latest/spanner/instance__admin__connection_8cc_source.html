241  future<StatusOr<gcsa::Instance>> AwaitCreateOrUpdateInstance(
242  google::longrunning::Operation operation) {
243  promise<StatusOr<gcsa::Instance>> pr;
244  auto f = pr.get_future();
245 
246  // TODO(#4038) - use background_threads_->cq() to run this loop.
247  std::thread t(
248  [](std::shared_ptr<spanner_internal::InstanceAdminStub> stub,
249  google::longrunning::Operation operation,
250  std::unique_ptr<PollingPolicy> polling_policy,
251  google::cloud::promise<StatusOr<gcsa::Instance>> promise,
252  char const* location) mutable {
253  auto result = google::cloud::internal::PollingLoop<
254  google::cloud::internal::PollingLoopResponseExtractor<
255  gcsa::Instance>>(
256  std::move(polling_policy),
257  [stub](grpc::ClientContext& context,
258  google::longrunning::GetOperationRequest const& request) {
259  return stub->GetOperation(context, request);
260  },
261  std::move(operation), location);
262 
263  // Drop our reference to stub; ideally we'd have std::moved into the
264  // lambda. Doing this also prevents a false leak from being reported
265  // when using googlemock.
266  stub.reset();
267  promise.set_value(std::move(result));
268  },
269  stub_, std::move(operation), polling_policy_prototype_->clone(),
270  std::move(pr), __func__);
271  t.detach();
272 
273  return f;
274  }
