188 StatusOr<CommitResult> Client::Commit(
189  std::function<StatusOr<Mutations>(Transaction)> const& mutator,
190  std::unique_ptr<TransactionRerunPolicy> rerun_policy,
191  std::unique_ptr<BackoffPolicy> backoff_policy) {
192  // The status-code discriminator of TransactionRerunPolicy.
193  using RerunnablePolicy = spanner_internal::SafeTransactionRerun;
194 
195  Transaction txn = MakeReadWriteTransaction();
196  for (int rerun = 0;; ++rerun) {
197  StatusOr<Mutations> mutations;
198 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
199  try {
200 #endif
201  mutations = mutator(txn);
202 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
203  } catch (RuntimeStatusError const& error) {
204  // Treat this like mutator() returned a bad Status.
205  Status status = error.status();
206  if (status.ok()) {
207  status = Status(StatusCode::kUnknown, "OK Status thrown from mutator");
208  }
209  mutations = status;
210  } catch (...) {
211  auto rb_status = Rollback(txn);
212  if (!RerunnablePolicy::IsOk(rb_status)) {
213  GCP_LOG(WARNING) << "Rollback() failure in Client::Commit(): "
214  << rb_status.message();
215  }
216  throw;
217  }
218 #endif
219  auto status = mutations.status();
220  if (RerunnablePolicy::IsOk(status)) {
221  auto result = Commit(txn, *mutations);
222  status = result.status();
223  if (!RerunnablePolicy::IsTransientFailure(status)) {
224  return result;
225  }
226  } else {
227  if (!RerunnablePolicy::IsTransientFailure(status)) {
228  auto rb_status = Rollback(txn);
229  if (!RerunnablePolicy::IsOk(rb_status)) {
230  GCP_LOG(WARNING) << "Rollback() failure in Client::Commit(): "
231  << rb_status.message();
232  }
233  return status;
234  }
235  }
236  // A transient failure (e.g., kAborted), so consider rerunning.
237  if (!rerun_policy->OnFailure(status)) {
238  return status; // reruns exhausted
239  }
240  if (spanner_internal::IsSessionNotFound(status)) {
241  // Marks the session bad and creates a new Transaction for the next loop.
242  spanner_internal::Visit(
243  txn, [](spanner_internal::SessionHolder& s,
244  StatusOr<google::spanner::v1::TransactionSelector> const&,
245  std::int64_t) {
246  if (s) s->set_bad();
247  return true;
248  });
249  txn = MakeReadWriteTransaction();
250  } else {
251  // Create a new transaction for the next loop, but reuse the session
252  // so that we have a slightly better chance of avoiding another abort.
253  txn = MakeReadWriteTransaction(txn);
254  }
255  std::this_thread::sleep_for(backoff_policy->OnCompletion());
256  }
257 }

259 StatusOr<CommitResult> Client::Commit(
260  std::function<StatusOr<Mutations>(Transaction)> const& mutator) {
261  auto const rerun_maximum_duration = std::chrono::minutes(10);
262  auto default_commit_rerun_policy =
263  LimitedTimeTransactionRerunPolicy(rerun_maximum_duration).clone();
264 
265  auto const backoff_initial_delay = std::chrono::milliseconds(100);
266  auto const backoff_maximum_delay = std::chrono::minutes(5);
267  auto const backoff_scaling = 2.0;
268  auto default_commit_backoff_policy =
269  ExponentialBackoffPolicy(backoff_initial_delay, backoff_maximum_delay,
270  backoff_scaling)
271  .clone();
272 
273  return Commit(mutator, std::move(default_commit_rerun_policy),
274  std::move(default_commit_backoff_policy));
275 }

295 // Returns a QueryOptions struct that has each field set according to the
296 // hierarchy that options specified as to the function call (i.e., `preferred`)
297 // are preferred, followed by options set at the Client level, followed by an
298 // environment variable. If none are set, the field's optional will be unset
299 // and nothing will be included in the proto sent to Spanner, in which case,
300 // the Database default will be used.
301 QueryOptions Client::OverlayQueryOptions(QueryOptions const& preferred) {
302  // GetEnv() is not super fast, so we look it up once and cache it.
303  static auto const* const kOptimizerVersionEnvValue =
304  new auto(google::cloud::internal::GetEnv("SPANNER_OPTIMIZER_VERSION"));
305 
306  QueryOptions const& fallback = opts_.query_options();
307  QueryOptions opts;
308 
309  // Choose the `optimizer_version` option.
310  if (preferred.optimizer_version().has_value()) {
311  opts.set_optimizer_version(preferred.optimizer_version());
312  } else if (fallback.optimizer_version().has_value()) {
313  opts.set_optimizer_version(fallback.optimizer_version());
314  } else if (kOptimizerVersionEnvValue->has_value()) {
315  opts.set_optimizer_version(*kOptimizerVersionEnvValue);
316  }
317 
318  return opts;
319 }

329 std::shared_ptr<Connection> MakeConnection(
330  Database const& db, ConnectionOptions const& connection_options,
331  SessionPoolOptions session_pool_options,
332  std::unique_ptr<RetryPolicy> retry_policy,
333  std::unique_ptr<BackoffPolicy> backoff_policy) {
334  std::vector<std::shared_ptr<spanner_internal::SpannerStub>> stubs;
335  int num_channels = std::max(connection_options.num_channels(), 1);
336  stubs.reserve(num_channels);
337  for (int channel_id = 0; channel_id < num_channels; ++channel_id) {
338  stubs.push_back(spanner_internal::CreateDefaultSpannerStub(
339  db, connection_options, channel_id));
340  }
341  return spanner_internal::MakeConnection(
342  db, std::move(stubs), connection_options, std::move(session_pool_options),
343  std::move(retry_policy), std::move(backoff_policy));
344 }
