191 StatusOr<CommitResult> Client::Commit(
192  std::function<StatusOr<Mutations>(Transaction)> const& mutator,
193  std::unique_ptr<TransactionRerunPolicy> rerun_policy,
194  std::unique_ptr<BackoffPolicy> backoff_policy,
195  CommitOptions const& options) {
196  // The status-code discriminator of TransactionRerunPolicy.
197  using RerunnablePolicy = spanner_internal::SafeTransactionRerun;
198 
199  Transaction txn = MakeReadWriteTransaction();
200  for (int rerun = 0;; ++rerun) {
201  StatusOr<Mutations> mutations;
202 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
203  try {
204 #endif
205  mutations = mutator(txn);
206 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
207  } catch (RuntimeStatusError const& error) {
208  // Treat this like mutator() returned a bad Status.
209  Status status = error.status();
210  if (status.ok()) {
211  status = Status(StatusCode::kUnknown, "OK Status thrown from mutator");
212  }
213  mutations = status;
214  } catch (...) {
215  auto rb_status = Rollback(txn);
216  if (!RerunnablePolicy::IsOk(rb_status)) {
217  GCP_LOG(WARNING) << "Rollback() failure in Client::Commit(): "
218  << rb_status.message();
219  }
220  throw;
221  }
222 #endif
223  auto status = mutations.status();
224  if (RerunnablePolicy::IsOk(status)) {
225  auto result = Commit(txn, *mutations, options);
226  status = result.status();
227  if (!RerunnablePolicy::IsTransientFailure(status)) {
228  return result;
229  }
230  } else {
231  if (!RerunnablePolicy::IsTransientFailure(status)) {
232  auto rb_status = Rollback(txn);
233  if (!RerunnablePolicy::IsOk(rb_status)) {
234  GCP_LOG(WARNING) << "Rollback() failure in Client::Commit(): "
235  << rb_status.message();
236  }
237  return status;
238  }
239  }
240  // A transient failure (e.g., kAborted), so consider rerunning.
241  if (!rerun_policy->OnFailure(status)) {
242  return status; // reruns exhausted
243  }
244  if (spanner_internal::IsSessionNotFound(status)) {
245  // Marks the session bad and creates a new Transaction for the next loop.
246  spanner_internal::Visit(
247  txn, [](spanner_internal::SessionHolder& s,
248  StatusOr<google::spanner::v1::TransactionSelector> const&,
249  std::int64_t) {
250  if (s) s->set_bad();
251  return true;
252  });
253  txn = MakeReadWriteTransaction();
254  } else {
255  // Create a new transaction for the next loop, but reuse the session
256  // so that we have a slightly better chance of avoiding another abort.
257  txn = MakeReadWriteTransaction(txn);
258  }
259  std::this_thread::sleep_for(backoff_policy->OnCompletion());
260  }
261 }

263 StatusOr<CommitResult> Client::Commit(
264  std::function<StatusOr<Mutations>(Transaction)> const& mutator,
265  CommitOptions const& options) {
266  auto const rerun_maximum_duration = std::chrono::minutes(10);
267  auto default_commit_rerun_policy =
268  LimitedTimeTransactionRerunPolicy(rerun_maximum_duration).clone();
269 
270  auto const backoff_initial_delay = std::chrono::milliseconds(100);
271  auto const backoff_maximum_delay = std::chrono::minutes(5);
272  auto const backoff_scaling = 2.0;
273  auto default_commit_backoff_policy =
274  ExponentialBackoffPolicy(backoff_initial_delay, backoff_maximum_delay,
275  backoff_scaling)
276  .clone();
277 
278  return Commit(mutator, std::move(default_commit_rerun_policy),
279  std::move(default_commit_backoff_policy), options);
280 }

304 // Returns a QueryOptions struct that has each field set according to the
305 // hierarchy that options specified as to the function call (i.e., `preferred`)
306 // are preferred, followed by options set at the Client level, followed by an
307 // environment variable. If none are set, the field's optional will be unset
308 // and nothing will be included in the proto sent to Spanner, in which case,
309 // the Database default will be used.
310 QueryOptions Client::OverlayQueryOptions(QueryOptions const& preferred) {
311  // GetEnv() is not super fast, so we look it up once and cache it.
312  static auto const* const kOptimizerVersionEnvValue =
313  new auto(google::cloud::internal::GetEnv("SPANNER_OPTIMIZER_VERSION"));
314 
315  QueryOptions const& fallback = opts_.query_options();
316  QueryOptions opts;
317 
318  // Choose the `optimizer_version` option.
319  if (preferred.optimizer_version().has_value()) {
320  opts.set_optimizer_version(preferred.optimizer_version());
321  } else if (fallback.optimizer_version().has_value()) {
322  opts.set_optimizer_version(fallback.optimizer_version());
323  } else if (kOptimizerVersionEnvValue->has_value()) {
324  opts.set_optimizer_version(*kOptimizerVersionEnvValue);
325  }
326 
327  return opts;
328 }

330 std::shared_ptr<spanner::Connection> MakeConnection(spanner::Database const& db,
331  Options opts) {
332  internal::CheckExpectedOptions<CommonOptionList, GrpcOptionList,
333  SessionPoolOptionList,
334  SpannerPolicyOptionList>(opts, __func__);
335  opts = spanner_internal::DefaultOptions(std::move(opts));
336  std::vector<std::shared_ptr<spanner_internal::SpannerStub>> stubs;
337  int num_channels = opts.get<GrpcNumChannelsOption>();
338  stubs.reserve(num_channels);
339  for (int channel_id = 0; channel_id < num_channels; ++channel_id) {
340  stubs.push_back(
341  spanner_internal::CreateDefaultSpannerStub(db, opts, channel_id));
342  }
343  return std::make_shared<spanner_internal::ConnectionImpl>(
344  std::move(db), std::move(stubs), opts);
345 }
